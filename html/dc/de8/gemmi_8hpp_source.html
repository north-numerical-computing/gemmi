154 void computeNormalisationVectors() {
155 // Compute normalisation vector.
156 const size_t iStride = this->iStride();
157 const size_t jStride = this->jStride();
158 for (size_t i = 0; i < this->otherDimension(); i++) {
159 this->powersVector[i] = 0.0;
160 for (size_t j = 0; j < this->innerProductDimension(); j++) {
161 const size_t index = i * iStride + j * jStride;
162 this->powersVector[i] = std::max(this->powersVector[i],
163 std::abs(this->matrix[index]));
164 }
165 // Compute the smallest power of 2 that is strictly greater than the
166 // maximum value in the row/column.
167 // NOTE 1: This is not the technique used in uoi24.
168 // NOTE 2: I use exponents instead of powers of 2, as I need the former
169 // to shift correctly.
170 frexp(this->powersVector[i], this->scalingExponents.data() + i);
171 this->powersVector[i] = std::ldexp(1.0, this->scalingExponents[i]);
172 }
173 }

187 void computeSplitsWithRoundToNearest() {
188 this->splitType = splittingStrategy::roundToNearest;
189 auto iStride = this->iStride();
190 auto jStride = this->jStride();
191 auto localMatrix = this->matrix;
192 for (size_t slice = 0; slice < numSplits; slice++) {
193 for (size_t i = 0; i < this->otherDimension(); i++) {
194 fp_t sigma = ldexp(0.75, computeNumFracBits<fp_t>() - this->bitsPerSlice * slice + 1 - this->bitsPerSlice) * powersVector[i];
195 for (size_t j = 0; j < this->innerProductDimension(); j++) {
196 auto value = (localMatrix[i * iStride + j * jStride] + sigma);
197 value -= sigma;
198 localMatrix[i * iStride + j * jStride] -= value;
199 value = value / powersVector[i] * ldexp(1.0, this->bitsPerSlice * slice + this->bitsPerSlice - 1);
200 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = value;
201 }
202 }
203 }

217 void computeSplitsWithBitMasking() {
218 this->splitType = splittingStrategy::bitMasking;
219 // Compute splits one row/column at a time.
220 auto numExpBits = computeNumExpBits<fp_t>();
221 auto numFracBits = computeNumFracBits<fp_t>();
222 auto iStride = this->iStride();
223 auto jStride = this->jStride();
224 std::vector<uint_t> tmp (this->innerProductDimension());
225 std::vector<bool> sign (this->innerProductDimension());
226 for (size_t i = 0; i < this->otherDimension(); i++) {
227 // Get binary representation of normalised row/column.
228 for (size_t j = 0; j < this->innerProductDimension(); j++) {
229 const size_t index = i * iStride + j * jStride;
230 fp_t value = this->matrix[index]; // powersVector[i];
231 tmp[j] = std::bit_cast<uint_t>(value); // To bitstring.
232 sign[j] = std::signbit(value); // Extract sign.
233 uint_t bitmask = (~((uint_t)(0))) >> (numExpBits + 1);
234 tmp[j] = tmp[j] & bitmask; // Remove exponent and sign.
235 // Restore implicit bit for normal numbers.
236 // NOTE: NaNs and infs are currently not supported.
237 if (std::fpclassify(value) == FP_NORMAL)
238 tmp[j] |= ((uint_t)1 << (numFracBits - 1));
239 }
240
241 // Create bitmask.
242 const uint_t smallBitmask = (1 << this->bitsPerSlice) - 1;
243 // Perform the split.
244 for (size_t j = 0; j < this->innerProductDimension(); j++) {
245 int16_t shiftCounter = numFracBits - this->bitsPerSlice;
246 int currentExponent;
247 frexp(this->matrix[i * iStride + j * jStride], &currentExponent);
248 int16_t exponentDifference = scalingExponents[i] - currentExponent;
249 for (size_t slice = 0; slice < numSplits; slice++) {
250 if (exponentDifference > (signed)this->bitsPerSlice) {
251 exponentDifference -= this->bitsPerSlice;
252 } else {
253 shiftCounter += exponentDifference;
254 exponentDifference = 0;
255 uint_t bitmask = shiftCounter > 0 ?
256 smallBitmask << shiftCounter :
257 smallBitmask >> -shiftCounter;
258 uint_t currentSlice = tmp[j] & bitmask;
259 uint_t currentSplit = shiftCounter > 0 ?
260 currentSlice >> shiftCounter :
261 currentSlice << -shiftCounter;
262 splitint_t value = (splitint_t)(currentSplit) * (sign[j] ? -1 : 1);
263 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = value;
264 shiftCounter -= this->bitsPerSlice;
265 }
266 }
267 }
268 }
269 }

310fp_t computeScalingConstantForSplittingStrategy(const MatrixSplit<splitint_t, fp_t> &A,
311 const MatrixSplit<splitint_t, fp_t> &B) {
312 // When splitting with round-to-nearest, the first slice has bitsPerSlice - 1 bits,
313 // and we need to account for this when scaling the final result.
314 fp_t scalingConstant = 1.0;
315 scalingConstant *= A.splitType == splittingStrategy::roundToNearest ? 2.0 : 1.0;
316 scalingConstant *= B.splitType == splittingStrategy::roundToNearest ? 2.0 : 1.0;
317 return scalingConstant;
318}

340template <typename splitint_t, typename accumulator_t, typename fp_t>
341std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
342 const MatrixSplit<splitint_t, fp_t> &B,
343 const size_t bitsPerSlice,
344 const size_t numDiagonals) {
345
346 std::vector<fp_t > C (A.m * B.n);
347
348 auto scalingConstant = computeScalingConstantForSplittingStrategy(A, B);
349
350 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
351 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
352 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
353 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
354 std::vector<accumulator_t> accumulator (A.m * B.n, 0.0);
355 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, Aindex, Bindex);
356 for (size_t i = 0; i < A.m; i++) {
357 for (size_t j = 0; j < B.n; j++) {
358 fp_t scaledSum = std::ldexp(accumulator[i + j * A.m], -(Aindex + 1 + Bindex + 1) * bitsPerSlice);
359 fp_t scalingFactor = A.powersVector[i] * B.powersVector[j] * scalingConstant;
360 C[i + j * A.m] += scaledSum * scalingFactor;
361 }
362 }
363 Aindex--;
364 Bindex++;
365 }
366 }
367
368 return C;
369}

439template <typename fp_t, typename splitint_t, typename accumulator_t>
440std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const std::vector<fp_t> &B,
441 const size_t m, const size_t k, const size_t n,
442 const size_t numSplitsA, const size_t numSplitsB,
443 const splittingStrategy splitType = splittingStrategy::roundToNearest,
444 const accumulationStrategy accType = accumulationStrategy::floatingPoint,
445 const multiplicationStrategy multType = multiplicationStrategy::reduced) {
446
447 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
448 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
449 assert(bitsPerInteger <= bitsInAccumulator / 2);
450 const size_t alpha = std::floor((bitsInAccumulator - log2(n)) / 2);
451 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
452
453 auto splitA = MatrixSplit<splitint_t, fp_t>(A, m, k, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
454 auto splitB = MatrixSplit<splitint_t, fp_t>(B, k, n, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
455
456 size_t numDiagonals;
457 switch (multType) {
458 case multiplicationStrategy::reduced:
459 // Products below the main anti-diagonal are ignored.
460 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
461 break;
462 case multiplicationStrategy::full:
463 // All products are computed.
464 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
465 break;
466 default:
467 std::cerr << "Unknown multiplication strategy requested.";
468 exit(1);
469 }
470
471 switch (accType) {
472 case accumulationStrategy::floatingPoint:
473 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, bitsPerSlice, numDiagonals);
474 case accumulationStrategy::integer:
475 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, bitsPerSlice, numDiagonals);
476 default:
477 std::cerr << "Unknown accumulation strategy requested.";
478 exit(1);
479 }
480}
