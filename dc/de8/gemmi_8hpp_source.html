424 void computeNormalisationVectors() {
425 // Compute normalisation vector.
426 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
427 this->powersVector[outer] = 0.0;
428 for (size_t j = 0; j < this->innerDimension(); j++) {
429 this->powersVector[outer] = std::max(this->powersVector[outer],
430 std::abs(this->operand(outer, j)));
431 }
432 // Compute the smallest power of 2 that is strictly greater than the
433 // maximum value in the row/column.
434 // NOTE 1: This is not the technique used in uoi24.
435 // NOTE 2: I use exponents instead of powers of 2, as I need the former
436 // to shift correctly.
437 this->scalingExponents[outer] = getStoredFloatingPointExponent(this->powersVector[outer]);
438 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);
439 }
440 }

447 int computeSliceBitOffset(size_t slice) const {
448 switch (splitType) {
449 case splittingStrategy::truncation:
450 // Slice k -> k * b
451 return static_cast<int>((slice + 1) * bitsPerSlice);
452
453 case splittingStrategy::roundToNearest:
454 // Slice 0 -> b - 1
455 // Slice k -> k * b - 1
456 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
457
458 case splittingStrategy::unsignedEncoding:
459 // Slice 0 -> b - 1
460 // Slice k -> (b - 1) + k * (b + 1)
461 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));

463 // LCOV_EXCL_START
464 default:
465 throw std::logic_error("Unhandled splittingStrategy");
466 // LCOV_EXCL_STOP

477 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
478 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
479 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
480 fp_t value = this->operand(outer, inner);
481 fraction[inner] = std::bit_cast<uint_t>(value);
482 sign[inner] = std::signbit(value);
483 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
484 fraction[inner] = fraction[inner] & bitmask;
485 // Restore implicit bit for normal numbers.
486 // NOTE: NaNs and infs are currently not supported.
487 if (std::fpclassify(value) == FP_NORMAL)
488 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
489 }
490 }

501 void computeSplitsWithTruncation() {
502 this->splitType = splittingStrategy::truncation;
503 // Compute splits one row/column at a time.
504 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
505 auto bitsPerSlice = this->bitsPerSlice;
506 std::vector<uint_t> fraction (this->innerDimension());
507 std::vector<bool> sign (this->innerDimension());
508 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
509 // Get binary representation of significands of normalised row/column.
510 computeFixedPointRepresentationVector(fraction, sign, outer);
511
512 // Create bitmask.
513 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
514
515 // Perform the split.
516 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
517 // NOTE: I could have a special path for 0.
518 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
519 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
520 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
521 for (size_t slice = 0; slice < numSplits; slice++) {
522 if (exponentDifference > (signed)bitsPerSlice) {
523 exponentDifference -= bitsPerSlice;
524 } else {
525 shiftCounter += exponentDifference;
526 exponentDifference = 0;
527 uint_t bitmask = shiftCounter > 0 ?
528 smallBitmask << shiftCounter :
529 smallBitmask >> -shiftCounter;
530 uint_t currentSlice = fraction[inner] & bitmask;
531 uint_t currentSplit = shiftCounter > 0 ?
532 currentSlice >> shiftCounter :
533 currentSlice << -shiftCounter;
534 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
535 this->splitValue(outer, inner, slice) = value;
536 shiftCounter -= bitsPerSlice;
537 }
538 }
539 }
540 }

554 void computeSplitsWithUnsignedEncoding() {
555 this->splitType = splittingStrategy::unsignedEncoding;
556 // Compute splits one row/column at a time.
557 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
558 auto bitsPerSlice = this->bitsPerSlice;
559 std::vector<uint_t> fraction (this->innerDimension());
560 std::vector<bool> sign (this->innerDimension());
561 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
562 // Get binary representation of significands of normalised row/column.
563 computeFixedPointRepresentationVector(fraction, sign, outer);
564
565 // Create bitmasks.
566 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
567 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
568 // is to keep the algorithm consistent with the bitmasking approach
569 // above.
570 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
571 // is not mentioned in sabb26, but it seems necessary to do this to
572 // ensure that the first (signed) slice does not overflow if the full
573 // bitwidth of splitint_t is used and the second slice is negative in
574 // two's complement.
575 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
576 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
577
578 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
579
580 auto matrixIndex = operandIndex(outer, inner);
581
582 // NOTE: I could have a special path for 0.
583 int16_t shiftCounter;
584 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
585 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
586
587 splitint_t value = 0;
588 uint_t remainder = 0;
589
590 // Slice 0.
591 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
592 // Value is too small to contribute explicit bits to slice 0.
593 value = sign[inner] ? -1 : 0;
594 remainder = sign[inner] ?
595 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
596 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
597 fraction[inner];
598 exponentDifference -= (bitsPerSlice - 1);
599 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
600 } else {
601 // Truncation.
602 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
603 exponentDifference = 0;
604 uint_t bitmask = (shiftCounter > 0)
605 ? (smallBitmask << shiftCounter)
606 : (smallBitmask >> -shiftCounter);
607 uint_t currentSlice = fraction[inner] & bitmask;
608 uint_t currentSplit = (shiftCounter > 0)
609 ? (currentSlice >> shiftCounter)
610 : (currentSlice << -shiftCounter);
611
612 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
613
614 // Rounding.
615 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
616 uint_t lowBits = fraction[inner] & lowMask;
617
618 if (lowBits != 0) {
619 value += (sign[inner] ? -1 : 0);
620 }
621 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
622 shiftCounter -= (bitsPerSlice + 1);
623 }
624 this->memory[matrixIndex] = value;
625
626 // Remaining slices.
627 const auto width = bitsPerSlice + 1;
628 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
629 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
630 const auto digitMax = cutoff - 1; // 2^b - 1
631 const auto digitMin = -cutoff; // -2^b
632 for (size_t slice = 1; slice < numSplits; slice++) {
633 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
634 exponentDifference -= (bitsPerSlice + 1);
635 if (sign[inner]) {
636 this->splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
637 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
638 }
639 } else {
640 shiftCounter += exponentDifference;
641 exponentDifference = 0;
642
643 uint_t bitmask = shiftCounter > 0 ?
644 largeBitmask << shiftCounter :
645 largeBitmask >> -shiftCounter;
646
647 uint_t currentSlice = remainder & bitmask;
648 uint_t currentSplit = shiftCounter > 0 ?
649 currentSlice >> shiftCounter :
650 currentSlice << -shiftCounter;
651
652 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
653 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
654 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
655 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
656 currentSplit |= maskToAdd;
657 }
658
659 // Update previous slices if currentSplit is negative in splitint_t.
660 // Carry must propoagate upward until slice 0 if necessary.
661 if (currentSplit > static_cast<uint_t>(digitMax)) {
662 int prevSliceIndex = static_cast<int>(slice) - 1;
663 while (true) {
664 auto newValue = static_cast<wideint_t>(this->splitValue(outer, inner, prevSliceIndex)) + 1;
665 if (newValue <= digitMax) {
666 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
667 break;
668 } else {
669 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
670 prevSliceIndex--;
671 }
672 }
673 }
674
675 // Store the slice as a signed (b+1)-bit value:
676 // [0, 2^b-1] -> unchanged
677 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
678 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
679 if (signedDigit >= cutoff) {
680 signedDigit -= base;
681 }
682
683 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
684 shiftCounter -= (bitsPerSlice + 1);
685 }
686 }
687 }
688 }

700 void computeSplitsWithRoundToNearest() {
701 this->splitType = splittingStrategy::roundToNearest;
702 auto bitsPerSlice = this->bitsPerSlice;
703 auto localMatrix = std::vector<fp_t>(this->matrix.data, this->matrix.data + this->matrix.size());
704 for (size_t slice = 0; slice < numSplits; slice++) {
705 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
706 fp_t sigma = ldexp(0.75,
707 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
708 powersVector[outer];
709 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
710 auto matrixIndex = operandIndex(outer, inner);
711 auto value = (localMatrix[matrixIndex] + sigma);
712 value -= sigma;
713 localMatrix[matrixIndex] -= value;
714 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
715 this->splitValue(outer, inner, slice) = value;
716 }
717 }
718 }
719 }

788template <typename splitint_t, typename accumulator_t, typename fp_t>
789std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const Decomposition<splitint_t, fp_t> &A,
790 const Decomposition<splitint_t, fp_t> &B,
791 const matrixLayout layoutC,
792 const size_t numDiagonals) {
793 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
794 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
795 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
796 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
797 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
798 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
799 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
800 for (size_t row = 0; row < A.rows(); row++) {
801 for (size_t col = 0; col < B.cols(); col++) {
802 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
803 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
804 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
805 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
806 C[index] += scaledSum * scalingFactor;
807 }
808 }
809 Aindex--;
810 Bindex++;
811 }
812 }
813
814 return C;
815}

837template <typename splitint_t, typename accumulator_t, typename fp_t>
838std::vector<fp_t> computeProductsWithIntegerAccumulation(const Decomposition<splitint_t, fp_t> &A,
839 const Decomposition<splitint_t, fp_t> &B,
840 const matrixLayout layoutC,
841 const size_t numDiagonals) {
842
843 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
844 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
845 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
846 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
847
848 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
849
850 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
851 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
852 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
853 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
854 Aindex--;
855 Bindex++;
856 }
857
858 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
859 for (size_t row = 0; row < A.rows(); row++) {
860 for (size_t col = 0; col < B.cols(); col++) {
861 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
862 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
863 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
864 C[index] += scaledSum * scalingFactor;
865 }
866 }
867 }
868
869 return C;
870}

891template <typename fp_t, typename splitint_t, typename accumulator_t>
892std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const matrixLayout layoutA,
893 const std::vector<fp_t> &B, const matrixLayout layoutB,
894 const size_t m, const size_t k, const size_t n,
895 const size_t numSplitsA, const size_t numSplitsB,
896 const matrixLayout layoutC = matrixLayout::columnMajor,
897 const multiterm::splittingStrategy splitType = multiterm::splittingStrategy::roundToNearest,
898 const multiterm::reductionStrategy accType = multiterm::reductionStrategy::floatingPoint,
899 const multiterm::multiplicationStrategy multType = multiterm::multiplicationStrategy::reduced) {
900
901 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
902 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
903 if (bitsPerInteger > bitsInAccumulator / 2) {
904 throw std::invalid_argument("Split integer type is too wide for the chosen accumulator type");
905 }
906 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
907 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
908
909 auto viewA = makeMatrixView(A, m, k, layoutA);
910 auto viewB = makeMatrixView(B, k, n, layoutB);
911
912 auto splitA = multiterm::Decomposition<splitint_t, fp_t>(viewA, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
913 auto splitB = multiterm::Decomposition<splitint_t, fp_t>(viewB, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
914
915 splitA.prepare();
916 splitB.prepare();
917
918 size_t numDiagonals;
919 switch (multType) {
920 case multiterm::multiplicationStrategy::reduced:
921 // Products below the main anti-diagonal are ignored.
922 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
923 break;
924 case multiterm::multiplicationStrategy::full:
925 // All products are computed.
926 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
927 break;
928 // LCOV_EXCL_START
929 default:
930 throw std::logic_error("Unhandled multiplicationStrategy");
931 // LCOV_EXCL_STOP
932 }
933
934 switch (accType) {
935 case multiterm::reductionStrategy::floatingPoint:
936 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
937 case multiterm::reductionStrategy::integer:
938 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
