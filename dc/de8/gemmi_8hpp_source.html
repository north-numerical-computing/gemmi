435 void computeNormalisationVectors() {
436 // Compute normalisation vector.
437 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
438 this->powersVector[outer] = 0.0;
439 for (size_t j = 0; j < this->innerDimension(); j++) {
440 this->powersVector[outer] = std::max(this->powersVector[outer],
441 std::abs(this->operand(outer, j)));
442 }
443 // Compute the smallest power of 2 that is strictly greater than the
444 // maximum value in the row/column.
445 // NOTE 1: This is not the technique used in uoi24.
446 // NOTE 2: I use exponents instead of powers of 2, as I need the former
447 // to shift correctly.
448 this->scalingExponents[outer] = getStoredFloatingPointExponent(this->powersVector[outer]);
449 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);
450 }
451 }

458 int computeSliceBitOffset(size_t slice) const {
459 switch (splitType) {
460 case splittingStrategy::truncation:
461 // Slice k -> k * b
462 return static_cast<int>((slice + 1) * bitsPerSlice);
463
464 case splittingStrategy::roundToNearest:
465 // Slice 0 -> b - 1
466 // Slice k -> k * b - 1
467 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
468
469 case splittingStrategy::unsignedEncoding:
470 // Slice 0 -> b - 1
471 // Slice k -> (b - 1) + k * (b + 1)
472 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));
473
474 // LCOV_EXCL_START
475 default:
476 throw std::logic_error("Unhandled splittingStrategy");
477 // LCOV_EXCL_STOP

488 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
489 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
490 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
491 fp_t value = this->operand(outer, inner);
492 fraction[inner] = std::bit_cast<uint_t>(value);
493 sign[inner] = std::signbit(value);
494 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
495 fraction[inner] = fraction[inner] & bitmask;
496 // Restore implicit bit for normal numbers.
497 // NOTE: NaNs and infs are currently not supported.
498 if (std::fpclassify(value) == FP_NORMAL)
499 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
500 }

512 void computeSplitsWithTruncation() {
513 this->splitType = splittingStrategy::truncation;
514 // Compute splits one row/column at a time.
515 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
516 auto bitsPerSlice = this->bitsPerSlice;
517 std::vector<uint_t> fraction (this->innerDimension());
518 std::vector<bool> sign (this->innerDimension());
519 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
520 // Get binary representation of significands of normalised row/column.
521 computeFixedPointRepresentationVector(fraction, sign, outer);
522
523 // Create bitmask.
524 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
525
526 // Perform the split.
527 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
528 // NOTE: I could have a special path for 0.
529 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
530 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
531 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
532 for (size_t slice = 0; slice < numSplits; slice++) {
533 if (exponentDifference > (signed)bitsPerSlice) {
534 exponentDifference -= bitsPerSlice;
535 } else {
536 shiftCounter += exponentDifference;
537 exponentDifference = 0;
538 uint_t bitmask = shiftCounter > 0 ?
539 smallBitmask << shiftCounter :
540 smallBitmask >> -shiftCounter;
541 uint_t currentSlice = fraction[inner] & bitmask;
542 uint_t currentSplit = shiftCounter > 0 ?
543 currentSlice >> shiftCounter :
544 currentSlice << -shiftCounter;
545 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
546 this->splitValue(outer, inner, slice) = value;
547 shiftCounter -= bitsPerSlice;
548 }
549 }
550 }
551 }

565 void computeSplitsWithUnsignedEncoding() {
566 this->splitType = splittingStrategy::unsignedEncoding;
567 // Compute splits one row/column at a time.
568 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
569 auto bitsPerSlice = this->bitsPerSlice;
570 std::vector<uint_t> fraction (this->innerDimension());
571 std::vector<bool> sign (this->innerDimension());
572 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
573 // Get binary representation of significands of normalised row/column.
574 computeFixedPointRepresentationVector(fraction, sign, outer);
575
576 // Create bitmasks.
577 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
578 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
579 // is to keep the algorithm consistent with the bitmasking approach
580 // above.
581 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
582 // is not mentioned in sabb26, but it seems necessary to do this to
583 // ensure that the first (signed) slice does not overflow if the full
584 // bitwidth of splitint_t is used and the second slice is negative in
585 // two's complement.
586 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
587 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
588
589 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
590
591 auto matrixIndex = operandIndex(outer, inner);
592
593 // NOTE: I could have a special path for 0.
594 int16_t shiftCounter;
595 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
596 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
597
598 splitint_t value = 0;
599 uint_t remainder = 0;
600
601 // Slice 0.
602 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
603 // Value is too small to contribute explicit bits to slice 0.
604 value = sign[inner] ? -1 : 0;
605 remainder = sign[inner] ?
606 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
607 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
608 fraction[inner];
609 exponentDifference -= (bitsPerSlice - 1);
610 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
611 } else {
612 // Truncation.
613 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
614 exponentDifference = 0;
615 uint_t bitmask = (shiftCounter > 0)
616 ? (smallBitmask << shiftCounter)
617 : (smallBitmask >> -shiftCounter);
618 uint_t currentSlice = fraction[inner] & bitmask;
619 uint_t currentSplit = (shiftCounter > 0)
620 ? (currentSlice >> shiftCounter)
621 : (currentSlice << -shiftCounter);
622
623 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
624
625 // Rounding.
626 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
627 uint_t lowBits = fraction[inner] & lowMask;
628
629 if (lowBits != 0) {
630 value += (sign[inner] ? -1 : 0);
631 }
632 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
633 shiftCounter -= (bitsPerSlice + 1);
634 }
635 this->memory[matrixIndex] = value;
636
637 // Remaining slices.
638 const auto width = bitsPerSlice + 1;
639 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
640 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
641 const auto digitMax = cutoff - 1; // 2^b - 1
642 const auto digitMin = -cutoff; // -2^b
643 for (size_t slice = 1; slice < numSplits; slice++) {
644 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
645 exponentDifference -= (bitsPerSlice + 1);
646 if (sign[inner]) {
647 this->splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
648 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
649 }
650 } else {
651 shiftCounter += exponentDifference;
652 exponentDifference = 0;
653
654 uint_t bitmask = shiftCounter > 0 ?
655 largeBitmask << shiftCounter :
656 largeBitmask >> -shiftCounter;
657
658 uint_t currentSlice = remainder & bitmask;
659 uint_t currentSplit = shiftCounter > 0 ?
660 currentSlice >> shiftCounter :
661 currentSlice << -shiftCounter;
662
663 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
664 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
665 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
666 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
667 currentSplit |= maskToAdd;
668 }
669
670 // Update previous slices if currentSplit is negative in splitint_t.
671 // Carry must propoagate upward until slice 0 if necessary.
672 if (currentSplit > static_cast<uint_t>(digitMax)) {
673 int prevSliceIndex = static_cast<int>(slice) - 1;
674 while (true) {
675 auto newValue = static_cast<wideint_t>(this->splitValue(outer, inner, prevSliceIndex)) + 1;
676 if (newValue <= digitMax) {
677 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
678 break;
679 } else {
680 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
681 prevSliceIndex--;
682 }
683 }
684 }
685
686 // Store the slice as a signed (b+1)-bit value:
687 // [0, 2^b-1] -> unchanged
688 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
689 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
690 if (signedDigit >= cutoff) {
691 signedDigit -= base;
692 }
693
694 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
695 shiftCounter -= (bitsPerSlice + 1);
696 }
697 }
698 }
699 }
700 }

711 void computeSplitsWithRoundToNearest() {
712 this->splitType = splittingStrategy::roundToNearest;
713 auto bitsPerSlice = this->bitsPerSlice;
714 auto localMatrix = std::vector<fp_t>(this->matrix.data, this->matrix.data + this->matrix.size());
715 for (size_t slice = 0; slice < numSplits; slice++) {
716 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
717 fp_t sigma = ldexp(0.75,
718 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
719 powersVector[outer];
720 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
721 auto matrixIndex = operandIndex(outer, inner);
722 auto value = (localMatrix[matrixIndex] + sigma);
723 value -= sigma;
724 localMatrix[matrixIndex] -= value;
725 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
726 this->splitValue(outer, inner, slice) = value;
727 }
728 }
729 }
730 }

799template <typename splitint_t, typename accumulator_t, typename fp_t>
800std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const Decomposition<splitint_t, fp_t> &A,
801 const Decomposition<splitint_t, fp_t> &B,
802 const matrixLayout layoutC,
803 const size_t numDiagonals) {
804 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
805 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
806 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
807 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
808 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
809 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
810 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
811 for (size_t row = 0; row < A.rows(); row++) {
812 for (size_t col = 0; col < B.cols(); col++) {
813 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
814 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
815 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
816 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
817 C[index] += scaledSum * scalingFactor;
818 }
819 }
820 Aindex--;
821 Bindex++;
822 }
823 }
824
825 return C;
826}

848template <typename splitint_t, typename accumulator_t, typename fp_t>
849std::vector<fp_t> computeProductsWithIntegerAccumulation(const Decomposition<splitint_t, fp_t> &A,
850 const Decomposition<splitint_t, fp_t> &B,
851 const matrixLayout layoutC,
852 const size_t numDiagonals) {
853
854 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
855 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
856 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
857 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
858
859 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
860
861 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
862 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
863 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
864 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
865 Aindex--;
866 Bindex++;
867 }
868
869 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
870 for (size_t row = 0; row < A.rows(); row++) {
871 for (size_t col = 0; col < B.cols(); col++) {
872 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
873 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
874 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
875 C[index] += scaledSum * scalingFactor;
876 }
877 }
878 }
879
880 return C;
881}

898template <typename fp_t, typename splitint_t, typename accumulator_t>
899std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const matrixLayout layoutA,
900 const std::vector<fp_t> &B, const matrixLayout layoutB,
901 const size_t m, const size_t k, const size_t n,
902 const matrixLayout layoutC,
903 const multiterm::config &config) {
904
905 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
906 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
907 if (bitsPerInteger > bitsInAccumulator / 2) {
908 throw std::invalid_argument("Split integer type is too wide for the chosen accumulator type");
909 }
910 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
911 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
912
913 auto viewA = makeMatrixView(A, m, k, layoutA);
914 auto viewB = makeMatrixView(B, k, n, layoutB);
915
916 auto splitA = multiterm::Decomposition<splitint_t, fp_t>(viewA, config.splitType, config.numSplitsA, bitsPerSlice, normalisationDimension::byRows);
917 auto splitB = multiterm::Decomposition<splitint_t, fp_t>(viewB, config.splitType, config.numSplitsB, bitsPerSlice, normalisationDimension::byCols);
918
919 splitA.prepare();
920 splitB.prepare();
921
922 size_t numDiagonals;
923 switch (config.multType) {
924 case multiterm::multiplicationStrategy::reduced:
925 // Products below the main anti-diagonal are ignored.
926 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
927 break;
928 case multiterm::multiplicationStrategy::full:
929 // All products are computed.
930 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
931 break;
932 // LCOV_EXCL_START
933 default:
934 throw std::logic_error("Unhandled multiplicationStrategy");
935 // LCOV_EXCL_STOP
936 }
937
938 switch (config.redType) {
939 case multiterm::reductionStrategy::floatingPoint:
940 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
941 case multiterm::reductionStrategy::integer:
942 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
