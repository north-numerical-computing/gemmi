422template <typename fp_t, typename splitint_t, typename accumulator_t>
423DerivedParameters deriveParameters(matrix::MatrixView<const fp_t> A,
424 matrix::MatrixView<const fp_t> B,
425 const config& config) {
426
427 // Compile-time type checks.
428 static_assert(std::is_floating_point_v<fp_t>,
429 "fp_t must be a floating-point type");
430 static_assert(std::is_integral_v<splitint_t> && std::is_signed_v<splitint_t>,
431 "splitint_t must be a signed integer type");
432 static_assert(std::is_integral_v<accumulator_t> && std::is_signed_v<accumulator_t>,
433 "accumulator_t must be a signed integer type");
434
435 // Matrix checks.
436 if (A.data == nullptr)
437 throw std::invalid_argument("Matrix A has a null data pointer");
438 if (B.data == nullptr)
439 throw std::invalid_argument("Matrix B has a null data pointer");
440 if (A.empty())
441 throw std::invalid_argument("Matrix A is empty (rows or cols is 0)");
442 if (B.empty())
443 throw std::invalid_argument("Matrix B is empty (rows or cols is 0)");
444
445 if (A.cols != B.rows)
446 throw std::invalid_argument(
447 "Dimension mismatch: A.cols (" + std::to_string(A.cols) +
448 ") != B.rows (" + std::to_string(B.rows) + ")");
449
450 // Split counts.
451 if (config.numSplitsA == 0)
452 throw std::invalid_argument("numSplitsA must be >= 1");
453 if (config.numSplitsB == 0)
454 throw std::invalid_argument("numSplitsB must be >= 1");
455
456 // Custom mask size.
457 if (std::holds_alternative<std::vector<bool>>(config.multSpecification)) {
458 const auto& mask = std::get<std::vector<bool>>(config.multSpecification);
459 const size_t expected = config.numSplitsA * config.numSplitsB;
460 if (mask.size() != expected)
461 throw std::invalid_argument(
462 "Custom mask size (" + std::to_string(mask.size()) +
463 ") does not match numSplitsA * numSplitsB (" +
464 std::to_string(expected) + ")");
465 }
466
467 // Type width constraints.
468 constexpr size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
469 constexpr size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
470 if (bitsPerInteger > bitsInAccumulator / 2)
471 throw std::invalid_argument(
472 "splitint_t (" + std::to_string(bitsPerInteger) + " bits) is too wide "
473 "for accumulator_t (" + std::to_string(bitsInAccumulator) + " bits): "
474 "require bitsPerInteger <= bitsInAccumulator / 2");
475
476 const size_t k = A.cols;
477 const double log2k = (k > 1) ? std::log2(static_cast<double>(k)) : 0.0;
478 const auto alpha = static_cast<size_t>(
479 std::max(0.0, std::floor((static_cast<double>(bitsInAccumulator) - log2k) / 2.0)));
480 const size_t bitsPerSlice = std::min(bitsPerInteger, alpha);
481 if (bitsPerSlice == 0)
482 throw std::invalid_argument(
483 "Computed bitsPerSlice is 0: inner dimension k=" + std::to_string(k) +
484 " is too large for accumulator_t (" +
485 std::to_string(bitsInAccumulator) + " bits)");
486
487 return DerivedParameters{bitsPerSlice};
488}

641template <typename splitint_t, typename fp_t>
642void computeNormalisationVectors(preparedOperand<splitint_t, fp_t>& operand) {
643 // Compute normalisation vector.
644 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
645 operand.powersVector[outer] = 0.0;
646 for (size_t j = 0; j < operand.innerDimension(); j++) {
647 operand.powersVector[outer] = std::max(operand.powersVector[outer],
648 std::abs(operand.operand(outer, j)));
649 }
650 // Compute the smallest power of 2 that is strictly greater than the
651 // maximum value in the row/column.
652 // NOTE 1: This is not the technique used in uoi24.
653 // NOTE 2: I use exponents instead of powers of 2, as I need the former
654 // to shift correctly.
655 operand.scalingExponents[outer] = fp::getStoredFloatingPointExponent(operand.powersVector[outer]);
656 operand.powersVector[outer] = std::ldexp(1.0, operand.scalingExponents[outer]);
657 }
658}

676template <typename splitint_t, typename fp_t>
677void computeFixedPointRepresentationVector(std::vector<typename fp::FloatingPointTraits<fp_t>::StorageType> &fraction,
678 std::vector<bool> &sign, size_t outer,
679 const preparedOperand<splitint_t, fp_t>& operand) {
680 using uint_t = typename fp::FloatingPointTraits<fp_t>::StorageType;
681 constexpr size_t numSignificandBits = fp::FloatingPointTraits<fp_t>::numSignificandBits;
682 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
683 fp_t value = operand.operand(outer, inner);
684 fraction[inner] = std::bit_cast<uint_t>(value);
685 sign[inner] = std::signbit(value);
686 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
687 fraction[inner] = fraction[inner] & bitmask;
688 // Restore implicit bit for normal numbers.
689 // NOTE: NaNs and infs are currently not supported.
690 if (std::fpclassify(value) == FP_NORMAL)
691 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
692 }
693}

708template <typename splitint_t, typename fp_t>
709void computeSplitsWithTruncation(preparedOperand<splitint_t, fp_t>& operand) {
710 using uint_t = typename fp::FloatingPointTraits<fp_t>::StorageType;
711
712 // Compute splits one row/column at a time.
713 constexpr size_t numSignificandBits = fp::FloatingPointTraits<fp_t>::numSignificandBits;
714 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
715 std::vector<uint_t> fraction (operand.innerDimension());
716 std::vector<bool> sign (operand.innerDimension());
717 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
718 // Get binary representation of significands of normalised row/column.
719 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
720
721 // Create bitmask.
722 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
723
724 // Perform the split.
725 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
726 // NOTE: I could have a special path for 0.
727 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
728 int currentExponent = fp::getStoredFloatingPointExponent(operand.operand(outer, inner));
729 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
730 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
731 if (exponentDifference > (signed)bitsPerSlice) {
732 exponentDifference -= bitsPerSlice;
733 } else {
734 shiftCounter += exponentDifference;
735 exponentDifference = 0;
736 uint_t bitmask = shiftCounter > 0 ?
737 smallBitmask << shiftCounter :
738 smallBitmask >> -shiftCounter;
739 uint_t currentSlice = fraction[inner] & bitmask;
740 uint_t currentSplit = shiftCounter > 0 ?
741 currentSlice >> shiftCounter :
742 currentSlice << -shiftCounter;
743 splitint_t value = static_cast<splitint_t>(currentSplit) * (sign[inner] ? -1 : 1);
744 operand.splitValue(outer, inner, slice) = value;
745 shiftCounter -= bitsPerSlice;
746 }
747 }
748 }
749 }

767template <typename splitint_t, typename fp_t>
768void computeSplitsWithUnsignedEncoding(preparedOperand<splitint_t, fp_t>& operand) {
769 using uint_t = typename fp::FloatingPointTraits<fp_t>::StorageType;
770 using wideint_t = std::conditional_t<(sizeof(splitint_t) < sizeof(int)), int, std::intmax_t>;
771
772 // Compute splits one row/column at a time.
773 constexpr size_t numSignificandBits = fp::FloatingPointTraits<fp_t>::numSignificandBits;
774 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
775 std::vector<uint_t> fraction (operand.innerDimension());
776 std::vector<bool> sign (operand.innerDimension());
777 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
778 // Get binary representation of significands of normalised row/column.
779 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
780
781 // Create bitmasks.
782 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
783 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
784 // is to keep the algorithm consistent with the bitmasking approach
785 // above.
786 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
787 // is not mentioned in sabb26, but it seems necessary to do this to
788 // ensure that the first (signed) slice does not overflow if the full
789 // bitwidth of splitint_t is used and the second slice is negative in
790 // two's complement.
791 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
792 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
793
794 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
795
796 auto matrixIndex = operand.operandIndex(outer, inner);
797
798 // NOTE: I could have a special path for 0.
799 int16_t shiftCounter;
800 int currentExponent = fp::getStoredFloatingPointExponent(operand.operand(outer, inner));
801 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
802
803 splitint_t value = 0;
804 uint_t remainder = 0;
805
806 // Slice 0.
807 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
808 // Value is too small to contribute explicit bits to slice 0.
809 value = sign[inner] ? -1 : 0;
810 remainder = sign[inner] ?
811 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
812 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
813 fraction[inner];
814 exponentDifference -= (bitsPerSlice - 1);
815 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
816 } else {
817 // Truncation.
818 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
819 exponentDifference = 0;
820 uint_t bitmask = (shiftCounter > 0)
821 ? (smallBitmask << shiftCounter)
822 : (smallBitmask >> -shiftCounter);
823 uint_t currentSlice = fraction[inner] & bitmask;
824 uint_t currentSplit = (shiftCounter > 0)
825 ? (currentSlice >> shiftCounter)
826 : (currentSlice << -shiftCounter);
827
828 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
829
830 // Rounding.
831 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
832 uint_t lowBits = fraction[inner] & lowMask;
833
834 if (lowBits != 0) {
835 value += (sign[inner] ? -1 : 0);
836 }
837 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
838 shiftCounter -= (bitsPerSlice + 1);
839 }
840 operand.memory[matrixIndex] = value;
841
842 // Remaining slices.
843 const auto width = bitsPerSlice + 1;
844 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
845 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
846 const auto digitMax = cutoff - 1; // 2^b - 1
847 const auto digitMin = -cutoff; // -2^b
848 for (size_t slice = 1; slice < operand.prepConfig.numSplits; slice++) {
849 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
850 exponentDifference -= (bitsPerSlice + 1);
851 if (sign[inner]) {
852 operand.splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
853 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
854 }
855 } else {
856 shiftCounter += exponentDifference;
857 exponentDifference = 0;
858
859 uint_t bitmask = shiftCounter > 0 ?
860 largeBitmask << shiftCounter :
861 largeBitmask >> -shiftCounter;
862
863 uint_t currentSlice = remainder & bitmask;
864 uint_t currentSplit = shiftCounter > 0 ?
865 currentSlice >> shiftCounter :
866 currentSlice << -shiftCounter;
867
868 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
869 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
870 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
871 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
872 currentSplit |= maskToAdd;
873 }
874
875 // Update previous slices if currentSplit is negative in splitint_t.
876 // Carry must propoagate upward until slice 0 if necessary.
877 if (currentSplit > static_cast<uint_t>(digitMax)) {
878 int prevSliceIndex = static_cast<int>(slice) - 1;
879 while (true) {
880 auto newValue = static_cast<wideint_t>(operand.splitValue(outer, inner, prevSliceIndex)) + 1;
881 if (newValue <= digitMax) {
882 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
883 break;
884 } else {
885 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
886 prevSliceIndex--;
887 }
888 }
889 }
890
891 // Store the slice as a signed (b+1)-bit value:
892 // [0, 2^b-1] -> unchanged
893 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
894 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
895 if (signedDigit >= cutoff) {
896 signedDigit -= base;
897 }
898
899 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
900 shiftCounter -= (bitsPerSlice + 1);
901 }
902 }
903 }
904 }
905}

920template <typename splitint_t, typename fp_t>
921void computeSplitsWithRoundToNearest(preparedOperand<splitint_t, fp_t>& operand) {
922 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
923 auto localMatrix = std::vector<fp_t>(operand.matrix.data, operand.matrix.data + operand.matrix.size());
924 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
925 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
926 // Compute exponent in signed arithmetic to avoid wraparound when
927 // bitsPerSlice * (slice + 1) approaches numSignificandBits.
928 int exponent = static_cast<int>(fp::FloatingPointTraits<fp_t>::numSignificandBits) - static_cast<int>(bitsPerSlice * (slice + 1)) + 1;
929 fp_t sigma = std::ldexp(0.75, exponent) * operand.powersVector[outer];
930 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
931 auto matrixIndex = operand.operandIndex(outer, inner);
932 auto value = (localMatrix[matrixIndex] + sigma);
933 value -= sigma;
934 localMatrix[matrixIndex] -= value;
935 value = value / operand.powersVector[outer] * std::ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
936 operand.splitValue(outer, inner, slice) = value;
937 }
938 }
939 }

947template <typename splitint_t, typename fp_t>
948preparedOperand<splitint_t, fp_t> prepareOperand(matrix::MatrixView<const fp_t> matrix,
949 const OperandPreparationConfig& prepConfig) {
950 preparedOperand<splitint_t, fp_t> operand;
951 operand.matrix = matrix;
952 operand.prepConfig = prepConfig;
953 operand.memory.resize(operand.matrix.size() * prepConfig.numSplits);
954 operand.powersVector.resize(operand.prepConfig.dimension == normalisationDimension::byRows ?
955 operand.rows() : operand.cols());
956 operand.scalingExponents.resize(operand.powersVector.size());
957
958 computeNormalisationVectors(operand);
959
960 switch (prepConfig.splitType) {
961 case splittingStrategy::truncation:
962 computeSplitsWithTruncation(operand);
963 break;
964 case splittingStrategy::unsignedEncoding:
965 computeSplitsWithUnsignedEncoding(operand);
966 break;
967 case splittingStrategy::roundToNearest:
968 computeSplitsWithRoundToNearest(operand);
969 break;
970 }
971
972 return operand;
973}

985 struct multiplicationSchedule {
986 size_t numSplitsA; // number of slices for A
987 size_t numSplitsB; // number of slices for B
988
989 std::vector<bool> mask;
990
996 multiplicationSchedule(size_t numSplitsA, size_t numSplitsB) :
997 numSplitsA(numSplitsA), numSplitsB(numSplitsB), mask(numSplitsA * numSplitsB, false) {}
998
999 inline std::vector<bool>::reference operator()(size_t i, size_t j) {
1000 return mask[i * numSplitsB + j];
1001 }
1002 inline bool operator()(size_t i, size_t j) const {
1003 return mask[i * numSplitsB + j];
1004 }
1005};
1006
1017inline multiplicationSchedule makeSchedule(const config& config) {
1018 multiplicationSchedule schedule(config.numSplitsA, config.numSplitsB);
1019
1020 std::visit([&](auto&& spec) {
1021 using T = std::decay_t<decltype(spec)>;
1022
1023 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
1024 if (spec == multiplicationStrategy::full) {
1025 std::fill(schedule.mask.begin(), schedule.mask.end(), true);
1026 } else {
1027 size_t limit = std::max(config.numSplitsA, config.numSplitsB) - 1;
1028 for (size_t i = 0; i < config.numSplitsA; ++i)
1029 for (size_t j = 0; j < config.numSplitsB; ++j)
1030 schedule(i, j) = i + j <= limit;
1031 }
1032 } else if constexpr (std::is_same_v<T, std::vector<bool>>) {
1033
1034 const auto& mask = spec;
1035
1036 if (mask.size() != config.numSplitsA * config.numSplitsB)
1037 throw std::invalid_argument("Mask size mismatch.");
1038
1039 schedule.mask = mask; // row‑major copy
1040 }
1041
1042 }, config.multSpecification);
1043

1095template <typename splitint_t, typename accumulator_t, typename fp_t>
1096std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const preparedOperand<splitint_t, fp_t> &A,
1097 const preparedOperand<splitint_t, fp_t> &B,
1098 const multiplicationSchedule &schedule,
1099 const matrix::matrixLayout layoutC) {
1100 auto numSplitsA = A.prepConfig.numSplits;
1101 auto numSplitsB = B.prepConfig.numSplits;
1102 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1103 for (size_t diagonal = 0; diagonal <= numSplitsA + numSplitsB - 1; diagonal++) {
1104 int Aindex = diagonal < numSplitsA ? static_cast<int>(diagonal) : static_cast<int>(numSplitsA - 1);
1105 size_t Bindex = diagonal > numSplitsA - 1 ? diagonal - numSplitsA + 1 : 0;
1106 while (Aindex >= 0 && Bindex <= std::min(diagonal, numSplitsB - 1)) {
1107 if (schedule(Aindex, Bindex)) {
1108 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
1109 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1110 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1111 for (size_t row = 0; row < A.rows(); row++) {
1112 for (size_t col = 0; col < B.cols(); col++) {
1113 auto index = (layoutC == matrix::matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1114 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1115 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1116 C[index] += scaledSum * scalingFactor;
1117 }
1118 }
1119 }
1120 Aindex--;
1121 Bindex++;
1122 }
1123 }
1124
1125 return C;
1126}

1148template <typename splitint_t, typename accumulator_t, typename fp_t>
1149std::vector<fp_t> computeProductsWithIntegerAccumulation(const preparedOperand<splitint_t, fp_t> &A,
1150 const preparedOperand<splitint_t, fp_t> &B,
1151 const multiplicationSchedule &schedule,
1152 const matrix::matrixLayout layoutC) {
1153 auto numSplitsA = A.prepConfig.numSplits;
1154 auto numSplitsB = B.prepConfig.numSplits;
1155
1156 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1157 for (size_t diagonal = 0; diagonal <= numSplitsA + numSplitsB - 1; diagonal++) {
1158 int Aindex = diagonal < numSplitsA ? static_cast<int>(diagonal) : static_cast<int>(numSplitsA - 1);
1159 size_t Bindex = diagonal > numSplitsA - 1 ? diagonal - numSplitsA + 1 : 0;
1160
1161 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1162
1163 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
1164 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
1165 while (Aindex >= 0 && Bindex <= std::min(diagonal, numSplitsB - 1)) {
1166 if (schedule(Aindex, Bindex))
1167 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1168 Aindex--;
1169 Bindex++;
1170 }
1171
1172 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
1173 for (size_t row = 0; row < A.rows(); row++) {
1174 for (size_t col = 0; col < B.cols(); col++) {
1175 auto index = (layoutC == matrix::matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1176 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1177 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1178 C[index] += scaledSum * scalingFactor;
1179 }
1180 }
1181 }
1182
1183 return C;
1184}
