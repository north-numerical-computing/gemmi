422 void computeNormalisationVectors() {
423 // Compute normalisation vector.
424 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
425 this->powersVector[outer] = 0.0;
426 for (size_t j = 0; j < this->innerDimension(); j++) {
427 this->powersVector[outer] = std::max(this->powersVector[outer],
428 std::abs(this->operand(outer, j)));
429 }
430 // Compute the smallest power of 2 that is strictly greater than the
431 // maximum value in the row/column.
432 // NOTE 1: This is not the technique used in uoi24.
433 // NOTE 2: I use exponents instead of powers of 2, as I need the former
434 // to shift correctly.
435 this->scalingExponents[outer] = getStoredFloatingPointExponent(this->powersVector[outer]);
436 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);
437 }

475 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
476 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
477 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
478 fp_t value = this->operand(outer, inner);
479 fraction[inner] = std::bit_cast<uint_t>(value);
480 sign[inner] = std::signbit(value);
481 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
482 fraction[inner] = fraction[inner] & bitmask;
483 // Restore implicit bit for normal numbers.
484 // NOTE: NaNs and infs are currently not supported.
485 if (std::fpclassify(value) == FP_NORMAL)
486 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
487 }

499 void computeSplitsWithTruncation() {
500 this->splitType = splittingStrategy::truncation;
501 // Compute splits one row/column at a time.
502 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
503 auto bitsPerSlice = this->bitsPerSlice;
504 std::vector<uint_t> fraction (this->innerDimension());
505 std::vector<bool> sign (this->innerDimension());
506 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
507 // Get binary representation of significands of normalised row/column.
508 computeFixedPointRepresentationVector(fraction, sign, outer);
509
510 // Create bitmask.
511 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
512
513 // Perform the split.
514 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
515 // NOTE: I could have a special path for 0.
516 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
517 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
518 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
519 for (size_t slice = 0; slice < numSplits; slice++) {
520 if (exponentDifference > (signed)bitsPerSlice) {
521 exponentDifference -= bitsPerSlice;
522 } else {
523 shiftCounter += exponentDifference;
524 exponentDifference = 0;
525 uint_t bitmask = shiftCounter > 0 ?
526 smallBitmask << shiftCounter :
527 smallBitmask >> -shiftCounter;
528 uint_t currentSlice = fraction[inner] & bitmask;
529 uint_t currentSplit = shiftCounter > 0 ?
530 currentSlice >> shiftCounter :
531 currentSlice << -shiftCounter;
532 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
533 this->splitValue(outer, inner, slice) = value;
534 shiftCounter -= bitsPerSlice;
535 }
536 }
537 }
538 }

552 void computeSplitsWithUnsignedEncoding() {
553 this->splitType = splittingStrategy::unsignedEncoding;
554 // Compute splits one row/column at a time.
555 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
556 auto bitsPerSlice = this->bitsPerSlice;
557 std::vector<uint_t> fraction (this->innerDimension());
558 std::vector<bool> sign (this->innerDimension());
559 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
560 // Get binary representation of significands of normalised row/column.
561 computeFixedPointRepresentationVector(fraction, sign, outer);
562
563 // Create bitmasks.
564 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
565 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
566 // is to keep the algorithm consistent with the bitmasking approach
567 // above.
568 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
569 // is not mentioned in sabb26, but it seems necessary to do this to
570 // ensure that the first (signed) slice does not overflow if the full
571 // bitwidth of splitint_t is used and the second slice is negative in
572 // two's complement.
573 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
574 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
575
576 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
577
578 auto matrixIndex = operandIndex(outer, inner);
579
580 // NOTE: I could have a special path for 0.
581 int16_t shiftCounter;
582 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
583 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
584
585 splitint_t value = 0;
586 uint_t remainder = 0;
587
588 // Slice 0.
589 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
590 // Value is too small to contribute explicit bits to slice 0.
591 value = sign[inner] ? -1 : 0;
592 remainder = sign[inner] ?
593 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
594 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
595 fraction[inner];
596 exponentDifference -= (bitsPerSlice - 1);
597 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
598 } else {
599 // Truncation.
600 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
601 exponentDifference = 0;
602 uint_t bitmask = (shiftCounter > 0)
603 ? (smallBitmask << shiftCounter)
604 : (smallBitmask >> -shiftCounter);
605 uint_t currentSlice = fraction[inner] & bitmask;
606 uint_t currentSplit = (shiftCounter > 0)
607 ? (currentSlice >> shiftCounter)
608 : (currentSlice << -shiftCounter);
609
610 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
611
612 // Rounding.
613 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
614 uint_t lowBits = fraction[inner] & lowMask;
615
616 if (lowBits != 0) {
617 value += (sign[inner] ? -1 : 0);
618 }
619 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
620 shiftCounter -= (bitsPerSlice + 1);
621 }
622 this->memory[matrixIndex] = value;
623
624 // Remaining slices.
625 const auto width = bitsPerSlice + 1;
626 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
627 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
628 const auto digitMax = cutoff - 1; // 2^b - 1
629 const auto digitMin = -cutoff; // -2^b
630 for (size_t slice = 1; slice < numSplits; slice++) {
631 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
632 exponentDifference -= (bitsPerSlice + 1);
633 if (sign[inner]) {
634 this->splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
635 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
636 }
637 } else {
638 shiftCounter += exponentDifference;
639 exponentDifference = 0;
640
641 uint_t bitmask = shiftCounter > 0 ?
642 largeBitmask << shiftCounter :
643 largeBitmask >> -shiftCounter;
644
645 uint_t currentSlice = remainder & bitmask;
646 uint_t currentSplit = shiftCounter > 0 ?
647 currentSlice >> shiftCounter :
648 currentSlice << -shiftCounter;
649
650 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
651 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
652 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
653 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
654 currentSplit |= maskToAdd;
655 }
656
657 // Update previous slices if currentSplit is negative in splitint_t.
658 // Carry must propoagate upward until slice 0 if necessary.
659 if (currentSplit > static_cast<uint_t>(digitMax)) {
660 int prevSliceIndex = static_cast<int>(slice) - 1;
661 while (true) {
662 auto newValue = static_cast<wideint_t>(this->splitValue(outer, inner, prevSliceIndex)) + 1;
663 if (newValue <= digitMax) {
664 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
665 break;
666 } else {
667 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
668 prevSliceIndex--;
669 }
670 }
671 }
672
673 // Store the slice as a signed (b+1)-bit value:
674 // [0, 2^b-1] -> unchanged
675 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
676 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
677 if (signedDigit >= cutoff) {
678 signedDigit -= base;
679 }
680
681 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
682 shiftCounter -= (bitsPerSlice + 1);
683 }
684 }
685 }
686 }
687 }

698 void computeSplitsWithRoundToNearest() {
699 this->splitType = splittingStrategy::roundToNearest;
700 auto bitsPerSlice = this->bitsPerSlice;
701 auto localMatrix = std::vector<fp_t>(this->matrix.data, this->matrix.data + this->matrix.size());
702 for (size_t slice = 0; slice < numSplits; slice++) {
703 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
704 fp_t sigma = ldexp(0.75,
705 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
706 powersVector[outer];
707 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
708 auto matrixIndex = operandIndex(outer, inner);
709 auto value = (localMatrix[matrixIndex] + sigma);
710 value -= sigma;
711 localMatrix[matrixIndex] -= value;
712 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
713 this->splitValue(outer, inner, slice) = value;
714 }
715 }
716 }
717 }

786template <typename splitint_t, typename accumulator_t, typename fp_t>
787std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
788 const MatrixSplit<splitint_t, fp_t> &B,
789 const matrixLayout layoutC,
790 const size_t numDiagonals) {
791 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
792 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
793 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
794 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
795 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
796 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
797 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
798 for (size_t row = 0; row < A.rows(); row++) {
799 for (size_t col = 0; col < B.cols(); col++) {
800 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
801 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
802 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
803 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
804 C[index] += scaledSum * scalingFactor;
805 }
806 }
807 Aindex--;
808 Bindex++;
809 }
810 }
811
812 return C;
813}

835template <typename splitint_t, typename accumulator_t, typename fp_t>
836std::vector<fp_t> computeProductsWithIntegerAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
837 const MatrixSplit<splitint_t, fp_t> &B,
838 const matrixLayout layoutC,
839 const size_t numDiagonals) {
840
841 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
842 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
843 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
844 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
845
846 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
847
848 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
849 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
850 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
851 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
852 Aindex--;
853 Bindex++;
854 }
855
856 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
857 for (size_t row = 0; row < A.rows(); row++) {
858 for (size_t col = 0; col < B.cols(); col++) {
859 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
860 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
861 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
862 C[index] += scaledSum * scalingFactor;
863 }
864 }
865 }
866
867 return C;
868}

887template <typename fp_t, typename splitint_t, typename accumulator_t>
888std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const matrixLayout layoutA,
889 const std::vector<fp_t> &B, const matrixLayout layoutB,
890 const size_t m, const size_t k, const size_t n,
891 const size_t numSplitsA, const size_t numSplitsB,
892 const matrixLayout layoutC = matrixLayout::columnMajor,
893 const splittingStrategy splitType = splittingStrategy::roundToNearest,
894 const accumulationStrategy accType = accumulationStrategy::floatingPoint,
895 const multiplicationStrategy multType = multiplicationStrategy::reduced) {
896
897 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
898 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
899 if (bitsPerInteger > bitsInAccumulator / 2) {
900 throw std::invalid_argument("Split integer type is too wide for the chosen accumulator type");
901 }
902 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
903 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
904
905 auto viewA = makeMatrixView(A, m, k, layoutA);
906 auto viewB = makeMatrixView(B, k, n, layoutB);
907
908 auto splitA = MatrixSplit<splitint_t, fp_t>(viewA, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
909 auto splitB = MatrixSplit<splitint_t, fp_t>(viewB, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
910
911 splitA.prepare();
912 splitB.prepare();
913
914 size_t numDiagonals;
915 switch (multType) {
916 case multiplicationStrategy::reduced:
917 // Products below the main anti-diagonal are ignored.
918 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
919 break;
920 case multiplicationStrategy::full:
921 // All products are computed.
922 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
923 break;
924 // LCOV_EXCL_START
925 default:
926 throw std::logic_error("Unhandled multiplicationStrategy");
927 // LCOV_EXCL_STOP
928 }
929
930 switch (accType) {
931 case accumulationStrategy::floatingPoint:
932 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
933 case accumulationStrategy::integer:
934 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
935 // LCOV_EXCL_START
936 default:
937 throw std::logic_error("Unhandled accumulationStrategy");
938 // LCOV_EXCL_STOP
939 }
940}
