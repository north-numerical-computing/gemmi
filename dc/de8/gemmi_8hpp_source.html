550template <typename splitint_t, typename fp_t>
551void computeNormalisationVectors(preparedOperand<splitint_t, fp_t>& operand) {
552 // Compute normalisation vector.
553 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
554 operand.powersVector[outer] = 0.0;
555 for (size_t j = 0; j < operand.innerDimension(); j++) {
556 operand.powersVector[outer] = std::max(operand.powersVector[outer],
557 std::abs(operand.operand(outer, j)));
558 }
559 // Compute the smallest power of 2 that is strictly greater than the
560 // maximum value in the row/column.
561 // NOTE 1: This is not the technique used in uoi24.
562 // NOTE 2: I use exponents instead of powers of 2, as I need the former
563 // to shift correctly.
564 operand.scalingExponents[outer] = getStoredFloatingPointExponent(operand.powersVector[outer]);
565 operand.powersVector[outer] = std::ldexp(1.0, operand.scalingExponents[outer]);
566 }

580template <typename splitint_t, typename fp_t>
581void computeFixedPointRepresentationVector(std::vector<typename FloatingPointTraits<fp_t>::StorageType> &fraction,
582 std::vector<bool> &sign, size_t outer,
583 const preparedOperand<splitint_t, fp_t>& operand) {
584 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
585 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
586 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
587 fp_t value = operand.operand(outer, inner);
588 fraction[inner] = std::bit_cast<uint_t>(value);
589 sign[inner] = std::signbit(value);
590 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
591 fraction[inner] = fraction[inner] & bitmask;
592 // Restore implicit bit for normal numbers.
593 // NOTE: NaNs and infs are currently not supported.
594 if (std::fpclassify(value) == FP_NORMAL)
595 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
596 }
597}

612template <typename splitint_t, typename fp_t>
613void computeSplitsWithTruncation(preparedOperand<splitint_t, fp_t>& operand) {
614 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
615
616 // Compute splits one row/column at a time.
617 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
618 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
619 std::vector<uint_t> fraction (operand.innerDimension());
620 std::vector<bool> sign (operand.innerDimension());
621 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
622 // Get binary representation of significands of normalised row/column.
623 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
624
625 // Create bitmask.
626 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
627
628 // Perform the split.
629 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
630 // NOTE: I could have a special path for 0.
631 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
632 int currentExponent = getStoredFloatingPointExponent(operand.operand(outer, inner));
633 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
634 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
635 if (exponentDifference > (signed)bitsPerSlice) {
636 exponentDifference -= bitsPerSlice;
637 } else {
638 shiftCounter += exponentDifference;
639 exponentDifference = 0;
640 uint_t bitmask = shiftCounter > 0 ?
641 smallBitmask << shiftCounter :
642 smallBitmask >> -shiftCounter;
643 uint_t currentSlice = fraction[inner] & bitmask;
644 uint_t currentSplit = shiftCounter > 0 ?
645 currentSlice >> shiftCounter :
646 currentSlice << -shiftCounter;
647 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
648 operand.splitValue(outer, inner, slice) = value;
649 shiftCounter -= bitsPerSlice;
650 }
651 }
652 }
653 }

671template <typename splitint_t, typename fp_t>
672void computeSplitsWithUnsignedEncoding(preparedOperand<splitint_t, fp_t>& operand) {
673 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
674 using wideint_t = std::conditional_t<(sizeof(splitint_t) < sizeof(int)), int, std::intmax_t>;
675
676 // Compute splits one row/column at a time.
677 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
678 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
679 std::vector<uint_t> fraction (operand.innerDimension());
680 std::vector<bool> sign (operand.innerDimension());
681 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
682 // Get binary representation of significands of normalised row/column.
683 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
684
685 // Create bitmasks.
686 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
687 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
688 // is to keep the algorithm consistent with the bitmasking approach
689 // above.
690 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
691 // is not mentioned in sabb26, but it seems necessary to do this to
692 // ensure that the first (signed) slice does not overflow if the full
693 // bitwidth of splitint_t is used and the second slice is negative in
694 // two's complement.
695 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
696 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
697
698 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
699
700 auto matrixIndex = operand.operandIndex(outer, inner);
701
702 // NOTE: I could have a special path for 0.
703 int16_t shiftCounter;
704 int currentExponent = getStoredFloatingPointExponent(operand.operand(outer, inner));
705 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
706
707 splitint_t value = 0;
708 uint_t remainder = 0;
709
710 // Slice 0.
711 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
712 // Value is too small to contribute explicit bits to slice 0.
713 value = sign[inner] ? -1 : 0;
714 remainder = sign[inner] ?
715 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
716 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
717 fraction[inner];
718 exponentDifference -= (bitsPerSlice - 1);
719 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
720 } else {
721 // Truncation.
722 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
723 exponentDifference = 0;
724 uint_t bitmask = (shiftCounter > 0)
725 ? (smallBitmask << shiftCounter)
726 : (smallBitmask >> -shiftCounter);
727 uint_t currentSlice = fraction[inner] & bitmask;
728 uint_t currentSplit = (shiftCounter > 0)
729 ? (currentSlice >> shiftCounter)
730 : (currentSlice << -shiftCounter);
731
732 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
733
734 // Rounding.
735 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
736 uint_t lowBits = fraction[inner] & lowMask;
737
738 if (lowBits != 0) {
739 value += (sign[inner] ? -1 : 0);
740 }
741 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
742 shiftCounter -= (bitsPerSlice + 1);
743 }
744 operand.memory[matrixIndex] = value;
745
746 // Remaining slices.
747 const auto width = bitsPerSlice + 1;
748 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
749 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
750 const auto digitMax = cutoff - 1; // 2^b - 1
751 const auto digitMin = -cutoff; // -2^b
752 for (size_t slice = 1; slice < operand.prepConfig.numSplits; slice++) {
753 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
754 exponentDifference -= (bitsPerSlice + 1);
755 if (sign[inner]) {
756 operand.splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
757 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
758 }
759 } else {
760 shiftCounter += exponentDifference;
761 exponentDifference = 0;
762
763 uint_t bitmask = shiftCounter > 0 ?
764 largeBitmask << shiftCounter :
765 largeBitmask >> -shiftCounter;
766
767 uint_t currentSlice = remainder & bitmask;
768 uint_t currentSplit = shiftCounter > 0 ?
769 currentSlice >> shiftCounter :
770 currentSlice << -shiftCounter;
771
772 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
773 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
774 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
775 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
776 currentSplit |= maskToAdd;
777 }
778
779 // Update previous slices if currentSplit is negative in splitint_t.
780 // Carry must propoagate upward until slice 0 if necessary.
781 if (currentSplit > static_cast<uint_t>(digitMax)) {
782 int prevSliceIndex = static_cast<int>(slice) - 1;
783 while (true) {
784 auto newValue = static_cast<wideint_t>(operand.splitValue(outer, inner, prevSliceIndex)) + 1;
785 if (newValue <= digitMax) {
786 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
787 break;
788 } else {
789 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
790 prevSliceIndex--;
791 }
792 }
793 }
794
795 // Store the slice as a signed (b+1)-bit value:
796 // [0, 2^b-1] -> unchanged
797 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
798 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
799 if (signedDigit >= cutoff) {
800 signedDigit -= base;
801 }
802
803 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
804 shiftCounter -= (bitsPerSlice + 1);
805 }
806 }
807 }
808 }
809}

824template <typename splitint_t, typename fp_t>
825void computeSplitsWithRoundToNearest(preparedOperand<splitint_t, fp_t>& operand) {
826 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
827 auto localMatrix = std::vector<fp_t>(operand.matrix.data, operand.matrix.data + operand.matrix.size());
828 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
829 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
830 fp_t sigma = ldexp(0.75,
831 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
832 operand.powersVector[outer];
833 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
834 auto matrixIndex = operand.operandIndex(outer, inner);
835 auto value = (localMatrix[matrixIndex] + sigma);
836 value -= sigma;
837 localMatrix[matrixIndex] -= value;
838 value = value / operand.powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
839 operand.splitValue(outer, inner, slice) = value;
840 }
841 }
842 }

850template <typename splitint_t, typename fp_t>
851preparedOperand<splitint_t, fp_t> prepareOperand(MatrixView<const fp_t> matrix,
852 const OperandPreparationConfig& prepConfig) {
853 preparedOperand<splitint_t, fp_t> operand;
854 operand.matrix = matrix;
855 operand.prepConfig = prepConfig;
856 operand.memory.resize(operand.matrix.size() * operand.prepConfig.numSplits * prepConfig.numSplits);
857 operand.powersVector.resize(operand.prepConfig.dimension == normalisationDimension::byRows ?
858 operand.rows() : operand.cols());
859 operand.scalingExponents.resize(operand.powersVector.size());
860
861 computeNormalisationVectors(operand);
862
863 switch (prepConfig.splitType) {
864 case splittingStrategy::truncation:
865 computeSplitsWithTruncation(operand);
866 break;
867 case splittingStrategy::unsignedEncoding:
868 computeSplitsWithUnsignedEncoding(operand);
869 break;
870 case splittingStrategy::roundToNearest:
871 computeSplitsWithRoundToNearest(operand);
872 break;
873 }
874
875 return operand;
876}

887 struct multiplicationSchedule {
888 size_t numSplitsA; // number of slices for A
889 size_t numSplitsB; // number of slices for B
890
891 std::vector<bool> mask;
892 inline std::vector<bool>::reference operator()(size_t i, size_t j) {
893 return mask[i * numSplitsB + j];
894 }
895 inline bool operator()(size_t i, size_t j) const {
896 return mask[i * numSplitsB + j];
897 }
898};
899
908inline multiplicationSchedule makeSchedule(const config& config) {
909 multiplicationSchedule schedule;
910 schedule.numSplitsA = config.numSplitsA;
911 schedule.numSplitsB = config.numSplitsB;
912 schedule.mask.resize(config.numSplitsA * config.numSplitsB, false);
913
914 std::visit([&](auto&& spec) {
915 using T = std::decay_t<decltype(spec)>;
916
917 // Predefined strategy (full or reduced).
918 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
919
920 if (spec == multiplicationStrategy::full) {
921 std::fill(schedule.mask.begin(), schedule.mask.end(), true);
922 }
923 else { // reduced (anti‑diagonal rule)
924 size_t limit = std::max(config.numSplitsA, config.numSplitsB) - 1;
925
926 for (size_t i = 0; i < config.numSplitsA; ++i)
927 for (size_t j = 0; j < config.numSplitsB; ++j)
928 if (i + j <= limit)
929 schedule(i, j) = true;
930 }
931 }
932
933 // Custom boolean mask.
934 else if constexpr (std::is_same_v<T, std::vector<bool>>) {
935
936 const auto& mask = spec;
937
938 if (mask.size() != config.numSplitsA * config.numSplitsB)
939 throw std::invalid_argument("Mask size mismatch.");
940
941 schedule.mask = mask; // row‑major copy
942 }
943
944 }, config.multSpecification);
945
946 return schedule;
947}

996template <typename splitint_t, typename accumulator_t, typename fp_t>
997std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const preparedOperand<splitint_t, fp_t> &A,
998 const preparedOperand<splitint_t, fp_t> &B,
999 const multiplicationSchedule &schedule,
1000 const matrixLayout layoutC) {
1001 auto numSplitsA = A.prepConfig.numSplits;
1002 auto numSplitsB = B.prepConfig.numSplits;
1003 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1004 for (size_t diagonal = 0; diagonal <= numSplitsA + numSplitsB - 1; diagonal++) {
1005 int Aindex = diagonal < numSplitsA - 1 ? diagonal : numSplitsA - 1;
1006 size_t Bindex = diagonal > numSplitsA - 1 ? diagonal - numSplitsA + 1 : 0;
1007 while (Aindex >= 0 && Bindex <= std::min(diagonal, numSplitsB - 1)) {
1008 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
1009 if (schedule(Aindex, Bindex)) {
1010 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1011 for (size_t row = 0; row < A.rows(); row++) {
1012 for (size_t col = 0; col < B.cols(); col++) {
1013 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1014 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1015 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1016 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1017 C[index] += scaledSum * scalingFactor;
1018 }
1019 }
1020 }
1021 Aindex--;
1022 Bindex++;
1023 }
1024 }

1050template <typename splitint_t, typename accumulator_t, typename fp_t>
1051std::vector<fp_t> computeProductsWithIntegerAccumulation(const preparedOperand<splitint_t, fp_t> &A,
1052 const preparedOperand<splitint_t, fp_t> &B,
1053 const multiplicationSchedule &schedule,
1054 const matrixLayout layoutC) {
1055 auto numSplitsA = A.prepConfig.numSplits;
1056 auto numSplitsB = B.prepConfig.numSplits;
1057
1058 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1059 for (size_t diagonal = 0; diagonal <= numSplitsA + numSplitsB - 1; diagonal++) {
1060 int Aindex = diagonal < numSplitsA ? static_cast<int>(diagonal) : static_cast<int>(numSplitsA - 1);
1061 size_t Bindex = diagonal > numSplitsA - 1 ? diagonal - numSplitsA + 1 : 0;
1062
1063 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1064
1065 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
1066 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
1067 while (Aindex >= 0 && Bindex <= std::min(diagonal, numSplitsB - 1)) {
1068 if (schedule(Aindex, Bindex))
1069 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1070 Aindex--;
1071 Bindex++;
1072 }
1073
1074 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
1075 for (size_t row = 0; row < A.rows(); row++) {
1076 for (size_t col = 0; col < B.cols(); col++) {
1077 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1078 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1079 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1080 C[index] += scaledSum * scalingFactor;
1081 }
1082 }
1083 }
1084
1085 return C;
1086}

1118 throw std::invalid_argument("Split integer type is too wide for the chosen accumulator type");

1121 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));

1130 multiterm::OperandPreparationConfig(config.splitType, config.numSplitsB, bitsPerSlice, normalisationDimension::byCols));
1131

1134
1135 // Execute multiplication based on reduction type.
1136 if (config.redType == multiterm::reductionStrategy::floatingPoint) {
