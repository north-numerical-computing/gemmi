194 void computeNormalisationVectors() {
195 // Compute normalisation vector.
196 const size_t iStride = this->iStride();
197 const size_t jStride = this->jStride();
198 for (size_t i = 0; i < this->otherDimension(); i++) {
199 this->powersVector[i] = 0.0;
200 for (size_t j = 0; j < this->innerProductDimension(); j++) {
201 const size_t index = i * iStride + j * jStride;
202 this->powersVector[i] = std::max(this->powersVector[i],
203 std::abs(this->matrix[index]));
204 }
205 // Compute the smallest power of 2 that is strictly greater than the
206 // maximum value in the row/column.
207 // NOTE 1: This is not the technique used in uoi24.
208 // NOTE 2: I use exponents instead of powers of 2, as I need the former
209 // to shift correctly.
210 this->scalingExponents[i] = getFloatingPointExponent(this->powersVector[i]);
211 this->powersVector[i] = std::ldexp(1.0, this->scalingExponents[i]);
212 }
213 }

220 int computeSliceBitOffset(size_t slice) const {
221 switch (splitType) {
222 case splittingStrategy::bitMasking:
223 // Slice k -> k * b
224 return static_cast<int>((slice + 1) * bitsPerSlice);
225
226 case splittingStrategy::roundToNearest:
227 // Slice 0 -> b - 1
228 // Slice k -> k * b - 1
229 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
230
231 case splittingStrategy::unsignedEncoding:
232 // Slice 0 -> b - 1
233 // Slice k -> (b - 1) + k * (b + 1)
234 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));
235 default:
236 UNREACHABLE();
237 }

247 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t i) {
248 auto numFracBits = computeNumFracBits<fp_t>();
249 auto iStride = this->iStride();
250 auto jStride = this->jStride();
251 for (size_t j = 0; j < this->innerProductDimension(); j++) {
252 const size_t index = i * iStride + j * jStride;
253 fp_t value = this->matrix[index]; // powersVector[i];
254 fraction[j] = std::bit_cast<uint_t>(value);
255 sign[j] = std::signbit(value);
256 uint_t bitmask = (1ull << (numFracBits - 1)) - 1;
257 fraction[j] = fraction[j] & bitmask;
258 // Restore implicit bit for normal numbers.
259 // NOTE: NaNs and infs are currently not supported.
260 if (std::fpclassify(value) == FP_NORMAL)
261 fraction[j] |= ((uint_t)1 << (numFracBits - 1));
262 }
263 }

274 void computeSplitsWithBitMasking() {
275 this->splitType = splittingStrategy::bitMasking;
276 // Compute splits one row/column at a time.
277 auto numFracBits = computeNumFracBits<fp_t>();
278 auto bitsPerSlice = this->bitsPerSlice;
279 auto iStride = this->iStride();
280 auto jStride = this->jStride();
281 std::vector<uint_t> fraction (this->innerProductDimension());
282 std::vector<bool> sign (this->innerProductDimension());
283 for (size_t i = 0; i < this->otherDimension(); i++) {
284 // Get binary representation of significands of normalised row/column.
285 computeFixedPointRepresentationVector(fraction, sign, i);
286
287 // Create bitmask.
288 const uint_t smallBitmask = (1 << bitsPerSlice) - 1;
289 // Perform the split.
290 for (size_t j = 0; j < this->innerProductDimension(); j++) {
291
292 // NOTE: I could have a special path for 0.
293 int16_t shiftCounter = numFracBits - bitsPerSlice;
294 int currentExponent = getFloatingPointExponent(this->matrix[i * iStride + j * jStride]);
295 int16_t exponentDifference = scalingExponents[i] - currentExponent;
296 for (size_t slice = 0; slice < numSplits; slice++) {
297 if (exponentDifference > (signed)bitsPerSlice) {
298 exponentDifference -= bitsPerSlice;
299 } else {
300 shiftCounter += exponentDifference;
301 exponentDifference = 0;
302 uint_t bitmask = shiftCounter > 0 ?
303 smallBitmask << shiftCounter :
304 smallBitmask >> -shiftCounter;
305 uint_t currentSlice = fraction[j] & bitmask;
306 uint_t currentSplit = shiftCounter > 0 ?
307 currentSlice >> shiftCounter :
308 currentSlice << -shiftCounter;
309 splitint_t value = (splitint_t)(currentSplit) * (sign[j] ? -1 : 1);
310 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = value;
311 shiftCounter -= bitsPerSlice;
312 }
313 }
314 }
315 }

329 void computeSplitsWithUnsignedEncoding() {
330 this->splitType = splittingStrategy::unsignedEncoding;
331 // Compute splits one row/column at a time.
332 auto numFracBits = computeNumFracBits<fp_t>();
333 auto bitsPerSlice = this->bitsPerSlice;
334 auto iStride = this->iStride();
335 auto jStride = this->jStride();
336 std::vector<uint_t> fraction (this->innerProductDimension());
337 std::vector<bool> sign (this->innerProductDimension());
338 for (size_t i = 0; i < this->otherDimension(); i++) {
339 // Get binary representation of significands of normalised row/column.
340 computeFixedPointRepresentationVector(fraction, sign, i);
341
342 // Create bitmasks.
343 // This algorithm uses bitsPerSlice bits for the first slice and bitsPerSlice + 1 bits for subsequent slices.
344 // This choice is to keep the algorithm consistent with the bitmasking approach above.
345 // NOTE: I am using a first slice with bitsPerSlice - 1 bits. This is not mentioned in sabb26, but it seems
346 // necessary to do this, if I want to ensure that the first slice does not overflow if the second slice
347 // is negative in two's complement.
348 const uint_t smallBitmask = (1 << (bitsPerSlice - 1)) - 1;
349 const uint_t largeBitmask = (1 << (bitsPerSlice + 1)) - 1;
350
351 for (size_t j = 0; j < this->innerProductDimension(); j++) {
352
353 // NOTE: I could have a special path for 0.
354 int currentExponent = getFloatingPointExponent(this->matrix[i * iStride + j * jStride]);
355 int16_t exponentDifference = scalingExponents[i] - currentExponent;
356
357 // Boundary below slice 0, fully aligned to this entry.
358 int16_t firstShift = numFracBits - (bitsPerSlice - 1) + exponentDifference;
359
360 splitint_t value = 0;
361 uint_t remainder = 0;
362
363 if (!sign[j]) {
364 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
365 // Too small to contribute explicit bits to slice 0.
366 value = 0;
367 remainder = fraction[j];
368 } else {
369 // Positive number: leading digit is truncation.
370 uint_t bitmask = (firstShift > 0)
371 ? (smallBitmask << firstShift)
372 : (smallBitmask >> -firstShift);
373
374 uint_t currentSlice = fraction[j] & bitmask;
375 uint_t currentSplit = (firstShift > 0)
376 ? (currentSlice >> firstShift)
377 : (currentSlice << -firstShift);
378
379 value = static_cast<splitint_t>(currentSplit);
380 remainder = fraction[j] & ((uint_t(1) << firstShift) - 1);
381 }
382 } else {
383 // Negative number.
384 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
385 // Too small to contribute explicit bits to slice 0:
386 // round toward -infinity gives -1.
387 value = -1;
388 remainder = ((1ull << (sizeof(uint_t) * 8 - 1)) - fraction[j]) | (uint_t(1) << (sizeof(uint_t) * 8 - 1));
389 } else {
390 // Normal truncation path for the leading slice.
391 uint_t bitmask = (firstShift > 0)
392 ? (smallBitmask << firstShift)
393 : (smallBitmask >> -firstShift);
394
395 uint_t currentSlice = fraction[j] & bitmask;
396 uint_t currentSplit = (firstShift > 0)
397 ? (currentSlice >> firstShift)
398 : (currentSlice << -firstShift);
399
400 value = -static_cast<splitint_t>(currentSplit);
401
402 uint_t lowMask = (uint_t(1) << firstShift) - 1;
403 uint_t lowBits = fraction[j] & lowMask;
404
405 if (lowBits != 0) {
406 value -= 1; // round toward -infinity
407 remainder = (1ull << firstShift) - lowBits;
408 } else {
409 remainder = 0;
410 }
411 }
412 }
413
414 this->memory[i * iStride + j * jStride] = value;
415
416 // From here on, the remainder is already aligned and nonnegative.
417 int shiftCounter = firstShift - (bitsPerSlice + 1);
418 exponentDifference = 0;
419
420 // Split (nonnegative) remainder with unsigned slice encoding.
421 for (size_t slice = 1; slice < numSplits; slice++) {
422 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
423 exponentDifference -= (bitsPerSlice + 1);
424 if (sign[j]) {
425 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = static_cast<splitint_t>(largeBitmask);
426 }
427 } else {
428 shiftCounter += exponentDifference;
429 exponentDifference = 0;
430
431 uint_t bitmask = shiftCounter > 0 ?
432 largeBitmask << shiftCounter :
433 largeBitmask >> -shiftCounter;
434
435 uint_t currentSlice = remainder & bitmask;
436 uint_t currentSplit = shiftCounter > 0 ?
437 currentSlice >> shiftCounter :
438 currentSlice << -shiftCounter;
439
440 if (sign[j] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
441 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
442 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
443 uint_t maskToAdd = ((1ull << bitsLost) - 1) << bitsUsed;
444 currentSplit |= maskToAdd;
445 }
446
447 // Width of sub-leading slices is (bitsPerSlice + 1) bits.
448 const auto width = bitsPerSlice + 1;
449 const auto cutoff = static_cast<wideint_t>(uint_t(1) << bitsPerSlice); // 2^b
450 const auto base = static_cast<wideint_t>(uint_t(1) << width); // 2^(b+1)
451 const auto digitMax = cutoff - 1; // 2^b - 1
452 const auto digitMin = -cutoff; // -2^b
453
454 // If the unsigned digit is in the upper half, propagate carry upward.
455 if (currentSplit >= static_cast<uint_t>(cutoff)) {
456 int nextSlice = static_cast<int>(slice) - 1;
457 while (true) {
458 const size_t prevIdx =
459 i * iStride + j * jStride + nextSlice * this->matrix.size();
460
461 auto newValue =
462 static_cast<wideint_t>(this->memory[prevIdx]) + 1;
463
464 if (newValue <= digitMax) {
465 this->memory[prevIdx] = static_cast<splitint_t>(newValue);
466 break;
467 } else {
468 this->memory[prevIdx] = static_cast<splitint_t>(digitMin);
469 nextSlice--;
470 }
471 }
472 }
473
474 // Store the slice as a signed (b+1)-bit value:
475 // [0, 2^b-1] -> unchanged
476 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
477 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
478 if (signedDigit >= cutoff) {
479 signedDigit -= base;
480 }
481
482 this->memory[
483 i * iStride + j * jStride + slice * this->matrix.size()
484 ] = static_cast<splitint_t>(signedDigit);
485
486 shiftCounter -= (bitsPerSlice + 1);
487 }
488 }
489 }
490 }
491 }

502 void computeSplitsWithRoundToNearest() {
503 this->splitType = splittingStrategy::roundToNearest;
504 auto bitsPerSlice = this->bitsPerSlice;
505 auto iStride = this->iStride();
506 auto jStride = this->jStride();
507 auto localMatrix = this->matrix;
508 for (size_t slice = 0; slice < numSplits; slice++) {
509 for (size_t i = 0; i < this->otherDimension(); i++) {
510 fp_t sigma = ldexp(0.75, computeNumFracBits<fp_t>() - bitsPerSlice * slice + 1 - bitsPerSlice) * powersVector[i];
511 for (size_t j = 0; j < this->innerProductDimension(); j++) {
512 auto value = (localMatrix[i * iStride + j * jStride] + sigma);
513 value -= sigma;
514 localMatrix[i * iStride + j * jStride] -= value;
515 value = value / powersVector[i] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
516 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = value;
517 }
518 }
519 }
520 }

568template <typename splitint_t, typename accumulator_t, typename fp_t>
569std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
570 const MatrixSplit<splitint_t, fp_t> &B,
571 const size_t numDiagonals) {
572
573 std::vector<fp_t > C (A.m * B.n);
574
575 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
576 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
577 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
578 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
579 std::vector<accumulator_t> accumulator (A.m * B.n, 0.0);
580 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, Aindex, Bindex);
581 for (size_t i = 0; i < A.m; i++) {
582 for (size_t j = 0; j < B.n; j++) {
583 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
584 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[i + j * A.m]), -totalShift);
585 fp_t scalingFactor = A.powersVector[i] * B.powersVector[j];
586 C[i + j * A.m] += scaledSum * scalingFactor;
587 }
588 }
589 Aindex--;
590 Bindex++;
591 }
592 }
593
594 return C;
595}

617template <typename splitint_t, typename accumulator_t, typename fp_t>
618std::vector<fp_t> computeProductsWithIntegerAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
619 const MatrixSplit<splitint_t, fp_t> &B,
620 const size_t numDiagonals) {
621 std::vector<fp_t> C(A.m * B.n);
622
623 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
624 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
625 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
626
627 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
628
629 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
630 std::vector<accumulator_t> accumulator(A.m * B.n, 0);
631 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
632 // assert(A.sliceBitOffset(static_cast<size_t>(Aindex)) + B.sliceBitOffset(Bindex) == totalShift);
633
634 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, Aindex, Bindex);
635 Aindex--;
636 Bindex++;
637 }
638
639 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
640 for (size_t i = 0; i < A.m; i++) {
641 for (size_t j = 0; j < B.n; j++) {
642 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[i + j * A.m]), -totalShift);
643 fp_t scalingFactor = A.powersVector[i] * B.powersVector[j];
644 C[i + j * A.m] += scaledSum * scalingFactor;
645 }
646 }
647 }
648
649 return C;
650}

669template <typename fp_t, typename splitint_t, typename accumulator_t>
670std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const std::vector<fp_t> &B,
671 const size_t m, const size_t k, const size_t n,
672 const size_t numSplitsA, const size_t numSplitsB,
673 const splittingStrategy splitType = splittingStrategy::roundToNearest,
674 const accumulationStrategy accType = accumulationStrategy::floatingPoint,
675 const multiplicationStrategy multType = multiplicationStrategy::reduced) {
676
677 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
678 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
679 assert(bitsPerInteger <= bitsInAccumulator / 2);
680 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
681 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
682
683 auto splitA = MatrixSplit<splitint_t, fp_t>(A, m, k, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
684 auto splitB = MatrixSplit<splitint_t, fp_t>(B, k, n, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
685
686 size_t numDiagonals;
687 switch (multType) {
688 case multiplicationStrategy::reduced:
689 // Products below the main anti-diagonal are ignored.
690 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
691 break;
692 case multiplicationStrategy::full:
693 // All products are computed.
694 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
695 break;
696 default:
697 UNREACHABLE();
698 }
699
700 switch (accType) {
701 case accumulationStrategy::floatingPoint:
702 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, numDiagonals);
703 case accumulationStrategy::integer:
704 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, numDiagonals);
705 default:
706 UNREACHABLE();
707 }
708}
