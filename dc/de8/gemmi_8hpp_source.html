414template <typename fp_t, typename splitint_t, typename accumulator_t>
415DerivedParameters deriveParameters(MatrixView<const fp_t> A,
416 MatrixView<const fp_t> B,
417 const config& config) {
418
419 // Compile-time type checks.
420 static_assert(std::is_floating_point_v<fp_t>,
421 "fp_t must be a floating-point type");
422 static_assert(std::is_integral_v<splitint_t> && std::is_signed_v<splitint_t>,
423 "splitint_t must be a signed integer type");
424 static_assert(std::is_integral_v<accumulator_t> && std::is_signed_v<accumulator_t>,
425 "accumulator_t must be a signed integer type");
426
427 // Matrix checks.
428 if (A.data == nullptr)
429 throw std::invalid_argument("Matrix A has a null data pointer");
430 if (B.data == nullptr)
431 throw std::invalid_argument("Matrix B has a null data pointer");
432 if (A.empty())
433 throw std::invalid_argument("Matrix A is empty (rows or cols is 0)");
434 if (B.empty())
435 throw std::invalid_argument("Matrix B is empty (rows or cols is 0)");
436
437 if (A.cols != B.rows)
438 throw std::invalid_argument(
439 "Dimension mismatch: A.cols (" + std::to_string(A.cols) +
440 ") != B.rows (" + std::to_string(B.rows) + ")");
441
442 // Split counts.
443 if (config.numSplitsA == 0)
444 throw std::invalid_argument("numSplitsA must be >= 1");
445 if (config.numSplitsB == 0)
446 throw std::invalid_argument("numSplitsB must be >= 1");
447
448 // Custom mask size.
449 if (std::holds_alternative<std::vector<bool>>(config.multSpecification)) {
450 const auto& mask = std::get<std::vector<bool>>(config.multSpecification);
451 const size_t expected = config.numSplitsA * config.numSplitsB;
452 if (mask.size() != expected)
453 throw std::invalid_argument(
454 "Custom mask size (" + std::to_string(mask.size()) +
455 ") does not match numSplitsA * numSplitsB (" +
456 std::to_string(expected) + ")");
457 }
458
459 // Type width constraints.
460 constexpr size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
461 constexpr size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
462 if (bitsPerInteger > bitsInAccumulator / 2)
463 throw std::invalid_argument(
464 "splitint_t (" + std::to_string(bitsPerInteger) + " bits) is too wide "
465 "for accumulator_t (" + std::to_string(bitsInAccumulator) + " bits): "
466 "require bitsPerInteger <= bitsInAccumulator / 2");
467
468 const size_t k = A.cols;
469 const double log2k = (k > 1) ? std::log2(static_cast<double>(k)) : 0.0;
470 const auto alpha = static_cast<size_t>(
471 std::max(0.0, std::floor((static_cast<double>(bitsInAccumulator) - log2k) / 2.0)));
472 const size_t bitsPerSlice = std::min(bitsPerInteger, alpha);
473 if (bitsPerSlice == 0)
474 throw std::invalid_argument(
475 "Computed bitsPerSlice is 0: inner dimension k=" + std::to_string(k) +
476 " is too large for accumulator_t (" +
477 std::to_string(bitsInAccumulator) + " bits)");
478
479 return DerivedParameters{bitsPerSlice};
480}

633template <typename splitint_t, typename fp_t>
634void computeNormalisationVectors(preparedOperand<splitint_t, fp_t>& operand) {
635 // Compute normalisation vector.
636 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
637 operand.powersVector[outer] = 0.0;
638 for (size_t j = 0; j < operand.innerDimension(); j++) {
639 operand.powersVector[outer] = std::max(operand.powersVector[outer],
640 std::abs(operand.operand(outer, j)));
641 }
642 // Compute the smallest power of 2 that is strictly greater than the
643 // maximum value in the row/column.
644 // NOTE 1: This is not the technique used in uoi24.
645 // NOTE 2: I use exponents instead of powers of 2, as I need the former
646 // to shift correctly.
647 operand.scalingExponents[outer] = getStoredFloatingPointExponent(operand.powersVector[outer]);
648 operand.powersVector[outer] = std::ldexp(1.0, operand.scalingExponents[outer]);
649 }
650}

665template <typename splitint_t, typename fp_t>
666void computeFixedPointRepresentationVector(std::vector<typename FloatingPointTraits<fp_t>::StorageType> &fraction,
667 std::vector<bool> &sign, size_t outer,
668 const preparedOperand<splitint_t, fp_t>& operand) {
669 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
670 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
671 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
672 fp_t value = operand.operand(outer, inner);
673 fraction[inner] = std::bit_cast<uint_t>(value);
674 sign[inner] = std::signbit(value);
675 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
676 fraction[inner] = fraction[inner] & bitmask;
677 // Restore implicit bit for normal numbers.
678 // NOTE: NaNs and infs are currently not supported.
679 if (std::fpclassify(value) == FP_NORMAL)
680 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
681 }
682}

697template <typename splitint_t, typename fp_t>
698void computeSplitsWithTruncation(preparedOperand<splitint_t, fp_t>& operand) {
699 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
700
701 // Compute splits one row/column at a time.
702 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
703 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
704 std::vector<uint_t> fraction (operand.innerDimension());
705 std::vector<bool> sign (operand.innerDimension());
706 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
707 // Get binary representation of significands of normalised row/column.
708 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
709
710 // Create bitmask.
711 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
712
713 // Perform the split.
714 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
715 // NOTE: I could have a special path for 0.
716 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
717 int currentExponent = getStoredFloatingPointExponent(operand.operand(outer, inner));
718 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
719 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
720 if (exponentDifference > (signed)bitsPerSlice) {
721 exponentDifference -= bitsPerSlice;
722 } else {
723 shiftCounter += exponentDifference;
724 exponentDifference = 0;
725 uint_t bitmask = shiftCounter > 0 ?
726 smallBitmask << shiftCounter :
727 smallBitmask >> -shiftCounter;
728 uint_t currentSlice = fraction[inner] & bitmask;
729 uint_t currentSplit = shiftCounter > 0 ?
730 currentSlice >> shiftCounter :
731 currentSlice << -shiftCounter;
732 splitint_t value = static_cast<splitint_t>(currentSplit) * (sign[inner] ? -1 : 1);
733 operand.splitValue(outer, inner, slice) = value;
734 shiftCounter -= bitsPerSlice;
735 }
736 }
737 }
738 }
739}

756template <typename splitint_t, typename fp_t>
757void computeSplitsWithUnsignedEncoding(preparedOperand<splitint_t, fp_t>& operand) {
758 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
759 using wideint_t = std::conditional_t<(sizeof(splitint_t) < sizeof(int)), int, std::intmax_t>;
760
761 // Compute splits one row/column at a time.
762 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
763 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
764 std::vector<uint_t> fraction (operand.innerDimension());
765 std::vector<bool> sign (operand.innerDimension());
766 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
767 // Get binary representation of significands of normalised row/column.
768 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
769
770 // Create bitmasks.
771 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
772 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
773 // is to keep the algorithm consistent with the bitmasking approach
774 // above.
775 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
776 // is not mentioned in sabb26, but it seems necessary to do this to
777 // ensure that the first (signed) slice does not overflow if the full
778 // bitwidth of splitint_t is used and the second slice is negative in
779 // two's complement.
780 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
781 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
782
783 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
784
785 auto matrixIndex = operand.operandIndex(outer, inner);
786
787 // NOTE: I could have a special path for 0.
788 int16_t shiftCounter;
789 int currentExponent = getStoredFloatingPointExponent(operand.operand(outer, inner));
790 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
791
792 splitint_t value = 0;
793 uint_t remainder = 0;
794
795 // Slice 0.
796 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
797 // Value is too small to contribute explicit bits to slice 0.
798 value = sign[inner] ? -1 : 0;
799 remainder = sign[inner] ?
800 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
801 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
802 fraction[inner];
803 exponentDifference -= (bitsPerSlice - 1);
804 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
805 } else {
806 // Truncation.
807 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
808 exponentDifference = 0;
809 uint_t bitmask = (shiftCounter > 0)
810 ? (smallBitmask << shiftCounter)
811 : (smallBitmask >> -shiftCounter);
812 uint_t currentSlice = fraction[inner] & bitmask;
813 uint_t currentSplit = (shiftCounter > 0)
814 ? (currentSlice >> shiftCounter)
815 : (currentSlice << -shiftCounter);
816
817 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
818
819 // Rounding.
820 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
821 uint_t lowBits = fraction[inner] & lowMask;
822
823 if (lowBits != 0) {
824 value += (sign[inner] ? -1 : 0);
825 }
826 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
827 shiftCounter -= (bitsPerSlice + 1);
828 }
829 operand.memory[matrixIndex] = value;
830
831 // Remaining slices.
832 const auto width = bitsPerSlice + 1;
833 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
834 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
835 const auto digitMax = cutoff - 1; // 2^b - 1
836 const auto digitMin = -cutoff; // -2^b
837 for (size_t slice = 1; slice < operand.prepConfig.numSplits; slice++) {
838 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
839 exponentDifference -= (bitsPerSlice + 1);
840 if (sign[inner]) {
841 operand.splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
842 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
843 }
844 } else {
845 shiftCounter += exponentDifference;
846 exponentDifference = 0;
847
848 uint_t bitmask = shiftCounter > 0 ?
849 largeBitmask << shiftCounter :
850 largeBitmask >> -shiftCounter;
851
852 uint_t currentSlice = remainder & bitmask;
853 uint_t currentSplit = shiftCounter > 0 ?
854 currentSlice >> shiftCounter :
855 currentSlice << -shiftCounter;
856
857 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
858 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
859 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
860 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
861 currentSplit |= maskToAdd;
862 }
863
864 // Update previous slices if currentSplit is negative in splitint_t.
865 // Carry must propoagate upward until slice 0 if necessary.
866 if (currentSplit > static_cast<uint_t>(digitMax)) {
867 int prevSliceIndex = static_cast<int>(slice) - 1;
868 while (true) {
869 auto newValue = static_cast<wideint_t>(operand.splitValue(outer, inner, prevSliceIndex)) + 1;
870 if (newValue <= digitMax) {
871 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
872 break;
873 } else {
874 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
875 prevSliceIndex--;
876 }
877 }
878 }
879
880 // Store the slice as a signed (b+1)-bit value:
881 // [0, 2^b-1] -> unchanged
882 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
883 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
884 if (signedDigit >= cutoff) {
885 signedDigit -= base;
886 }
887
888 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
889 shiftCounter -= (bitsPerSlice + 1);
890 }
891 }
892 }
893 }

909template <typename splitint_t, typename fp_t>
910void computeSplitsWithRoundToNearest(preparedOperand<splitint_t, fp_t>& operand) {
911 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
912 auto localMatrix = std::vector<fp_t>(operand.matrix.data, operand.matrix.data + operand.matrix.size());
913 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
914 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
915 // Compute exponent in signed arithmetic to avoid wraparound when
916 // bitsPerSlice * (slice + 1) approaches numSignificandBits.
917 int exponent = static_cast<int>(FloatingPointTraits<fp_t>::numSignificandBits) - static_cast<int>(bitsPerSlice * (slice + 1)) + 1;
918 fp_t sigma = std::ldexp(0.75, exponent) * operand.powersVector[outer];
919 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
920 auto matrixIndex = operand.operandIndex(outer, inner);
921 auto value = (localMatrix[matrixIndex] + sigma);
922 value -= sigma;
923 localMatrix[matrixIndex] -= value;
924 value = value / operand.powersVector[outer] * std::ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
925 operand.splitValue(outer, inner, slice) = value;
926 }
927 }
928 }
929}

936template <typename splitint_t, typename fp_t>
937preparedOperand<splitint_t, fp_t> prepareOperand(MatrixView<const fp_t> matrix,
938 const OperandPreparationConfig& prepConfig) {
939 preparedOperand<splitint_t, fp_t> operand;
940 operand.matrix = matrix;
941 operand.prepConfig = prepConfig;
942 operand.memory.resize(operand.matrix.size() * prepConfig.numSplits);
943 operand.powersVector.resize(operand.prepConfig.dimension == normalisationDimension::byRows ?
944 operand.rows() : operand.cols());
945 operand.scalingExponents.resize(operand.powersVector.size());
946
947 computeNormalisationVectors(operand);
948
949 switch (prepConfig.splitType) {
950 case splittingStrategy::truncation:
951 computeSplitsWithTruncation(operand);
952 break;
953 case splittingStrategy::unsignedEncoding:
954 computeSplitsWithUnsignedEncoding(operand);
955 break;
956 case splittingStrategy::roundToNearest:
957 computeSplitsWithRoundToNearest(operand);
958 break;
959 }
960
961 return operand;
962}

974 struct multiplicationSchedule {
975 size_t numSplitsA; // number of slices for A
976 size_t numSplitsB; // number of slices for B
977
978 std::vector<bool> mask;
979
985 multiplicationSchedule(size_t numSplitsA, size_t numSplitsB) :
986 numSplitsA(numSplitsA), numSplitsB(numSplitsB), mask(numSplitsA * numSplitsB, false) {}
987
988 inline std::vector<bool>::reference operator()(size_t i, size_t j) {
989 return mask[i * numSplitsB + j];
990 }
991 inline bool operator()(size_t i, size_t j) const {
992 return mask[i * numSplitsB + j];
993 }
994};
995
1006inline multiplicationSchedule makeSchedule(const config& config) {
1007 multiplicationSchedule schedule(config.numSplitsA, config.numSplitsB);
1008
1009 std::visit([&](auto&& spec) {
1010 using T = std::decay_t<decltype(spec)>;
1011
1012 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
1013 if (spec == multiplicationStrategy::full) {
1014 std::fill(schedule.mask.begin(), schedule.mask.end(), true);
1015 } else {
1016 size_t limit = std::max(config.numSplitsA, config.numSplitsB) - 1;
1017 for (size_t i = 0; i < config.numSplitsA; ++i)
1018 for (size_t j = 0; j < config.numSplitsB; ++j)
1019 schedule(i, j) = i + j <= limit;
1020 }

1102 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1103 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];

1105 }
1106 }
1107 }
1108 Aindex--;
1109 Bindex++;
1110 }

1137std::vector<fp_t> computeProductsWithIntegerAccumulation(const preparedOperand<splitint_t, fp_t> &A,
1138 const preparedOperand<splitint_t, fp_t> &B,

1157 Bindex++;

1161 for (size_t row = 0; row < A.rows(); row++) {
1162 for (size_t col = 0; col < B.cols(); col++) {

1165 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1166 C[index] += scaledSum * scalingFactor;
1167 }
1168 }
1169 }
