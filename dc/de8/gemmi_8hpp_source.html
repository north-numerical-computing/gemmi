432 void computeNormalisationVectors() {
433 // Compute normalisation vector.
434 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
435 this->powersVector[outer] = 0.0;
436 for (size_t j = 0; j < this->innerDimension(); j++) {
437 this->powersVector[outer] = std::max(this->powersVector[outer],
438 std::abs(this->operand(outer, j)));
439 }
440 // Compute the smallest power of 2 that is strictly greater than the
441 // maximum value in the row/column.
442 // NOTE 1: This is not the technique used in uoi24.
443 // NOTE 2: I use exponents instead of powers of 2, as I need the former
444 // to shift correctly.
445 this->scalingExponents[outer] = getStoredFloatingPointExponent(this->powersVector[outer]);
446 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);
447 }

455 int computeSliceBitOffset(size_t slice) const {
456 switch (splitType) {
457 case splittingStrategy::truncation:
458 // Slice k -> k * b
459 return static_cast<int>((slice + 1) * bitsPerSlice);
460
461 case splittingStrategy::roundToNearest:
462 // Slice 0 -> b - 1
463 // Slice k -> k * b - 1
464 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
465
466 case splittingStrategy::unsignedEncoding:
467 // Slice 0 -> b - 1
468 // Slice k -> (b - 1) + k * (b + 1)
469 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));

472 default:
473 throw std::logic_error("Unhandled splittingStrategy");

485 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
486 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
487 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
488 fp_t value = this->operand(outer, inner);
489 fraction[inner] = std::bit_cast<uint_t>(value);
490 sign[inner] = std::signbit(value);
491 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
492 fraction[inner] = fraction[inner] & bitmask;
493 // Restore implicit bit for normal numbers.
494 // NOTE: NaNs and infs are currently not supported.
495 if (std::fpclassify(value) == FP_NORMAL)
496 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
497 }
498 }

509 void computeSplitsWithTruncation() {
510 this->splitType = splittingStrategy::truncation;
511 // Compute splits one row/column at a time.
512 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
513 auto bitsPerSlice = this->bitsPerSlice;
514 std::vector<uint_t> fraction (this->innerDimension());
515 std::vector<bool> sign (this->innerDimension());
516 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
517 // Get binary representation of significands of normalised row/column.
518 computeFixedPointRepresentationVector(fraction, sign, outer);
519
520 // Create bitmask.
521 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
522
523 // Perform the split.
524 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
525 // NOTE: I could have a special path for 0.
526 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
527 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
528 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
529 for (size_t slice = 0; slice < numSplits; slice++) {
530 if (exponentDifference > (signed)bitsPerSlice) {
531 exponentDifference -= bitsPerSlice;
532 } else {
533 shiftCounter += exponentDifference;
534 exponentDifference = 0;
535 uint_t bitmask = shiftCounter > 0 ?
536 smallBitmask << shiftCounter :
537 smallBitmask >> -shiftCounter;
538 uint_t currentSlice = fraction[inner] & bitmask;
539 uint_t currentSplit = shiftCounter > 0 ?
540 currentSlice >> shiftCounter :
541 currentSlice << -shiftCounter;
542 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
543 this->splitValue(outer, inner, slice) = value;
544 shiftCounter -= bitsPerSlice;
545 }
546 }
547 }
548 }

562 void computeSplitsWithUnsignedEncoding() {
563 this->splitType = splittingStrategy::unsignedEncoding;
564 // Compute splits one row/column at a time.
565 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
566 auto bitsPerSlice = this->bitsPerSlice;
567 std::vector<uint_t> fraction (this->innerDimension());
568 std::vector<bool> sign (this->innerDimension());
569 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
570 // Get binary representation of significands of normalised row/column.
571 computeFixedPointRepresentationVector(fraction, sign, outer);
572
573 // Create bitmasks.
574 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
575 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
576 // is to keep the algorithm consistent with the bitmasking approach
577 // above.
578 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
579 // is not mentioned in sabb26, but it seems necessary to do this to
580 // ensure that the first (signed) slice does not overflow if the full
581 // bitwidth of splitint_t is used and the second slice is negative in
582 // two's complement.
583 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
584 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
585
586 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
587
588 auto matrixIndex = operandIndex(outer, inner);
589
590 // NOTE: I could have a special path for 0.
591 int16_t shiftCounter;
592 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
593 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
594
595 splitint_t value = 0;
596 uint_t remainder = 0;
597
598 // Slice 0.
599 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
600 // Value is too small to contribute explicit bits to slice 0.
601 value = sign[inner] ? -1 : 0;
602 remainder = sign[inner] ?
603 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
604 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
605 fraction[inner];
606 exponentDifference -= (bitsPerSlice - 1);
607 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
608 } else {
609 // Truncation.
610 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
611 exponentDifference = 0;
612 uint_t bitmask = (shiftCounter > 0)
613 ? (smallBitmask << shiftCounter)
614 : (smallBitmask >> -shiftCounter);
615 uint_t currentSlice = fraction[inner] & bitmask;
616 uint_t currentSplit = (shiftCounter > 0)
617 ? (currentSlice >> shiftCounter)
618 : (currentSlice << -shiftCounter);
619
620 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
621
622 // Rounding.
623 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
624 uint_t lowBits = fraction[inner] & lowMask;
625
626 if (lowBits != 0) {
627 value += (sign[inner] ? -1 : 0);
628 }
629 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
630 shiftCounter -= (bitsPerSlice + 1);
631 }
632 this->memory[matrixIndex] = value;
633
634 // Remaining slices.
635 const auto width = bitsPerSlice + 1;
636 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
637 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
638 const auto digitMax = cutoff - 1; // 2^b - 1
639 const auto digitMin = -cutoff; // -2^b
640 for (size_t slice = 1; slice < numSplits; slice++) {
641 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
642 exponentDifference -= (bitsPerSlice + 1);
643 if (sign[inner]) {
644 this->splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
645 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
646 }
647 } else {
648 shiftCounter += exponentDifference;
649 exponentDifference = 0;
650
651 uint_t bitmask = shiftCounter > 0 ?
652 largeBitmask << shiftCounter :
653 largeBitmask >> -shiftCounter;
654
655 uint_t currentSlice = remainder & bitmask;
656 uint_t currentSplit = shiftCounter > 0 ?
657 currentSlice >> shiftCounter :
658 currentSlice << -shiftCounter;
659
660 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
661 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
662 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
663 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
664 currentSplit |= maskToAdd;
665 }
666
667 // Update previous slices if currentSplit is negative in splitint_t.
668 // Carry must propoagate upward until slice 0 if necessary.
669 if (currentSplit > static_cast<uint_t>(digitMax)) {
670 int prevSliceIndex = static_cast<int>(slice) - 1;
671 while (true) {
672 auto newValue = static_cast<wideint_t>(this->splitValue(outer, inner, prevSliceIndex)) + 1;
673 if (newValue <= digitMax) {
674 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
675 break;
676 } else {
677 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
678 prevSliceIndex--;
679 }
680 }
681 }
682
683 // Store the slice as a signed (b+1)-bit value:
684 // [0, 2^b-1] -> unchanged
685 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
686 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
687 if (signedDigit >= cutoff) {
688 signedDigit -= base;
689 }
690
691 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
692 shiftCounter -= (bitsPerSlice + 1);
693 }
694 }
695 }
696 }
697 }

708 void computeSplitsWithRoundToNearest() {
709 this->splitType = splittingStrategy::roundToNearest;
710 auto bitsPerSlice = this->bitsPerSlice;
711 auto localMatrix = std::vector<fp_t>(this->matrix.data, this->matrix.data + this->matrix.size());
712 for (size_t slice = 0; slice < numSplits; slice++) {
713 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
714 fp_t sigma = ldexp(0.75,
715 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
716 powersVector[outer];
717 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
718 auto matrixIndex = operandIndex(outer, inner);
719 auto value = (localMatrix[matrixIndex] + sigma);
720 value -= sigma;
721 localMatrix[matrixIndex] -= value;
722 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
723 this->splitValue(outer, inner, slice) = value;
724 }
725 }
726 }
727 }

796template <typename splitint_t, typename accumulator_t, typename fp_t>
797std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
798 const MatrixSplit<splitint_t, fp_t> &B,
799 const matrixLayout layoutC,
800 const size_t numDiagonals) {
801 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
802 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
803 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
804 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
805 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
806 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
807 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
808 for (size_t row = 0; row < A.rows(); row++) {
809 for (size_t col = 0; col < B.cols(); col++) {
810 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
811 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
812 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
813 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
814 C[index] += scaledSum * scalingFactor;
815 }
816 }
817 Aindex--;
818 Bindex++;
819 }
820 }
821
822 return C;
823}

845template <typename splitint_t, typename accumulator_t, typename fp_t>
846std::vector<fp_t> computeProductsWithIntegerAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
847 const MatrixSplit<splitint_t, fp_t> &B,
848 const matrixLayout layoutC,
849 const size_t numDiagonals) {
850
851 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
852 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
853 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
854 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
855
856 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
857
858 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
859 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
860 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
861 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
862 Aindex--;
863 Bindex++;
864 }
865
866 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
867 for (size_t row = 0; row < A.rows(); row++) {
868 for (size_t col = 0; col < B.cols(); col++) {
869 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
870 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
871 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
872 C[index] += scaledSum * scalingFactor;
873 }
874 }
875 }
876
877 return C;
878}

899template <typename fp_t, typename splitint_t, typename accumulator_t>
900std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const matrixLayout layoutA,
901 const std::vector<fp_t> &B, const matrixLayout layoutB,
902 const size_t m, const size_t k, const size_t n,
903 const size_t numSplitsA, const size_t numSplitsB,
904 const matrixLayout layoutC = matrixLayout::columnMajor,
905 const multiterm::splittingStrategy splitType = multiterm::splittingStrategy::roundToNearest,
906 const multiterm::reductionStrategy accType = multiterm::reductionStrategy::floatingPoint,
907 const multiterm::multiplicationStrategy multType = multiterm::multiplicationStrategy::reduced) {
908
909 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
910 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
911 if (bitsPerInteger > bitsInAccumulator / 2) {
912 throw std::invalid_argument("Split integer type is too wide for the chosen accumulator type");
913 }
914 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
915 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
916
917 auto viewA = makeMatrixView(A, m, k, layoutA);
918 auto viewB = makeMatrixView(B, k, n, layoutB);
919
920 auto splitA = multiterm::MatrixSplit<splitint_t, fp_t>(viewA, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
921 auto splitB = multiterm::MatrixSplit<splitint_t, fp_t>(viewB, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
922
923 splitA.prepare();
924 splitB.prepare();
925
926 size_t numDiagonals;
927 switch (multType) {
928 case multiterm::multiplicationStrategy::reduced:
929 // Products below the main anti-diagonal are ignored.
930 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
931 break;
932 case multiterm::multiplicationStrategy::full:
933 // All products are computed.
934 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
935 break;
936 // LCOV_EXCL_START
937 default:
938 throw std::logic_error("Unhandled multiplicationStrategy");
939 // LCOV_EXCL_STOP
940 }
941
942 switch (accType) {
943 case multiterm::reductionStrategy::floatingPoint:
