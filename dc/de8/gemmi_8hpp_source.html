214 void computeNormalisationVectors() {
215 // Compute normalisation vector.
216 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
217 this->powersVector[outer] = 0.0;
218 for (size_t j = 0; j < this->innerDimension(); j++) {
219 const size_t matrixIndex = operandIndex(outer, j);
220 this->powersVector[outer] = std::max(this->powersVector[outer],
221 std::abs(this->matrix[matrixIndex]));
222 }
223 // Compute the smallest power of 2 that is strictly greater than the
224 // maximum value in the row/column.
225 // NOTE 1: This is not the technique used in uoi24.
226 // NOTE 2: I use exponents instead of powers of 2, as I need the former
227 // to shift correctly.
228 this->scalingExponents[outer] = getStoredFloatingPointExponent(this->powersVector[outer]);
229 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);
230 }
231 }

238 int computeSliceBitOffset(size_t slice) const {
239 switch (splitType) {
240 case splittingStrategy::truncation:
241 // Slice k -> k * b
242 return static_cast<int>((slice + 1) * bitsPerSlice);
243
244 case splittingStrategy::roundToNearest:
245 // Slice 0 -> b - 1
246 // Slice k -> k * b - 1
247 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
248
249 case splittingStrategy::unsignedEncoding:
250 // Slice 0 -> b - 1
251 // Slice k -> (b - 1) + k * (b + 1)
252 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));

268 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
269 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
270 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
271 const size_t matrixIndex = operandIndex(outer, inner);
272 fp_t value = this->matrix[matrixIndex];
273 fraction[inner] = std::bit_cast<uint_t>(value);
274 sign[inner] = std::signbit(value);
275 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
276 fraction[inner] = fraction[inner] & bitmask;
277 // Restore implicit bit for normal numbers.
278 // NOTE: NaNs and infs are currently not supported.
279 if (std::fpclassify(value) == FP_NORMAL)
280 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
281 }

293 void computeSplitsWithTruncation() {
294 this->splitType = splittingStrategy::truncation;
295 // Compute splits one row/column at a time.
296 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
297 auto bitsPerSlice = this->bitsPerSlice;
298 std::vector<uint_t> fraction (this->innerDimension());
299 std::vector<bool> sign (this->innerDimension());
300 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
301 // Get binary representation of significands of normalised row/column.
302 computeFixedPointRepresentationVector(fraction, sign, outer);
303
304 // Create bitmask.
305 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
306
307 // Perform the split.
308 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
309 // NOTE: I could have a special path for 0.
310 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
311 int currentExponent = getStoredFloatingPointExponent(this->matrix[operandIndex(outer, inner)]);
312 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
313 for (size_t slice = 0; slice < numSplits; slice++) {
314 if (exponentDifference > (signed)bitsPerSlice) {
315 exponentDifference -= bitsPerSlice;
316 } else {
317 shiftCounter += exponentDifference;
318 exponentDifference = 0;
319 uint_t bitmask = shiftCounter > 0 ?
320 smallBitmask << shiftCounter :
321 smallBitmask >> -shiftCounter;
322 uint_t currentSlice = fraction[inner] & bitmask;
323 uint_t currentSplit = shiftCounter > 0 ?
324 currentSlice >> shiftCounter :
325 currentSlice << -shiftCounter;
326 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
327 this->memory[splitIndex(outer, inner, slice)] = value;
328 shiftCounter -= bitsPerSlice;
329 }
330 }
331 }
332 }

346 void computeSplitsWithUnsignedEncoding() {
347 this->splitType = splittingStrategy::unsignedEncoding;
348 // Compute splits one row/column at a time.
349 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
350 auto bitsPerSlice = this->bitsPerSlice;
351 std::vector<uint_t> fraction (this->innerDimension());
352 std::vector<bool> sign (this->innerDimension());
353 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
354 // Get binary representation of significands of normalised row/column.
355 computeFixedPointRepresentationVector(fraction, sign, outer);
356
357 // Create bitmasks.
358 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
359 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
360 // is to keep the algorithm consistent with the bitmasking approach
361 // above.
362 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
363 // is not mentioned in sabb26, but it seems necessary to do this to
364 // ensure that the first (signed) slice does not overflow if the full
365 // bitwidth of splitint_t is used and the second slice is negative in
366 // two's complement.
367 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
368 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
369
370 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
371
372 auto matrixIndex = operandIndex(outer, inner);
373
374 // NOTE: I could have a special path for 0.
375 int16_t shiftCounter;
376 int currentExponent = getStoredFloatingPointExponent(this->matrix[matrixIndex]);
377 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
378
379 splitint_t value = 0;
380 uint_t remainder = 0;
381
382 // Slice 0.
383 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
384 // Value is too small to contribute explicit bits to slice 0.
385 value = sign[inner] ? -1 : 0;
386 remainder = sign[inner] ?
387 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
388 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
389 fraction[inner];
390 exponentDifference -= (bitsPerSlice - 1);
391 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
392 } else {
393 // Truncation.
394 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
395 exponentDifference = 0;
396 uint_t bitmask = (shiftCounter > 0)
397 ? (smallBitmask << shiftCounter)
398 : (smallBitmask >> -shiftCounter);
399 uint_t currentSlice = fraction[inner] & bitmask;
400 uint_t currentSplit = (shiftCounter > 0)
401 ? (currentSlice >> shiftCounter)
402 : (currentSlice << -shiftCounter);
403
404 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
405
406 // Rounding.
407 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
408 uint_t lowBits = fraction[inner] & lowMask;
409
410 if (lowBits != 0) {
411 value += (sign[inner] ? -1 : 0);
412 }
413 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
414 shiftCounter -= (bitsPerSlice + 1);
415 }
416 this->memory[matrixIndex] = value;
417
418 // Remaining slices.
419 const auto width = bitsPerSlice + 1;
420 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
421 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
422 const auto digitMax = cutoff - 1; // 2^b - 1
423 const auto digitMin = -cutoff; // -2^b
424 for (size_t slice = 1; slice < numSplits; slice++) {
425 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
426 exponentDifference -= (bitsPerSlice + 1);
427 if (sign[inner]) {
428 this->memory[splitIndex(outer, inner, slice - 1)] = static_cast<splitint_t>(0);
429 this->memory[splitIndex(outer, inner, slice)] = static_cast<splitint_t>(-1);
430 }
431 } else {
432 shiftCounter += exponentDifference;
433 exponentDifference = 0;
434
435 uint_t bitmask = shiftCounter > 0 ?
436 largeBitmask << shiftCounter :
437 largeBitmask >> -shiftCounter;
438
439 uint_t currentSlice = remainder & bitmask;
440 uint_t currentSplit = shiftCounter > 0 ?
441 currentSlice >> shiftCounter :
442 currentSlice << -shiftCounter;
443
444 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
445 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
446 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
447 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
448 currentSplit |= maskToAdd;
449 }
450
451 // Update previous slices if currentSplit is negative in splitint_t.
452 // Carry must propoagate upward until slice 0 if necessary.
453 if (currentSplit > static_cast<uint_t>(digitMax)) {
454 int nextSlice = static_cast<int>(slice) - 1;
455 while (true) {
456 const size_t prevIdx = splitIndex(outer, inner, nextSlice);
457 auto newValue = static_cast<wideint_t>(this->memory[prevIdx]) + 1;
458
459 if (newValue <= digitMax) {
460 this->memory[prevIdx] = static_cast<splitint_t>(newValue);
461 break;
462 } else {
463 this->memory[prevIdx] = static_cast<splitint_t>(digitMin);
464 nextSlice--;
465 }
466 }
467 }
468
469 // Store the slice as a signed (b+1)-bit value:
470 // [0, 2^b-1] -> unchanged
471 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
472 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
473 if (signedDigit >= cutoff) {
474 signedDigit -= base;
475 }
476
477 this->memory[splitIndex(outer, inner, slice)] = static_cast<splitint_t>(signedDigit);
478 shiftCounter -= (bitsPerSlice + 1);
479 }
480 }
481 }
482 }
483 }

494 void computeSplitsWithRoundToNearest() {
495 this->splitType = splittingStrategy::roundToNearest;
496 auto bitsPerSlice = this->bitsPerSlice;
497 auto localMatrix = this->matrix;
498 for (size_t slice = 0; slice < numSplits; slice++) {
499 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
500 fp_t sigma = ldexp(0.75,
501 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
502 powersVector[outer];
503 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
504 auto matrixIndex = operandIndex(outer, inner);
505 auto value = (localMatrix[matrixIndex] + sigma);
506 value -= sigma;
507 localMatrix[matrixIndex] -= value;
508 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
509 this->memory[splitIndex(outer, inner, slice)] = value;
510 }
511 }
512 }
513 }

563template <typename splitint_t, typename accumulator_t, typename fp_t>
564std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
565 const MatrixSplit<splitint_t, fp_t> &B,
566 const matrixLayout layoutC,
567 const size_t numDiagonals) {
568 std::vector<fp_t> C (A.m * B.n, 0.0);
569 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
570 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
571 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
572 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
573 std::vector<accumulator_t> accumulator (A.m * B.n, 0.0);
574 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
575 for (size_t row = 0; row < A.m; row++) {
576 for (size_t col = 0; col < B.n; col++) {
577 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
578 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.m) : (col + row * B.n);
579 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
580 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
581 C[index] += scaledSum * scalingFactor;
582 }
583 }
584 Aindex--;
585 Bindex++;
586 }
587 }
588
589 return C;
590}

612template <typename splitint_t, typename accumulator_t, typename fp_t>
613std::vector<fp_t> computeProductsWithIntegerAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
614 const MatrixSplit<splitint_t, fp_t> &B,
615 const matrixLayout layoutC,
616 const size_t numDiagonals) {
617
618 std::vector<fp_t> C (A.m * B.n, 0.0);
619 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
620 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
621 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
622
623 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
624
625 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
626 std::vector<accumulator_t> accumulator(A.m * B.n, 0);
627 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
628 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
629 Aindex--;
630 Bindex++;
631 }
632
633 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
634 for (size_t row = 0; row < A.m; row++) {
635 for (size_t col = 0; col < B.n; col++) {
636 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.m) : (col + row * B.n);
637 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
638 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
639 C[index] += scaledSum * scalingFactor;
640 }
641 }
642 }
643
644 return C;
645}

664template <typename fp_t, typename splitint_t, typename accumulator_t>
665std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const matrixLayout layoutA,
666 const std::vector<fp_t> &B, const matrixLayout layoutB,
667 const size_t m, const size_t k, const size_t n,
668 const size_t numSplitsA, const size_t numSplitsB,
669 const matrixLayout layoutC = matrixLayout::columnMajor,
670 const splittingStrategy splitType = splittingStrategy::roundToNearest,
671 const accumulationStrategy accType = accumulationStrategy::floatingPoint,
672 const multiplicationStrategy multType = multiplicationStrategy::reduced) {
673
674 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
675 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
676 if (bitsPerInteger > bitsInAccumulator / 2) {
677 throw std::invalid_argument("Split integer type is too wide for the chosen accumulator type");
678 }
679 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
680 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
681
682 auto splitA = MatrixSplit<splitint_t, fp_t>(A, m, k, layoutA, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
683 auto splitB = MatrixSplit<splitint_t, fp_t>(B, k, n, layoutB, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
684
685 size_t numDiagonals;
686 switch (multType) {
687 case multiplicationStrategy::reduced:
688 // Products below the main anti-diagonal are ignored.
689 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
690 break;
691 case multiplicationStrategy::full:
692 // All products are computed.
693 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
694 break;
695 // LCOV_EXCL_START
696 default:
697 throw std::logic_error("Unhandled multiplicationStrategy");
698 // LCOV_EXCL_STOP
699 }
700
701 switch (accType) {
702 case accumulationStrategy::floatingPoint:
703 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
704 case accumulationStrategy::integer:
705 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
706 // LCOV_EXCL_START
707 default:
708 throw std::logic_error("Unhandled accumulationStrategy");
709 // LCOV_EXCL_STOP
710 }
711}
