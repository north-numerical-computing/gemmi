534 void computeNormalisationVectors() {
535 // Compute normalisation vector.
536 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
537 this->powersVector[outer] = 0.0;
538 for (size_t j = 0; j < this->innerDimension(); j++) {
539 this->powersVector[outer] = std::max(this->powersVector[outer],
540 std::abs(this->operand(outer, j)));
541 }
542 // Compute the smallest power of 2 that is strictly greater than the
543 // maximum value in the row/column.
544 // NOTE 1: This is not the technique used in uoi24.
545 // NOTE 2: I use exponents instead of powers of 2, as I need the former
546 // to shift correctly.
547 this->scalingExponents[outer] = getStoredFloatingPointExponent(this->powersVector[outer]);
548 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);
549 }

557 int computeSliceBitOffset(size_t slice) const {
558 switch (splitType) {
559 case splittingStrategy::truncation:
560 // Slice k -> k * b
561 return static_cast<int>((slice + 1) * bitsPerSlice);
562
563 case splittingStrategy::roundToNearest:
564 // Slice 0 -> b - 1
565 // Slice k -> k * b - 1
566 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
567
568 case splittingStrategy::unsignedEncoding:
569 // Slice 0 -> b - 1
570 // Slice k -> (b - 1) + k * (b + 1)
571 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));
572
573 // LCOV_EXCL_START
574 default:
575 throw std::logic_error("Unhandled splittingStrategy");
576 // LCOV_EXCL_STOP
577 }

587 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
588 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
589 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
590 fp_t value = this->operand(outer, inner);
591 fraction[inner] = std::bit_cast<uint_t>(value);
592 sign[inner] = std::signbit(value);
593 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
594 fraction[inner] = fraction[inner] & bitmask;
595 // Restore implicit bit for normal numbers.
596 // NOTE: NaNs and infs are currently not supported.
597 if (std::fpclassify(value) == FP_NORMAL)
598 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
599 }
600 }

611 void computeSplitsWithTruncation() {
612 this->splitType = splittingStrategy::truncation;
613 // Compute splits one row/column at a time.
614 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
615 auto bitsPerSlice = this->bitsPerSlice;
616 std::vector<uint_t> fraction (this->innerDimension());
617 std::vector<bool> sign (this->innerDimension());
618 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
619 // Get binary representation of significands of normalised row/column.
620 computeFixedPointRepresentationVector(fraction, sign, outer);
621
622 // Create bitmask.
623 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
624
625 // Perform the split.
626 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
627 // NOTE: I could have a special path for 0.
628 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
629 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
630 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
631 for (size_t slice = 0; slice < numSplits; slice++) {
632 if (exponentDifference > (signed)bitsPerSlice) {
633 exponentDifference -= bitsPerSlice;
634 } else {
635 shiftCounter += exponentDifference;
636 exponentDifference = 0;
637 uint_t bitmask = shiftCounter > 0 ?
638 smallBitmask << shiftCounter :
639 smallBitmask >> -shiftCounter;
640 uint_t currentSlice = fraction[inner] & bitmask;
641 uint_t currentSplit = shiftCounter > 0 ?
642 currentSlice >> shiftCounter :
643 currentSlice << -shiftCounter;
644 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
645 this->splitValue(outer, inner, slice) = value;
646 shiftCounter -= bitsPerSlice;
647 }
648 }
649 }
650 }
651 }

664 void computeSplitsWithUnsignedEncoding() {
665 this->splitType = splittingStrategy::unsignedEncoding;
666 // Compute splits one row/column at a time.
667 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
668 auto bitsPerSlice = this->bitsPerSlice;
669 std::vector<uint_t> fraction (this->innerDimension());
670 std::vector<bool> sign (this->innerDimension());
671 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
672 // Get binary representation of significands of normalised row/column.
673 computeFixedPointRepresentationVector(fraction, sign, outer);
674
675 // Create bitmasks.
676 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
677 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
678 // is to keep the algorithm consistent with the bitmasking approach
679 // above.
680 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
681 // is not mentioned in sabb26, but it seems necessary to do this to
682 // ensure that the first (signed) slice does not overflow if the full
683 // bitwidth of splitint_t is used and the second slice is negative in
684 // two's complement.
685 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
686 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
687
688 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
689
690 auto matrixIndex = operandIndex(outer, inner);
691
692 // NOTE: I could have a special path for 0.
693 int16_t shiftCounter;
694 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
695 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
696
697 splitint_t value = 0;
698 uint_t remainder = 0;
699
700 // Slice 0.
701 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
702 // Value is too small to contribute explicit bits to slice 0.
703 value = sign[inner] ? -1 : 0;
704 remainder = sign[inner] ?
705 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
706 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
707 fraction[inner];
708 exponentDifference -= (bitsPerSlice - 1);
709 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
710 } else {
711 // Truncation.
712 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
713 exponentDifference = 0;
714 uint_t bitmask = (shiftCounter > 0)
715 ? (smallBitmask << shiftCounter)
716 : (smallBitmask >> -shiftCounter);
717 uint_t currentSlice = fraction[inner] & bitmask;
718 uint_t currentSplit = (shiftCounter > 0)
719 ? (currentSlice >> shiftCounter)
720 : (currentSlice << -shiftCounter);
721
722 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
723
724 // Rounding.
725 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
726 uint_t lowBits = fraction[inner] & lowMask;
727
728 if (lowBits != 0) {
729 value += (sign[inner] ? -1 : 0);
730 }
731 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
732 shiftCounter -= (bitsPerSlice + 1);
733 }
734 this->memory[matrixIndex] = value;
735
736 // Remaining slices.
737 const auto width = bitsPerSlice + 1;
738 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
739 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
740 const auto digitMax = cutoff - 1; // 2^b - 1
741 const auto digitMin = -cutoff; // -2^b
742 for (size_t slice = 1; slice < numSplits; slice++) {
743 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
744 exponentDifference -= (bitsPerSlice + 1);
745 if (sign[inner]) {
746 this->splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
747 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
748 }
749 } else {
750 shiftCounter += exponentDifference;
751 exponentDifference = 0;
752
753 uint_t bitmask = shiftCounter > 0 ?
754 largeBitmask << shiftCounter :
755 largeBitmask >> -shiftCounter;
756
757 uint_t currentSlice = remainder & bitmask;
758 uint_t currentSplit = shiftCounter > 0 ?
759 currentSlice >> shiftCounter :
760 currentSlice << -shiftCounter;
761
762 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
763 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
764 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
765 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
766 currentSplit |= maskToAdd;
767 }
768
769 // Update previous slices if currentSplit is negative in splitint_t.
770 // Carry must propoagate upward until slice 0 if necessary.
771 if (currentSplit > static_cast<uint_t>(digitMax)) {
772 int prevSliceIndex = static_cast<int>(slice) - 1;
773 while (true) {
774 auto newValue = static_cast<wideint_t>(this->splitValue(outer, inner, prevSliceIndex)) + 1;
775 if (newValue <= digitMax) {
776 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
777 break;
778 } else {
779 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
780 prevSliceIndex--;
781 }
782 }
783 }
784
785 // Store the slice as a signed (b+1)-bit value:
786 // [0, 2^b-1] -> unchanged
787 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
788 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
789 if (signedDigit >= cutoff) {
790 signedDigit -= base;
791 }
792
793 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
794 shiftCounter -= (bitsPerSlice + 1);
795 }
796 }
797 }
798 }
799 }

810 void computeSplitsWithRoundToNearest() {
811 this->splitType = splittingStrategy::roundToNearest;
812 auto bitsPerSlice = this->bitsPerSlice;
813 auto localMatrix = std::vector<fp_t>(this->matrix.data, this->matrix.data + this->matrix.size());
814 for (size_t slice = 0; slice < numSplits; slice++) {
815 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
816 fp_t sigma = ldexp(0.75,
817 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
818 powersVector[outer];
819 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
820 auto matrixIndex = operandIndex(outer, inner);
821 auto value = (localMatrix[matrixIndex] + sigma);
822 value -= sigma;
823 localMatrix[matrixIndex] -= value;
824 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
825 this->splitValue(outer, inner, slice) = value;
826 }
827 }
828 }
829 }

856 struct multiplicationSchedule {
857 size_t numSplitsA; // number of slices for A
858 size_t numSplitsB; // number of slices for B
859
860 std::vector<bool> mask;
861 bool operator()(size_t i, size_t j) const {
862 return mask[i * numSplitsB + j];
863 }
864};
865
874inline multiplicationSchedule makeSchedule(const config& config) {
875 multiplicationSchedule sched;
876 sched.numSplitsA = config.numSplitsA;
877 sched.numSplitsB = config.numSplitsB;
878 sched.mask.resize(config.numSplitsA * config.numSplitsB, false);
879
880 std::visit([&](auto&& spec) {
881 using T = std::decay_t<decltype(spec)>;
882
883 // Predefined strategy (full or reduced).
884 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
885
886 if (spec == multiplicationStrategy::full) {
887 std::fill(sched.mask.begin(), sched.mask.end(), true);
888 }
889 else { // reduced (anti‑diagonal rule)
890 size_t limit = std::max(config.numSplitsA, config.numSplitsB) - 1;
891
892 for (size_t i = 0; i < config.numSplitsA; ++i)
893 for (size_t j = 0; j < config.numSplitsB; ++j)
894 if (i + j <= limit)
895 sched.mask[i * config.numSplitsB + j] = true;
896 }
897 }
898
899 // Custom boolean mask.
900 else if constexpr (std::is_same_v<T, std::vector<bool>>) {
901
902 const auto& mask = spec;
903
904 if (mask.size() != config.numSplitsA * config.numSplitsB)
905 throw std::invalid_argument("Mask size mismatch.");
906
907 sched.mask = mask; // row‑major copy
908 }
909
910 }, config.multSpecification);
911
912 return sched;
913}

962template <typename splitint_t, typename accumulator_t, typename fp_t>
963std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const Decomposition<splitint_t, fp_t> &A,
964 const Decomposition<splitint_t, fp_t> &B,
965 const multiplicationSchedule &sched,
966 const matrixLayout layoutC) {
967 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
968 for (size_t diagonal = 0; diagonal <= A.numSplits + B.numSplits - 1; diagonal++) {
969 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
970 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
971 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
972 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
973 if (sched(Aindex, Bindex)) {
974 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
975 for (size_t row = 0; row < A.rows(); row++) {
976 for (size_t col = 0; col < B.cols(); col++) {
977 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
978 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
979 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
980 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
981 C[index] += scaledSum * scalingFactor;
982 }
983 }
984 }
985 Aindex--;
986 Bindex++;
987 }
988 }
989
990 return C;
991}

1014template <typename splitint_t, typename accumulator_t, typename fp_t>
1015std::vector<fp_t> computeProductsWithIntegerAccumulation(const Decomposition<splitint_t, fp_t> &A,
1016 const Decomposition<splitint_t, fp_t> &B,
1017 const multiplicationSchedule &sched,
1018 const matrixLayout layoutC) {
1019
1020 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1021 for (size_t diagonal = 0; diagonal <= A.numSplits + B.numSplits - 1; diagonal++) {
1022 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
1023 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
1024
1025 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1026
1027 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
1028 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
1029 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
1030 if (sched(Aindex, Bindex))
1031 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1032 Aindex--;
1033 Bindex++;
1034 }
1035
1036 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
1037 for (size_t row = 0; row < A.rows(); row++) {
1038 for (size_t col = 0; col < B.cols(); col++) {
1039 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1040 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1041 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1042 C[index] += scaledSum * scalingFactor;
1043 }
1044 }
1045 }
1046
1047 return C;
1048}

1092 splitA.prepare();
1093 splitB.prepare();

1097
1098 // Execute multiplication based on reduction type.
1099 if (config.redType == multiterm::reductionStrategy::floatingPoint) {
1100 return multiterm::computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(
1101 splitA, splitB, multiplicationSchedule, layoutC);
1102 } else {
1103 return multiterm::computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(
1104 splitA, splitB, multiplicationSchedule, layoutC);
