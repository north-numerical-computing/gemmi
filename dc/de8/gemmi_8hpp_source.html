414template <typename fp_t, typename splitint_t, typename accumulator_t>
415DerivedParameters deriveParameters(matrix::MatrixView<const fp_t> A,
416 matrix::MatrixView<const fp_t> B,
417 const config& config) {
418
419 // Compile-time type checks.
420 static_assert(std::is_floating_point_v<fp_t>,
421 "fp_t must be a floating-point type");
422 static_assert(std::is_integral_v<splitint_t> && std::is_signed_v<splitint_t>,
423 "splitint_t must be a signed integer type");
424 static_assert(std::is_integral_v<accumulator_t> && std::is_signed_v<accumulator_t>,
425 "accumulator_t must be a signed integer type");
426
427 // Matrix checks.
428 if (A.data == nullptr)
429 throw std::invalid_argument("Matrix A has a null data pointer");
430 if (B.data == nullptr)
431 throw std::invalid_argument("Matrix B has a null data pointer");
432 if (A.empty())
433 throw std::invalid_argument("Matrix A is empty (rows or cols is 0)");
434 if (B.empty())
435 throw std::invalid_argument("Matrix B is empty (rows or cols is 0)");
436
437 if (A.cols != B.rows)
438 throw std::invalid_argument(
439 "Dimension mismatch: A.cols (" + std::to_string(A.cols) +
440 ") != B.rows (" + std::to_string(B.rows) + ")");
441
442 // Split counts.
443 if (config.numSplitsA == 0)
444 throw std::invalid_argument("numSplitsA must be >= 1");
445 if (config.numSplitsB == 0)
446 throw std::invalid_argument("numSplitsB must be >= 1");
447
448 // Custom mask size.
449 if (std::holds_alternative<std::vector<bool>>(config.multSpecification)) {
450 const auto& mask = std::get<std::vector<bool>>(config.multSpecification);
451 const size_t expected = config.numSplitsA * config.numSplitsB;
452 if (mask.size() != expected)
453 throw std::invalid_argument(
454 "Custom mask size (" + std::to_string(mask.size()) +
455 ") does not match numSplitsA * numSplitsB (" +
456 std::to_string(expected) + ")");
457 }
458
459 // Type width constraints.
460 constexpr size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
461 constexpr size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
462 if (bitsPerInteger > bitsInAccumulator / 2)
463 throw std::invalid_argument(
464 "splitint_t (" + std::to_string(bitsPerInteger) + " bits) is too wide "
465 "for accumulator_t (" + std::to_string(bitsInAccumulator) + " bits): "
466 "require bitsPerInteger <= bitsInAccumulator / 2");
467
468 const size_t k = A.cols;
469 const double log2k = (k > 1) ? std::log2(static_cast<double>(k)) : 0.0;
470 const auto alpha = static_cast<size_t>(
471 std::max(0.0, std::floor((static_cast<double>(bitsInAccumulator) - log2k) / 2.0)));
472 const size_t bitsPerSlice = std::min(bitsPerInteger, alpha);
473 if (bitsPerSlice == 0)
474 throw std::invalid_argument(
475 "Computed bitsPerSlice is 0: inner dimension k=" + std::to_string(k) +
476 " is too large for accumulator_t (" +
477 std::to_string(bitsInAccumulator) + " bits)");
478
479 return DerivedParameters{bitsPerSlice};
480}

633template <typename splitint_t, typename fp_t>
634void computeNormalisationVectors(preparedOperand<splitint_t, fp_t>& operand) {
635 // Compute normalisation vector.
636 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
637 operand.powersVector[outer] = 0.0;
638 for (size_t j = 0; j < operand.innerDimension(); j++) {
639 operand.powersVector[outer] = std::max(operand.powersVector[outer],
640 std::abs(operand.operand(outer, j)));
641 }
642 // Compute the smallest power of 2 that is strictly greater than the
643 // maximum value in the row/column.
644 // NOTE 1: This is not the technique used in uoi24.
645 // NOTE 2: I use exponents instead of powers of 2, as I need the former
646 // to shift correctly.
647 operand.scalingExponents[outer] = fp::getStoredFloatingPointExponent(operand.powersVector[outer]);
648 operand.powersVector[outer] = std::ldexp(1.0, operand.scalingExponents[outer]);
649 }
650}

668template <typename splitint_t, typename fp_t>
669void computeFixedPointRepresentationVector(std::vector<typename fp::FloatingPointTraits<fp_t>::StorageType> &fraction,
670 std::vector<bool> &sign, size_t outer,
671 const preparedOperand<splitint_t, fp_t>& operand) {
672 using uint_t = typename fp::FloatingPointTraits<fp_t>::StorageType;
673 constexpr size_t numSignificandBits = fp::FloatingPointTraits<fp_t>::numSignificandBits;
674 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
675 fp_t value = operand.operand(outer, inner);
676 fraction[inner] = std::bit_cast<uint_t>(value);
677 sign[inner] = std::signbit(value);
678 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
679 fraction[inner] = fraction[inner] & bitmask;
680 // Restore implicit bit for normal numbers.
681 // NOTE: NaNs and infs are currently not supported.
682 if (std::fpclassify(value) == FP_NORMAL)
683 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
684 }
685}

700template <typename splitint_t, typename fp_t>
701void computeSplitsWithTruncation(preparedOperand<splitint_t, fp_t>& operand) {
702 using uint_t = typename fp::FloatingPointTraits<fp_t>::StorageType;
703
704 // Compute splits one row/column at a time.
705 constexpr size_t numSignificandBits = fp::FloatingPointTraits<fp_t>::numSignificandBits;
706 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
707 std::vector<uint_t> fraction (operand.innerDimension());
708 std::vector<bool> sign (operand.innerDimension());
709 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
710 // Get binary representation of significands of normalised row/column.
711 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
712
713 // Create bitmask.
714 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
715
716 // Perform the split.
717 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
718 // NOTE: I could have a special path for 0.
719 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
720 int currentExponent = fp::getStoredFloatingPointExponent(operand.operand(outer, inner));
721 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
722 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
723 if (exponentDifference > (signed)bitsPerSlice) {
724 exponentDifference -= bitsPerSlice;
725 } else {
726 shiftCounter += exponentDifference;
727 exponentDifference = 0;
728 uint_t bitmask = shiftCounter > 0 ?
729 smallBitmask << shiftCounter :
730 smallBitmask >> -shiftCounter;
731 uint_t currentSlice = fraction[inner] & bitmask;
732 uint_t currentSplit = shiftCounter > 0 ?
733 currentSlice >> shiftCounter :
734 currentSlice << -shiftCounter;
735 splitint_t value = static_cast<splitint_t>(currentSplit) * (sign[inner] ? -1 : 1);
736 operand.splitValue(outer, inner, slice) = value;
737 shiftCounter -= bitsPerSlice;
738 }
739 }
740 }
741 }

759template <typename splitint_t, typename fp_t>
760void computeSplitsWithUnsignedEncoding(preparedOperand<splitint_t, fp_t>& operand) {
761 using uint_t = typename fp::FloatingPointTraits<fp_t>::StorageType;
762 using wideint_t = std::conditional_t<(sizeof(splitint_t) < sizeof(int)), int, std::intmax_t>;
763
764 // Compute splits one row/column at a time.
765 constexpr size_t numSignificandBits = fp::FloatingPointTraits<fp_t>::numSignificandBits;
766 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
767 std::vector<uint_t> fraction (operand.innerDimension());
768 std::vector<bool> sign (operand.innerDimension());
769 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
770 // Get binary representation of significands of normalised row/column.
771 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
772
773 // Create bitmasks.
774 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
775 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
776 // is to keep the algorithm consistent with the bitmasking approach
777 // above.
778 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
779 // is not mentioned in sabb26, but it seems necessary to do this to
780 // ensure that the first (signed) slice does not overflow if the full
781 // bitwidth of splitint_t is used and the second slice is negative in
782 // two's complement.
783 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
784 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
785
786 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
787
788 auto matrixIndex = operand.operandIndex(outer, inner);
789
790 // NOTE: I could have a special path for 0.
791 int16_t shiftCounter;
792 int currentExponent = fp::getStoredFloatingPointExponent(operand.operand(outer, inner));
793 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
794
795 splitint_t value = 0;
796 uint_t remainder = 0;
797
798 // Slice 0.
799 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
800 // Value is too small to contribute explicit bits to slice 0.
801 value = sign[inner] ? -1 : 0;
802 remainder = sign[inner] ?
803 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
804 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
805 fraction[inner];
806 exponentDifference -= (bitsPerSlice - 1);
807 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
808 } else {
809 // Truncation.
810 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
811 exponentDifference = 0;
812 uint_t bitmask = (shiftCounter > 0)
813 ? (smallBitmask << shiftCounter)
814 : (smallBitmask >> -shiftCounter);
815 uint_t currentSlice = fraction[inner] & bitmask;
816 uint_t currentSplit = (shiftCounter > 0)
817 ? (currentSlice >> shiftCounter)
818 : (currentSlice << -shiftCounter);
819
820 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
821
822 // Rounding.
823 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
824 uint_t lowBits = fraction[inner] & lowMask;
825
826 if (lowBits != 0) {
827 value += (sign[inner] ? -1 : 0);
828 }
829 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
830 shiftCounter -= (bitsPerSlice + 1);
831 }
832 operand.memory[matrixIndex] = value;
833
834 // Remaining slices.
835 const auto width = bitsPerSlice + 1;
836 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
837 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
838 const auto digitMax = cutoff - 1; // 2^b - 1
839 const auto digitMin = -cutoff; // -2^b
840 for (size_t slice = 1; slice < operand.prepConfig.numSplits; slice++) {
841 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
842 exponentDifference -= (bitsPerSlice + 1);
843 if (sign[inner]) {
844 operand.splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
845 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
846 }
847 } else {
848 shiftCounter += exponentDifference;
849 exponentDifference = 0;
850
851 uint_t bitmask = shiftCounter > 0 ?
852 largeBitmask << shiftCounter :
853 largeBitmask >> -shiftCounter;
854
855 uint_t currentSlice = remainder & bitmask;
856 uint_t currentSplit = shiftCounter > 0 ?
857 currentSlice >> shiftCounter :
858 currentSlice << -shiftCounter;
859
860 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
861 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
862 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
863 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
864 currentSplit |= maskToAdd;
865 }
866
867 // Update previous slices if currentSplit is negative in splitint_t.
868 // Carry must propoagate upward until slice 0 if necessary.
869 if (currentSplit > static_cast<uint_t>(digitMax)) {
870 int prevSliceIndex = static_cast<int>(slice) - 1;
871 while (true) {
872 auto newValue = static_cast<wideint_t>(operand.splitValue(outer, inner, prevSliceIndex)) + 1;
873 if (newValue <= digitMax) {
874 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
875 break;
876 } else {
877 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
878 prevSliceIndex--;
879 }
880 }
881 }
882
883 // Store the slice as a signed (b+1)-bit value:
884 // [0, 2^b-1] -> unchanged
885 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
886 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
887 if (signedDigit >= cutoff) {
888 signedDigit -= base;
889 }
890
891 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
892 shiftCounter -= (bitsPerSlice + 1);
893 }
894 }
895 }
896 }
897}

912template <typename splitint_t, typename fp_t>
913void computeSplitsWithRoundToNearest(preparedOperand<splitint_t, fp_t>& operand) {
914 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
915 auto localMatrix = std::vector<fp_t>(operand.matrix.data, operand.matrix.data + operand.matrix.size());
916 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
917 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
918 // Compute exponent in signed arithmetic to avoid wraparound when
919 // bitsPerSlice * (slice + 1) approaches numSignificandBits.
920 int exponent = static_cast<int>(fp::FloatingPointTraits<fp_t>::numSignificandBits) - static_cast<int>(bitsPerSlice * (slice + 1)) + 1;
921 fp_t sigma = std::ldexp(0.75, exponent) * operand.powersVector[outer];
922 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
923 auto matrixIndex = operand.operandIndex(outer, inner);
924 auto value = (localMatrix[matrixIndex] + sigma);
925 value -= sigma;
926 localMatrix[matrixIndex] -= value;
927 value = value / operand.powersVector[outer] * std::ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
928 operand.splitValue(outer, inner, slice) = value;
929 }
930 }
931 }
932}

939template <typename splitint_t, typename fp_t>
940preparedOperand<splitint_t, fp_t> prepareOperand(matrix::MatrixView<const fp_t> matrix,
941 const OperandPreparationConfig& prepConfig) {
942 preparedOperand<splitint_t, fp_t> operand;
943 operand.matrix = matrix;
944 operand.prepConfig = prepConfig;
945 operand.memory.resize(operand.matrix.size() * prepConfig.numSplits);
946 operand.powersVector.resize(operand.prepConfig.dimension == normalisationDimension::byRows ?
947 operand.rows() : operand.cols());
948 operand.scalingExponents.resize(operand.powersVector.size());
949
950 computeNormalisationVectors(operand);
951
952 switch (prepConfig.splitType) {
953 case splittingStrategy::truncation:
954 computeSplitsWithTruncation(operand);
955 break;
956 case splittingStrategy::unsignedEncoding:
957 computeSplitsWithUnsignedEncoding(operand);
958 break;
959 case splittingStrategy::roundToNearest:
960 computeSplitsWithRoundToNearest(operand);
961 break;
962 }
963
964 return operand;
965}

977 struct multiplicationSchedule {
978 size_t numSplitsA; // number of slices for A
979 size_t numSplitsB; // number of slices for B
980
981 std::vector<bool> mask;
982
988 multiplicationSchedule(size_t numSplitsA, size_t numSplitsB) :
989 numSplitsA(numSplitsA), numSplitsB(numSplitsB), mask(numSplitsA * numSplitsB, false) {}
990
991 inline std::vector<bool>::reference operator()(size_t i, size_t j) {
992 return mask[i * numSplitsB + j];
993 }
994 inline bool operator()(size_t i, size_t j) const {
995 return mask[i * numSplitsB + j];
996 }
997};
998
1009inline multiplicationSchedule makeSchedule(const config& config) {
1010 multiplicationSchedule schedule(config.numSplitsA, config.numSplitsB);
1011
1012 std::visit([&](auto&& spec) {
1013 using T = std::decay_t<decltype(spec)>;
1014
1015 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
1016 if (spec == multiplicationStrategy::full) {
1017 std::fill(schedule.mask.begin(), schedule.mask.end(), true);
1018 } else {

1020 for (size_t i = 0; i < config.numSplitsA; ++i)
1021 for (size_t j = 0; j < config.numSplitsB; ++j)
1022 schedule(i, j) = i + j <= limit;
1023 }
1024 } else if constexpr (std::is_same_v<T, std::vector<bool>>) {
1025
1026 const auto& mask = spec;
1027

1029 throw std::invalid_argument("Mask size mismatch.");
1030
1031 schedule.mask = mask; // row‑major copy
1032 }
1033
1034 }, config.multSpecification);

1087template <typename splitint_t, typename accumulator_t, typename fp_t>
1088std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const preparedOperand<splitint_t, fp_t> &A,
1089 const preparedOperand<splitint_t, fp_t> &B,
1090 const multiplicationSchedule &schedule,
1091 const matrix::matrixLayout layoutC) {
1092 auto numSplitsA = A.prepConfig.numSplits;
1093 auto numSplitsB = B.prepConfig.numSplits;
1094 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1095 for (size_t diagonal = 0; diagonal <= numSplitsA + numSplitsB - 1; diagonal++) {
1096 int Aindex = diagonal < numSplitsA ? static_cast<int>(diagonal) : static_cast<int>(numSplitsA - 1);
1097 size_t Bindex = diagonal > numSplitsA - 1 ? diagonal - numSplitsA + 1 : 0;
1098 while (Aindex >= 0 && Bindex <= std::min(diagonal, numSplitsB - 1)) {
1099 if (schedule(Aindex, Bindex)) {
1100 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
1101 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1102 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1103 for (size_t row = 0; row < A.rows(); row++) {
1104 for (size_t col = 0; col < B.cols(); col++) {
1105 auto index = (layoutC == matrix::matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1106 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1107 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1108 C[index] += scaledSum * scalingFactor;
1109 }
1110 }
1111 }
1112 Aindex--;
1113 Bindex++;
1114 }
1115 }
1116
1117 return C;
1118}

1140template <typename splitint_t, typename accumulator_t, typename fp_t>
1141std::vector<fp_t> computeProductsWithIntegerAccumulation(const preparedOperand<splitint_t, fp_t> &A,
1142 const preparedOperand<splitint_t, fp_t> &B,
1143 const multiplicationSchedule &schedule,
1144 const matrix::matrixLayout layoutC) {
1145 auto numSplitsA = A.prepConfig.numSplits;
1146 auto numSplitsB = B.prepConfig.numSplits;
1147
1148 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1149 for (size_t diagonal = 0; diagonal <= numSplitsA + numSplitsB - 1; diagonal++) {
1150 int Aindex = diagonal < numSplitsA ? static_cast<int>(diagonal) : static_cast<int>(numSplitsA - 1);
1151 size_t Bindex = diagonal > numSplitsA - 1 ? diagonal - numSplitsA + 1 : 0;
1152
1153 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1154
1155 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
1156 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
1157 while (Aindex >= 0 && Bindex <= std::min(diagonal, numSplitsB - 1)) {
1158 if (schedule(Aindex, Bindex))
1159 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1160 Aindex--;
1161 Bindex++;
1162 }
1163
1164 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
1165 for (size_t row = 0; row < A.rows(); row++) {
1166 for (size_t col = 0; col < B.cols(); col++) {
1167 auto index = (layoutC == matrix::matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1168 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1169 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1170 C[index] += scaledSum * scalingFactor;
1171 }
1172 }
1173 }
1174
1175 return C;
1176}
