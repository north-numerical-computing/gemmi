359template <typename fp_t, typename splitint_t, typename accumulator_t>
360void validateParameters(matrix::MatrixView<const fp_t> A,
361 matrix::MatrixView<const fp_t> B,
362 const config& config) {
363
364 // Compile-time type checks.
365 static_assert(std::is_floating_point_v<fp_t>,
366 "fp_t must be a floating-point type");
367 static_assert(std::is_integral_v<splitint_t> && std::is_signed_v<splitint_t>,
368 "splitint_t must be a signed integer type");
369 static_assert(std::is_integral_v<accumulator_t> && std::is_signed_v<accumulator_t>,
370 "accumulator_t must be a signed integer type");
371
372 // Matrix checks.
373 if (A.data == nullptr)
374 throw std::invalid_argument("Matrix A has a null data pointer");
375 if (B.data == nullptr)
376 throw std::invalid_argument("Matrix B has a null data pointer");
377 if (A.empty())
378 throw std::invalid_argument("Matrix A is empty (rows or cols is 0)");
379 if (B.empty())
380 throw std::invalid_argument("Matrix B is empty (rows or cols is 0)");
381
382 if (A.cols != B.rows)
383 throw std::invalid_argument(
384 "Dimension mismatch: A.cols (" + std::to_string(A.cols) +
385 ") != B.rows (" + std::to_string(B.rows) + ")");
386
387 // Split counts.
388 if (config.numSplitsA == 0)
389 throw std::invalid_argument("numSplitsA must be >= 1");
390 if (config.numSplitsB == 0)
391 throw std::invalid_argument("numSplitsB must be >= 1");
392
393 // Custom mask size.
394 if (std::holds_alternative<std::vector<bool>>(config.multSpecification)) {
395 const auto& mask = std::get<std::vector<bool>>(config.multSpecification);
396 const size_t expected = config.numSplitsA * config.numSplitsB;
397 if (mask.size() != expected)
398 throw std::invalid_argument(
399 "Custom mask size (" + std::to_string(mask.size()) +
400 ") does not match numSplitsA * numSplitsB (" +
401 std::to_string(expected) + ")");
402 }
403
404 // Type width constraints.
405 constexpr size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
406 constexpr size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
407 if (bitsPerInteger > bitsInAccumulator / 2)
408 throw std::invalid_argument(
409 "splitint_t (" + std::to_string(bitsPerInteger) + " bits) is too wide "
410 "for accumulator_t (" + std::to_string(bitsInAccumulator) + " bits): "
411 "require bitsPerInteger <= bitsInAccumulator / 2");
412}

423template <typename splitint_t, typename accumulator_t>
424size_t computeBitsPerSlice(size_t innerDimension) {
425 constexpr size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
426 constexpr size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
427
428 const double log2k = (innerDimension > 1) ? std::log2(static_cast<double>(innerDimension)) : 0.0;
429 const auto alpha = static_cast<size_t>(
430 std::max(0.0, std::floor((static_cast<double>(bitsInAccumulator) - log2k) / 2.0)));
431 const size_t bitsPerSlice = std::min(bitsPerInteger, alpha);
432 if (bitsPerSlice == 0) {
433 throw std::invalid_argument(
434 "Computed bitsPerSlice is 0: inner dimension k=" + std::to_string(innerDimension) +
435 " is too large for accumulator_t (" +
436 std::to_string(bitsInAccumulator) + " bits)");
437 }
438
439 return bitsPerSlice;
440}

593template <typename splitint_t, typename fp_t>
594void computeNormalisationVectors(preparedOperand<splitint_t, fp_t>& operand) {
595 // Compute normalisation vector.
596 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
597 operand.powersVector[outer] = 0.0;
598 for (size_t j = 0; j < operand.innerDimension(); j++) {
599 operand.powersVector[outer] = std::max(operand.powersVector[outer],
600 std::abs(operand.operand(outer, j)));
601 }
602 // Compute the smallest power of 2 that is strictly greater than the
603 // maximum value in the row/column.
604 // NOTE 1: This is not the technique used in uoi24.
605 // NOTE 2: I use exponents instead of powers of 2, as I need the former
606 // to shift correctly.
607 operand.scalingExponents[outer] = fp::getStoredFloatingPointExponent(operand.powersVector[outer]);
608 operand.powersVector[outer] = std::ldexp(1.0, operand.scalingExponents[outer]);
609 }
610}

628template <typename splitint_t, typename fp_t>
629void computeFixedPointRepresentationVector(std::vector<typename fp::FloatingPointTraits<fp_t>::StorageType> &fraction,
630 std::vector<bool> &sign, size_t outer,
631 const preparedOperand<splitint_t, fp_t>& operand) {
632 using uint_t = typename fp::FloatingPointTraits<fp_t>::StorageType;
633 constexpr size_t numSignificandBits = fp::FloatingPointTraits<fp_t>::numSignificandBits;
634 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
635 fp_t value = operand.operand(outer, inner);
636 fraction[inner] = std::bit_cast<uint_t>(value);
637 sign[inner] = std::signbit(value);
638 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
639 fraction[inner] = fraction[inner] & bitmask;
640 // Restore implicit bit for normal numbers.
641 // NOTE: NaNs and infs are currently not supported.
642 if (std::fpclassify(value) == FP_NORMAL)
643 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
644 }
645}

660template <typename splitint_t, typename fp_t>
661void computeSplitsWithTruncation(preparedOperand<splitint_t, fp_t>& operand) {
662 using uint_t = typename fp::FloatingPointTraits<fp_t>::StorageType;
663
664 // Compute splits one row/column at a time.
665 constexpr size_t numSignificandBits = fp::FloatingPointTraits<fp_t>::numSignificandBits;
666 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
667 std::vector<uint_t> fraction (operand.innerDimension());
668 std::vector<bool> sign (operand.innerDimension());
669 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
670 // Get binary representation of significands of normalised row/column.
671 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
672
673 // Create bitmask.
674 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
675
676 // Perform the split.
677 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
678 // NOTE: I could have a special path for 0.
679 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
680 int currentExponent = fp::getStoredFloatingPointExponent(operand.operand(outer, inner));
681 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
682 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
683 if (exponentDifference > (signed)bitsPerSlice) {
684 exponentDifference -= bitsPerSlice;
685 } else {
686 shiftCounter += exponentDifference;
687 exponentDifference = 0;
688 uint_t bitmask = shiftCounter > 0 ?
689 smallBitmask << shiftCounter :
690 smallBitmask >> -shiftCounter;
691 uint_t currentSlice = fraction[inner] & bitmask;
692 uint_t currentSplit = shiftCounter > 0 ?
693 currentSlice >> shiftCounter :
694 currentSlice << -shiftCounter;
695 splitint_t value = static_cast<splitint_t>(currentSplit) * (sign[inner] ? -1 : 1);
696 operand.splitValue(outer, inner, slice) = value;
697 shiftCounter -= bitsPerSlice;
698 }
699 }
700 }
701 }

719template <typename splitint_t, typename fp_t>
720void computeSplitsWithUnsignedEncoding(preparedOperand<splitint_t, fp_t>& operand) {
721 using uint_t = typename fp::FloatingPointTraits<fp_t>::StorageType;
722 using wideint_t = std::conditional_t<(sizeof(splitint_t) < sizeof(int)), int, std::intmax_t>;
723
724 // Compute splits one row/column at a time.
725 constexpr size_t numSignificandBits = fp::FloatingPointTraits<fp_t>::numSignificandBits;
726 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
727 std::vector<uint_t> fraction (operand.innerDimension());
728 std::vector<bool> sign (operand.innerDimension());
729 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
730 // Get binary representation of significands of normalised row/column.
731 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
732
733 // Create bitmasks.
734 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
735 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
736 // is to keep the algorithm consistent with the bitmasking approach
737 // above.
738 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
739 // is not mentioned in sabb26, but it seems necessary to do this to
740 // ensure that the first (signed) slice does not overflow if the full
741 // bitwidth of splitint_t is used and the second slice is negative in
742 // two's complement.
743 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
744 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
745
746 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
747
748 auto matrixIndex = operand.operandIndex(outer, inner);
749
750 // NOTE: I could have a special path for 0.
751 int16_t shiftCounter;
752 int currentExponent = fp::getStoredFloatingPointExponent(operand.operand(outer, inner));
753 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
754
755 splitint_t value = 0;
756 uint_t remainder = 0;
757
758 // Slice 0.
759 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
760 // Value is too small to contribute explicit bits to slice 0.
761 value = sign[inner] ? -1 : 0;
762 remainder = sign[inner] ?
763 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
764 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
765 fraction[inner];
766 exponentDifference -= (bitsPerSlice - 1);
767 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
768 } else {
769 // Truncation.
770 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
771 exponentDifference = 0;
772 uint_t bitmask = (shiftCounter > 0)
773 ? (smallBitmask << shiftCounter)
774 : (smallBitmask >> -shiftCounter);
775 uint_t currentSlice = fraction[inner] & bitmask;
776 uint_t currentSplit = (shiftCounter > 0)
777 ? (currentSlice >> shiftCounter)
778 : (currentSlice << -shiftCounter);
779
780 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
781
782 // Rounding.
783 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
784 uint_t lowBits = fraction[inner] & lowMask;
785
786 if (lowBits != 0) {
787 value += (sign[inner] ? -1 : 0);
788 }
789 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
790 shiftCounter -= (bitsPerSlice + 1);
791 }
792 operand.memory[matrixIndex] = value;
793
794 // Remaining slices.
795 const auto width = bitsPerSlice + 1;
796 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
797 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
798 const auto digitMax = cutoff - 1; // 2^b - 1
799 const auto digitMin = -cutoff; // -2^b
800 for (size_t slice = 1; slice < operand.prepConfig.numSplits; slice++) {
801 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
802 exponentDifference -= (bitsPerSlice + 1);
803 if (sign[inner]) {
804 operand.splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
805 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
806 }
807 } else {
808 shiftCounter += exponentDifference;
809 exponentDifference = 0;
810
811 uint_t bitmask = shiftCounter > 0 ?
812 largeBitmask << shiftCounter :
813 largeBitmask >> -shiftCounter;
814
815 uint_t currentSlice = remainder & bitmask;
816 uint_t currentSplit = shiftCounter > 0 ?
817 currentSlice >> shiftCounter :
818 currentSlice << -shiftCounter;
819
820 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
821 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
822 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
823 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
824 currentSplit |= maskToAdd;
825 }
826
827 // Update previous slices if currentSplit is negative in splitint_t.
828 // Carry must propoagate upward until slice 0 if necessary.
829 if (currentSplit > static_cast<uint_t>(digitMax)) {
830 int prevSliceIndex = static_cast<int>(slice) - 1;
831 while (true) {
832 auto newValue = static_cast<wideint_t>(operand.splitValue(outer, inner, prevSliceIndex)) + 1;
833 if (newValue <= digitMax) {
834 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
835 break;
836 } else {
837 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
838 prevSliceIndex--;
839 }
840 }
841 }
842
843 // Store the slice as a signed (b+1)-bit value:
844 // [0, 2^b-1] -> unchanged
845 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
846 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
847 if (signedDigit >= cutoff) {
848 signedDigit -= base;
849 }
850
851 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
852 shiftCounter -= (bitsPerSlice + 1);
853 }
854 }
855 }
856 }
857}

872template <typename splitint_t, typename fp_t>
873void computeSplitsWithRoundToNearest(preparedOperand<splitint_t, fp_t>& operand) {
874 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
875 auto localMatrix = std::vector<fp_t>(operand.matrix.data, operand.matrix.data + operand.matrix.size());
876 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
877 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
878 // Compute exponent in signed arithmetic to avoid wraparound when
879 // bitsPerSlice * (slice + 1) approaches numSignificandBits.
880 int exponent = static_cast<int>(fp::FloatingPointTraits<fp_t>::numSignificandBits) - static_cast<int>(bitsPerSlice * (slice + 1)) + 1;
881 fp_t sigma = std::ldexp(0.75, exponent) * operand.powersVector[outer];
882 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
883 auto matrixIndex = operand.operandIndex(outer, inner);
884 auto value = (localMatrix[matrixIndex] + sigma);
885 value -= sigma;
886 localMatrix[matrixIndex] -= value;
887 value = value / operand.powersVector[outer] * std::ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
888 operand.splitValue(outer, inner, slice) = value;
889 }
890 }
891 }
892}

899template <typename splitint_t, typename fp_t>
900preparedOperand<splitint_t, fp_t> prepareOperand(matrix::MatrixView<const fp_t> matrix,
901 const OperandPreparationConfig& prepConfig) {
902 preparedOperand<splitint_t, fp_t> operand;
903 operand.matrix = matrix;
904 operand.prepConfig = prepConfig;
905 operand.memory.resize(operand.matrix.size() * prepConfig.numSplits);
906 operand.powersVector.resize(operand.prepConfig.dimension == normalisationDimension::byRows ?
907 operand.rows() : operand.cols());
908 operand.scalingExponents.resize(operand.powersVector.size());
909
910 computeNormalisationVectors(operand);
911
912 switch (prepConfig.splitType) {
913 case splittingStrategy::truncation:
914 computeSplitsWithTruncation(operand);
915 break;
916 case splittingStrategy::unsignedEncoding:
917 computeSplitsWithUnsignedEncoding(operand);
918 break;
919 case splittingStrategy::roundToNearest:
920 computeSplitsWithRoundToNearest(operand);
921 break;
922 }
923
924 return operand;
925}

948 multiplicationSchedule(size_t numSplitsA, size_t numSplitsB) :
949 numSplitsA(numSplitsA), numSplitsB(numSplitsB), mask(numSplitsA * numSplitsB, false) {}
950
951 inline std::vector<bool>::reference operator()(size_t i, size_t j) {
952 return mask[i * numSplitsB + j];
953 }
954 inline bool operator()(size_t i, size_t j) const {
955 return mask[i * numSplitsB + j];

969inline multiplicationSchedule makeSchedule(const config& config) {
970 multiplicationSchedule schedule(config.numSplitsA, config.numSplitsB);
971
972 std::visit([&](auto&& spec) {
973 using T = std::decay_t<decltype(spec)>;
974
975 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
976 if (spec == multiplicationStrategy::full) {
977 std::fill(schedule.mask.begin(), schedule.mask.end(), true);
978 } else {
979 size_t limit = std::max(config.numSplitsA, config.numSplitsB) - 1;
980 for (size_t i = 0; i < config.numSplitsA; ++i)
981 for (size_t j = 0; j < config.numSplitsB; ++j)
982 schedule(i, j) = i + j <= limit;
983 }

1047std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const preparedOperand<splitint_t, fp_t> &A,
1048 const preparedOperand<splitint_t, fp_t> &B,
1049 const multiplicationSchedule &schedule,
1050 const matrix::matrixLayout layoutC) {

1052 auto numSplitsB = B.prepConfig.numSplits;
1053 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1054 for (size_t diagonal = 0; diagonal <= numSplitsA + numSplitsB - 1; diagonal++) {
1055 int Aindex = diagonal < numSplitsA ? static_cast<int>(diagonal) : static_cast<int>(numSplitsA - 1);
1056 size_t Bindex = diagonal > numSplitsA - 1 ? diagonal - numSplitsA + 1 : 0;
1057 while (Aindex >= 0 && Bindex <= std::min(diagonal, numSplitsB - 1)) {
1058 if (schedule(Aindex, Bindex)) {
1059 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
1060 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1061 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);

1063 for (size_t col = 0; col < B.cols(); col++) {
1064 auto index = (layoutC == matrix::matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1065 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1066 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1067 C[index] += scaledSum * scalingFactor;
1068 }
1069 }
1070 }
1071 Aindex--;
1072 Bindex++;
1073 }
1074 }

1130 }
1131 }
