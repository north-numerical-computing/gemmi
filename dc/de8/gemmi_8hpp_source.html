433template <typename fp_t, typename splitint_t, typename accumulator_t>
434DerivedParameters deriveParameters(MatrixView<const fp_t> A,
435 MatrixView<const fp_t> B,
436 const config& config) {
437
438 validateConfig(config);
439
440 // Compile-time type checks.
441 static_assert(std::is_floating_point_v<fp_t>,
442 "fp_t must be a floating-point type");
443 static_assert(std::is_integral_v<splitint_t> && std::is_signed_v<splitint_t>,
444 "splitint_t must be a signed integer type");
445 static_assert(std::is_integral_v<accumulator_t> && std::is_signed_v<accumulator_t>,
446 "accumulator_t must be a signed integer type");
447
448 // Matrix checks.
449 if (A.data == nullptr)
450 throw std::invalid_argument("Matrix A has a null data pointer");
451 if (B.data == nullptr)
452 throw std::invalid_argument("Matrix B has a null data pointer");
453 if (A.empty())
454 throw std::invalid_argument("Matrix A is empty (rows or cols is 0)");
455 if (B.empty())
456 throw std::invalid_argument("Matrix B is empty (rows or cols is 0)");
457
458 if (A.cols != B.rows)
459 throw std::invalid_argument(
460 "Dimension mismatch: A.cols (" + std::to_string(A.cols) +
461 ") != B.rows (" + std::to_string(B.rows) + ")");
462
463 // Split counts.
464 if (config.numSplitsA == 0)
465 throw std::invalid_argument("numSplitsA must be >= 1");
466 if (config.numSplitsB == 0)
467 throw std::invalid_argument("numSplitsB must be >= 1");
468
469 // Custom mask size.
470 if (std::holds_alternative<std::vector<bool>>(config.multSpecification)) {
471 const auto& mask = std::get<std::vector<bool>>(config.multSpecification);
472 const size_t expected = config.numSplitsA * config.numSplitsB;
473 if (mask.size() != expected)
474 throw std::invalid_argument(
475 "Custom mask size (" + std::to_string(mask.size()) +
476 ") does not match numSplitsA * numSplitsB (" +
477 std::to_string(expected) + ")");
478 }
479
480 // Type width constraints.
481 constexpr size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
482 constexpr size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
483 if (bitsPerInteger > bitsInAccumulator / 2)
484 throw std::invalid_argument(
485 "splitint_t (" + std::to_string(bitsPerInteger) + " bits) is too wide "
486 "for accumulator_t (" + std::to_string(bitsInAccumulator) + " bits): "
487 "require bitsPerInteger <= bitsInAccumulator / 2");
488
489 const size_t k = A.cols;
490 const double log2k = (k > 1) ? std::log2(static_cast<double>(k)) : 0.0;
491 const auto alpha = static_cast<size_t>(
492 std::max(0.0, std::floor((static_cast<double>(bitsInAccumulator) - log2k) / 2.0)));
493 const size_t bitsPerSlice = std::min(bitsPerInteger, alpha);
494 if (bitsPerSlice == 0)
495 throw std::invalid_argument(
496 "Computed bitsPerSlice is 0: inner dimension k=" + std::to_string(k) +
497 " is too large for accumulator_t (" +
498 std::to_string(bitsInAccumulator) + " bits)");
499
500 return DerivedParameters{bitsPerSlice};
501}

626 auto bitsPerSlice = prepConfig.bitsPerSlice;
627 switch (prepConfig.splitType) {
628 case splittingStrategy::truncation:
629 // Slice k -> k * b
630 return static_cast<int>((slice + 1) * bitsPerSlice);
631

653template <typename splitint_t, typename fp_t>
654void computeNormalisationVectors(preparedOperand<splitint_t, fp_t>& operand) {
655 // Compute normalisation vector.
656 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
657 operand.powersVector[outer] = 0.0;
658 for (size_t j = 0; j < operand.innerDimension(); j++) {
659 operand.powersVector[outer] = std::max(operand.powersVector[outer],
660 std::abs(operand.operand(outer, j)));
661 }
662 // Compute the smallest power of 2 that is strictly greater than the
663 // maximum value in the row/column.
664 // NOTE 1: This is not the technique used in uoi24.
665 // NOTE 2: I use exponents instead of powers of 2, as I need the former
666 // to shift correctly.
667 operand.scalingExponents[outer] = getStoredFloatingPointExponent(operand.powersVector[outer]);
668 operand.powersVector[outer] = std::ldexp(1.0, operand.scalingExponents[outer]);
669 }
670}

683template <typename splitint_t, typename fp_t>
684void computeFixedPointRepresentationVector(std::vector<typename FloatingPointTraits<fp_t>::StorageType> &fraction,
685 std::vector<bool> &sign, size_t outer,
686 const preparedOperand<splitint_t, fp_t>& operand) {
687 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
688 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
689 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
690 fp_t value = operand.operand(outer, inner);
691 fraction[inner] = std::bit_cast<uint_t>(value);
692 sign[inner] = std::signbit(value);
693 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
694 fraction[inner] = fraction[inner] & bitmask;
695 // Restore implicit bit for normal numbers.
696 // NOTE: NaNs and infs are currently not supported.
697 if (std::fpclassify(value) == FP_NORMAL)
698 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
699 }
700}

715template <typename splitint_t, typename fp_t>
716void computeSplitsWithTruncation(preparedOperand<splitint_t, fp_t>& operand) {
717 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
718
719 // Compute splits one row/column at a time.
720 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
721 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
722 std::vector<uint_t> fraction (operand.innerDimension());
723 std::vector<bool> sign (operand.innerDimension());
724 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
725 // Get binary representation of significands of normalised row/column.
726 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
727
728 // Create bitmask.
729 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
730
731 // Perform the split.
732 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
733 // NOTE: I could have a special path for 0.
734 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
735 int currentExponent = getStoredFloatingPointExponent(operand.operand(outer, inner));
736 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
737 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
738 if (exponentDifference > (signed)bitsPerSlice) {
739 exponentDifference -= bitsPerSlice;
740 } else {
741 shiftCounter += exponentDifference;
742 exponentDifference = 0;
743 uint_t bitmask = shiftCounter > 0 ?
744 smallBitmask << shiftCounter :
745 smallBitmask >> -shiftCounter;
746 uint_t currentSlice = fraction[inner] & bitmask;
747 uint_t currentSplit = shiftCounter > 0 ?
748 currentSlice >> shiftCounter :
749 currentSlice << -shiftCounter;
750 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
751 operand.splitValue(outer, inner, slice) = value;
752 shiftCounter -= bitsPerSlice;
753 }
754 }
755 }
756 }

774template <typename splitint_t, typename fp_t>
775void computeSplitsWithUnsignedEncoding(preparedOperand<splitint_t, fp_t>& operand) {
776 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
777 using wideint_t = std::conditional_t<(sizeof(splitint_t) < sizeof(int)), int, std::intmax_t>;
778
779 // Compute splits one row/column at a time.
780 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
781 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
782 std::vector<uint_t> fraction (operand.innerDimension());
783 std::vector<bool> sign (operand.innerDimension());
784 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
785 // Get binary representation of significands of normalised row/column.
786 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
787
788 // Create bitmasks.
789 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
790 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
791 // is to keep the algorithm consistent with the bitmasking approach
792 // above.
793 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
794 // is not mentioned in sabb26, but it seems necessary to do this to
795 // ensure that the first (signed) slice does not overflow if the full
796 // bitwidth of splitint_t is used and the second slice is negative in
797 // two's complement.
798 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
799 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
800
801 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
802
803 auto matrixIndex = operand.operandIndex(outer, inner);
804
805 // NOTE: I could have a special path for 0.
806 int16_t shiftCounter;
807 int currentExponent = getStoredFloatingPointExponent(operand.operand(outer, inner));
808 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
809
810 splitint_t value = 0;
811 uint_t remainder = 0;
812
813 // Slice 0.
814 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
815 // Value is too small to contribute explicit bits to slice 0.
816 value = sign[inner] ? -1 : 0;
817 remainder = sign[inner] ?
818 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
819 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
820 fraction[inner];
821 exponentDifference -= (bitsPerSlice - 1);
822 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
823 } else {
824 // Truncation.
825 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
826 exponentDifference = 0;
827 uint_t bitmask = (shiftCounter > 0)
828 ? (smallBitmask << shiftCounter)
829 : (smallBitmask >> -shiftCounter);
830 uint_t currentSlice = fraction[inner] & bitmask;
831 uint_t currentSplit = (shiftCounter > 0)
832 ? (currentSlice >> shiftCounter)
833 : (currentSlice << -shiftCounter);
834
835 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
836
837 // Rounding.
838 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
839 uint_t lowBits = fraction[inner] & lowMask;
840
841 if (lowBits != 0) {
842 value += (sign[inner] ? -1 : 0);
843 }
844 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
845 shiftCounter -= (bitsPerSlice + 1);
846 }
847 operand.memory[matrixIndex] = value;
848
849 // Remaining slices.
850 const auto width = bitsPerSlice + 1;
851 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
852 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
853 const auto digitMax = cutoff - 1; // 2^b - 1
854 const auto digitMin = -cutoff; // -2^b
855 for (size_t slice = 1; slice < operand.prepConfig.numSplits; slice++) {
856 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
857 exponentDifference -= (bitsPerSlice + 1);
858 if (sign[inner]) {
859 operand.splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
860 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
861 }
862 } else {
863 shiftCounter += exponentDifference;
864 exponentDifference = 0;
865
866 uint_t bitmask = shiftCounter > 0 ?
867 largeBitmask << shiftCounter :
868 largeBitmask >> -shiftCounter;
869
870 uint_t currentSlice = remainder & bitmask;
871 uint_t currentSplit = shiftCounter > 0 ?
872 currentSlice >> shiftCounter :
873 currentSlice << -shiftCounter;
874
875 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
876 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
877 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
878 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
879 currentSplit |= maskToAdd;
880 }
881
882 // Update previous slices if currentSplit is negative in splitint_t.
883 // Carry must propoagate upward until slice 0 if necessary.
884 if (currentSplit > static_cast<uint_t>(digitMax)) {
885 int prevSliceIndex = static_cast<int>(slice) - 1;
886 while (true) {
887 auto newValue = static_cast<wideint_t>(operand.splitValue(outer, inner, prevSliceIndex)) + 1;
888 if (newValue <= digitMax) {
889 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
890 break;
891 } else {
892 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
893 prevSliceIndex--;
894 }
895 }
896 }
897
898 // Store the slice as a signed (b+1)-bit value:
899 // [0, 2^b-1] -> unchanged
900 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
901 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
902 if (signedDigit >= cutoff) {
903 signedDigit -= base;
904 }
905
906 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
907 shiftCounter -= (bitsPerSlice + 1);
908 }
909 }
910 }
911 }
912}

927template <typename splitint_t, typename fp_t>
928void computeSplitsWithRoundToNearest(preparedOperand<splitint_t, fp_t>& operand) {
929 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
930 auto localMatrix = std::vector<fp_t>(operand.matrix.data, operand.matrix.data + operand.matrix.size());
931 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
932 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
933 fp_t sigma = ldexp(0.75,
934 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
935 operand.powersVector[outer];
936 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
937 auto matrixIndex = operand.operandIndex(outer, inner);
938 auto value = (localMatrix[matrixIndex] + sigma);
939 value -= sigma;
940 localMatrix[matrixIndex] -= value;
941 value = value / operand.powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
942 operand.splitValue(outer, inner, slice) = value;
943 }
944 }
945 }
946}

953template <typename splitint_t, typename fp_t>
954preparedOperand<splitint_t, fp_t> prepareOperand(MatrixView<const fp_t> matrix,
955 const OperandPreparationConfig& prepConfig) {
956 preparedOperand<splitint_t, fp_t> operand;
957 operand.matrix = matrix;
958 operand.prepConfig = prepConfig;
959 operand.memory.resize(operand.matrix.size() * operand.prepConfig.numSplits * prepConfig.numSplits);
960 operand.powersVector.resize(operand.prepConfig.dimension == normalisationDimension::byRows ?
961 operand.rows() : operand.cols());
962 operand.scalingExponents.resize(operand.powersVector.size());
963
964 computeNormalisationVectors(operand);
965
966 switch (prepConfig.splitType) {
967 case splittingStrategy::truncation:
968 computeSplitsWithTruncation(operand);
969 break;
970 case splittingStrategy::unsignedEncoding:
971 computeSplitsWithUnsignedEncoding(operand);
972 break;
973 case splittingStrategy::roundToNearest:
974 computeSplitsWithRoundToNearest(operand);
975 break;
976 }
977
978 return operand;
979}

992 size_t numSplitsB; // number of slices for B

995 inline std::vector<bool>::reference operator()(size_t i, size_t j) {

1011inline multiplicationSchedule makeSchedule(const config& config) {
1012 multiplicationSchedule schedule;
1013 schedule.numSplitsA = config.numSplitsA;
1014 schedule.numSplitsB = config.numSplitsB;
1015 schedule.mask.resize(config.numSplitsA * config.numSplitsB, false);
1016
1017 std::visit([&](auto&& spec) {
1018 using T = std::decay_t<decltype(spec)>;
1019
1020 // Predefined strategy (full or reduced).
1021 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
1022
1023 if (spec == multiplicationStrategy::full) {
1024 std::fill(schedule.mask.begin(), schedule.mask.end(), true);
1025 }
1026 else { // reduced (anti‑diagonal rule)
1027 size_t limit = std::max(config.numSplitsA, config.numSplitsB) - 1;
1028
1029 for (size_t i = 0; i < config.numSplitsA; ++i)
1030 for (size_t j = 0; j < config.numSplitsB; ++j)
1031 if (i + j <= limit)
1032 schedule(i, j) = true;
1033 }
1034 }
1035
1036 // Custom boolean mask.
1037 else if constexpr (std::is_same_v<T, std::vector<bool>>) {
1038
1039 const auto& mask = spec;
1040
1041 if (mask.size() != config.numSplitsA * config.numSplitsB)
1042 throw std::invalid_argument("Mask size mismatch.");
1043
1044 schedule.mask = mask; // row‑major copy
1045 }
1046
1047 }, config.multSpecification);
1048
1049 return schedule;
1050}

1099template <typename splitint_t, typename accumulator_t, typename fp_t>
1100std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const preparedOperand<splitint_t, fp_t> &A,
1101 const preparedOperand<splitint_t, fp_t> &B,
1102 const multiplicationSchedule &schedule,
1103 const matrixLayout layoutC) {
1104 auto numSplitsA = A.prepConfig.numSplits;
1105 auto numSplitsB = B.prepConfig.numSplits;
1106 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1107 for (size_t diagonal = 0; diagonal <= numSplitsA + numSplitsB - 1; diagonal++) {
1108 int Aindex = diagonal < numSplitsA - 1 ? diagonal : numSplitsA - 1;
1109 size_t Bindex = diagonal > numSplitsA - 1 ? diagonal - numSplitsA + 1 : 0;
1110 while (Aindex >= 0 && Bindex <= std::min(diagonal, numSplitsB - 1)) {
1111 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
1112 if (schedule(Aindex, Bindex)) {
1113 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1114 for (size_t row = 0; row < A.rows(); row++) {
1115 for (size_t col = 0; col < B.cols(); col++) {
1116 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1117 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1118 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1119 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1120 C[index] += scaledSum * scalingFactor;
1121 }
1122 }
1123 }
1124 Aindex--;
1125 Bindex++;
1126 }
1127 }
1128
1129 return C;

1153template <typename splitint_t, typename accumulator_t, typename fp_t>
1154std::vector<fp_t> computeProductsWithIntegerAccumulation(const preparedOperand<splitint_t, fp_t> &A,
1155 const preparedOperand<splitint_t, fp_t> &B,
1156 const multiplicationSchedule &schedule,
1157 const matrixLayout layoutC) {
1158 auto numSplitsA = A.prepConfig.numSplits;
1159 auto numSplitsB = B.prepConfig.numSplits;
1160
1161 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
1162 for (size_t diagonal = 0; diagonal <= numSplitsA + numSplitsB - 1; diagonal++) {
1163 int Aindex = diagonal < numSplitsA ? static_cast<int>(diagonal) : static_cast<int>(numSplitsA - 1);
1164 size_t Bindex = diagonal > numSplitsA - 1 ? diagonal - numSplitsA + 1 : 0;
1165
1166 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
1167
1168 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
1169 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
1170 while (Aindex >= 0 && Bindex <= std::min(diagonal, numSplitsB - 1)) {
1171 if (schedule(Aindex, Bindex))
1172 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1173 Aindex--;
1174 Bindex++;
1175 }
1176
1177 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
1178 for (size_t row = 0; row < A.rows(); row++) {
1179 for (size_t col = 0; col < B.cols(); col++) {
1180 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1181 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1182 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1183 C[index] += scaledSum * scalingFactor;
1184 }
1185 }
1186 }
1187
1188 return C;
1189}
