514 void computeNormalisationVectors() {
515 // Compute normalisation vector.
516 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
517 this->powersVector[outer] = 0.0;
518 for (size_t j = 0; j < this->innerDimension(); j++) {
519 this->powersVector[outer] = std::max(this->powersVector[outer],
520 std::abs(this->operand(outer, j)));
521 }
522 // Compute the smallest power of 2 that is strictly greater than the
523 // maximum value in the row/column.
524 // NOTE 1: This is not the technique used in uoi24.
525 // NOTE 2: I use exponents instead of powers of 2, as I need the former
526 // to shift correctly.
527 this->scalingExponents[outer] = getStoredFloatingPointExponent(this->powersVector[outer]);
528 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);

537 int computeSliceBitOffset(size_t slice) const {
538 switch (splitType) {
539 case splittingStrategy::truncation:
540 // Slice k -> k * b
541 return static_cast<int>((slice + 1) * bitsPerSlice);

544 // Slice 0 -> b - 1
545 // Slice k -> k * b - 1
546 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
547
548 case splittingStrategy::unsignedEncoding:
549 // Slice 0 -> b - 1
550 // Slice k -> (b - 1) + k * (b + 1)
551 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));

555 throw std::logic_error("Unhandled splittingStrategy");

567 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
568 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
569 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
570 fp_t value = this->operand(outer, inner);
571 fraction[inner] = std::bit_cast<uint_t>(value);
572 sign[inner] = std::signbit(value);
573 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
574 fraction[inner] = fraction[inner] & bitmask;
575 // Restore implicit bit for normal numbers.
576 // NOTE: NaNs and infs are currently not supported.
577 if (std::fpclassify(value) == FP_NORMAL)
578 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
579 }

591 void computeSplitsWithTruncation() {
592 this->splitType = splittingStrategy::truncation;
593 // Compute splits one row/column at a time.
594 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
595 auto bitsPerSlice = this->bitsPerSlice;
596 std::vector<uint_t> fraction (this->innerDimension());
597 std::vector<bool> sign (this->innerDimension());
598 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
599 // Get binary representation of significands of normalised row/column.
600 computeFixedPointRepresentationVector(fraction, sign, outer);
601
602 // Create bitmask.
603 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
604
605 // Perform the split.
606 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
607 // NOTE: I could have a special path for 0.
608 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
609 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
610 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
611 for (size_t slice = 0; slice < numSplits; slice++) {
612 if (exponentDifference > (signed)bitsPerSlice) {
613 exponentDifference -= bitsPerSlice;
614 } else {
615 shiftCounter += exponentDifference;
616 exponentDifference = 0;
617 uint_t bitmask = shiftCounter > 0 ?
618 smallBitmask << shiftCounter :
619 smallBitmask >> -shiftCounter;
620 uint_t currentSlice = fraction[inner] & bitmask;
621 uint_t currentSplit = shiftCounter > 0 ?
622 currentSlice >> shiftCounter :
623 currentSlice << -shiftCounter;
624 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
625 this->splitValue(outer, inner, slice) = value;
626 shiftCounter -= bitsPerSlice;
627 }
628 }
629 }
630 }

644 void computeSplitsWithUnsignedEncoding() {
645 this->splitType = splittingStrategy::unsignedEncoding;
646 // Compute splits one row/column at a time.
647 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
648 auto bitsPerSlice = this->bitsPerSlice;
649 std::vector<uint_t> fraction (this->innerDimension());
650 std::vector<bool> sign (this->innerDimension());
651 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
652 // Get binary representation of significands of normalised row/column.
653 computeFixedPointRepresentationVector(fraction, sign, outer);
654
655 // Create bitmasks.
656 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
657 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
658 // is to keep the algorithm consistent with the bitmasking approach
659 // above.
660 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
661 // is not mentioned in sabb26, but it seems necessary to do this to
662 // ensure that the first (signed) slice does not overflow if the full
663 // bitwidth of splitint_t is used and the second slice is negative in
664 // two's complement.
665 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
666 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
667
668 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
669
670 auto matrixIndex = operandIndex(outer, inner);
671
672 // NOTE: I could have a special path for 0.
673 int16_t shiftCounter;
674 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
675 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
676
677 splitint_t value = 0;
678 uint_t remainder = 0;
679
680 // Slice 0.
681 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
682 // Value is too small to contribute explicit bits to slice 0.
683 value = sign[inner] ? -1 : 0;
684 remainder = sign[inner] ?
685 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
686 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
687 fraction[inner];
688 exponentDifference -= (bitsPerSlice - 1);
689 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
690 } else {
691 // Truncation.
692 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
693 exponentDifference = 0;
694 uint_t bitmask = (shiftCounter > 0)
695 ? (smallBitmask << shiftCounter)
696 : (smallBitmask >> -shiftCounter);
697 uint_t currentSlice = fraction[inner] & bitmask;
698 uint_t currentSplit = (shiftCounter > 0)
699 ? (currentSlice >> shiftCounter)
700 : (currentSlice << -shiftCounter);
701
702 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
703
704 // Rounding.
705 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
706 uint_t lowBits = fraction[inner] & lowMask;
707
708 if (lowBits != 0) {
709 value += (sign[inner] ? -1 : 0);
710 }
711 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
712 shiftCounter -= (bitsPerSlice + 1);
713 }
714 this->memory[matrixIndex] = value;
715
716 // Remaining slices.
717 const auto width = bitsPerSlice + 1;
718 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
719 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
720 const auto digitMax = cutoff - 1; // 2^b - 1
721 const auto digitMin = -cutoff; // -2^b
722 for (size_t slice = 1; slice < numSplits; slice++) {
723 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
724 exponentDifference -= (bitsPerSlice + 1);
725 if (sign[inner]) {
726 this->splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
727 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
728 }
729 } else {
730 shiftCounter += exponentDifference;
731 exponentDifference = 0;
732
733 uint_t bitmask = shiftCounter > 0 ?
734 largeBitmask << shiftCounter :
735 largeBitmask >> -shiftCounter;
736
737 uint_t currentSlice = remainder & bitmask;
738 uint_t currentSplit = shiftCounter > 0 ?
739 currentSlice >> shiftCounter :
740 currentSlice << -shiftCounter;
741
742 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
743 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
744 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
745 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
746 currentSplit |= maskToAdd;
747 }
748
749 // Update previous slices if currentSplit is negative in splitint_t.
750 // Carry must propoagate upward until slice 0 if necessary.
751 if (currentSplit > static_cast<uint_t>(digitMax)) {
752 int prevSliceIndex = static_cast<int>(slice) - 1;
753 while (true) {
754 auto newValue = static_cast<wideint_t>(this->splitValue(outer, inner, prevSliceIndex)) + 1;
755 if (newValue <= digitMax) {
756 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
757 break;
758 } else {
759 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
760 prevSliceIndex--;
761 }
762 }
763 }
764
765 // Store the slice as a signed (b+1)-bit value:
766 // [0, 2^b-1] -> unchanged
767 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
768 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
769 if (signedDigit >= cutoff) {
770 signedDigit -= base;
771 }
772
773 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
774 shiftCounter -= (bitsPerSlice + 1);
775 }
776 }
777 }
778 }
779 }

790 void computeSplitsWithRoundToNearest() {
791 this->splitType = splittingStrategy::roundToNearest;
792 auto bitsPerSlice = this->bitsPerSlice;
793 auto localMatrix = std::vector<fp_t>(this->matrix.data, this->matrix.data + this->matrix.size());
794 for (size_t slice = 0; slice < numSplits; slice++) {
795 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
796 fp_t sigma = ldexp(0.75,
797 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
798 powersVector[outer];
799 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
800 auto matrixIndex = operandIndex(outer, inner);
801 auto value = (localMatrix[matrixIndex] + sigma);
802 value -= sigma;
803 localMatrix[matrixIndex] -= value;
804 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
805 this->splitValue(outer, inner, slice) = value;
806 }
807 }
808 }
809 }

859struct multiplicationSchedule {
860 size_t numSplitsA; // number of slices for A
861 size_t numSplitsB; // number of slices for B
862
863 std::vector<bool> mask;
864 bool operator()(size_t i, size_t j) const {
865 return mask[i * numSplitsB + j];
866 }
867};
868
869
870inline multiplicationSchedule makeSchedule(const config& config) {
871 multiplicationSchedule sched;
872 sched.numSplitsA = config.numSplitsA;
873 sched.numSplitsB = config.numSplitsB;
874 sched.mask.resize(config.numSplitsA * config.numSplitsB, false);
875
876 std::visit([&](auto&& spec) {
877 using T = std::decay_t<decltype(spec)>;
878
879 // Case 1: predefined strategy (full or reduced)
880 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
881
882 if (spec == multiplicationStrategy::full) {
883 std::fill(sched.mask.begin(), sched.mask.end(), true);
884 }
885 else { // reduced (anti‑diagonal rule)
886 size_t limit = std::max(config.numSplitsA, config.numSplitsB) - 1;
887
888 for (size_t i = 0; i < config.numSplitsA; ++i)
889 for (size_t j = 0; j < config.numSplitsB; ++j)
890 if (i + j <= limit)
891 sched.mask[i * config.numSplitsB + j] = true;
892 }
893 }
894
895 // Case 2: custom mask
896 else if constexpr (std::is_same_v<T, std::vector<bool>>) {
897
898 const auto& mask = spec;
899
900 if (mask.size() != config.numSplitsA * config.numSplitsB)
901 throw std::invalid_argument("Mask size mismatch.");
902
903 sched.mask = mask; // row‑major copy
904 }
905
906 }, config.multSpecification);
907
908 return sched;
909}

932template <typename splitint_t, typename accumulator_t, typename fp_t>
933std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const Decomposition<splitint_t, fp_t> &A,
934 const Decomposition<splitint_t, fp_t> &B,
935 const multiplicationSchedule &sched,
936 const matrixLayout layoutC) {
937 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
938 for (size_t diagonal = 0; diagonal <= A.numSplits + B.numSplits - 1; diagonal++) {
939 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
940 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
941 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
942 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
943 if (sched(Aindex, Bindex)) {
944 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
945 for (size_t row = 0; row < A.rows(); row++) {
946 for (size_t col = 0; col < B.cols(); col++) {
947 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
948 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
949 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
950 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
951 C[index] += scaledSum * scalingFactor;
952 }
953 }
954 }
955 Aindex--;
956 Bindex++;
957 }
958 }
959
960 return C;
961}

984template <typename splitint_t, typename accumulator_t, typename fp_t>
985std::vector<fp_t> computeProductsWithIntegerAccumulation(const Decomposition<splitint_t, fp_t> &A,
986 const Decomposition<splitint_t, fp_t> &B,
987 const multiplicationSchedule &sched,
988 const matrixLayout layoutC) {
989
990 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
991 for (size_t diagonal = 0; diagonal <= A.numSplits + B.numSplits - 1; diagonal++) {
992 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
993 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
994
995 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
996
997 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
998 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
999 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
1000 if (sched(Aindex, Bindex))
1001 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1002 Aindex--;
1003 Bindex++;
1004 }
1005
1006 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
1007 for (size_t row = 0; row < A.rows(); row++) {
1008 for (size_t col = 0; col < B.cols(); col++) {

1036std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const matrixLayout layoutA,
1037 const std::vector<fp_t> &B, const matrixLayout layoutB,
1038 const size_t m, const size_t k, const size_t n,
1039 const matrixLayout layoutC,
1040 const multiterm::config &config) {

1042
1043 // Validate configuration parameters.
1044 multiterm::validateConfig(config);
1045
1046 // Derive bitsPerSlice.
1047 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
1048 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
1049 if (bitsPerInteger > bitsInAccumulator / 2) {
1050 throw std::invalid_argument("Split integer type is too wide for the chosen accumulator type");
1051 }
1052 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
1053 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
1054
1055 // Slice operands.
1056 auto viewA = makeMatrixView(A, m, k, layoutA);
1057 auto viewB = makeMatrixView(B, k, n, layoutB);

1061
1062 splitA.prepare();
1063 splitB.prepare();
1064
1065 // Build multiplication schedule.
1066 auto multiplicationSchedule = multiterm::makeSchedule(config);
1067
1068 // Execute multiplication based on reduction type.
1069 if (config.redType == multiterm::reductionStrategy::floatingPoint) {
1070 return multiterm::computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(
1071 splitA, splitB, multiplicationSchedule, layoutC);
1072 } else {
1073 return multiterm::computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(
1074 splitA, splitB, multiplicationSchedule, layoutC);
1075 }
