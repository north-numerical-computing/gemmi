178 void computeNormalisationVectors() {
179 // Compute normalisation vector.
180 const size_t iStride = this->iStride();
181 const size_t jStride = this->jStride();
182 for (size_t i = 0; i < this->otherDimension(); i++) {
183 this->powersVector[i] = 0.0;
184 for (size_t j = 0; j < this->innerProductDimension(); j++) {
185 const size_t index = i * iStride + j * jStride;
186 this->powersVector[i] = std::max(this->powersVector[i],
187 std::abs(this->matrix[index]));
188 }
189 // Compute the smallest power of 2 that is strictly greater than the
190 // maximum value in the row/column.
191 // NOTE 1: This is not the technique used in uoi24.
192 // NOTE 2: I use exponents instead of powers of 2, as I need the former
193 // to shift correctly.
194 frexp(this->powersVector[i], this->scalingExponents.data() + i);
195 this->powersVector[i] = std::ldexp(1.0, this->scalingExponents[i]);
196 }

204 int computeSliceBitOffset(size_t slice) const {
205 switch (splitType) {
206 case splittingStrategy::bitMasking:
207 // Slice k -> k * b
208 return static_cast<int>((slice + 1) * bitsPerSlice);
209
210 case splittingStrategy::roundToNearest:
211 // Slice 0 -> b - 1
212 // Slice k -> k * b - 1
213 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
214
215 case splittingStrategy::unsignedEncoding:
216 // Slice 0 -> b - 1
217 // Slice k -> (b - 1) + k * (b + 1)
218 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));
219 default:

231 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t i) {
232 auto numExpBits = computeNumExpBits<fp_t>();
233 auto numFracBits = computeNumFracBits<fp_t>();
234 auto iStride = this->iStride();
235 auto jStride = this->jStride();
236 for (size_t j = 0; j < this->innerProductDimension(); j++) {
237 const size_t index = i * iStride + j * jStride;
238 fp_t value = this->matrix[index]; // powersVector[i];
239 fraction[j] = std::bit_cast<uint_t>(value); // To bitstring.
240 sign[j] = std::signbit(value); // Extract sign.
241 uint_t bitmask = (~((uint_t)(0))) >> (numExpBits + 1);
242 fraction[j] = fraction[j] & bitmask; // Remove exponent and sign.
243 // Restore implicit bit for normal numbers.
244 // NOTE: NaNs and infs are currently not supported.
245 if (std::fpclassify(value) == FP_NORMAL)
246 fraction[j] |= ((uint_t)1 << (numFracBits - 1));
247 }
248 }

259 void computeSplitsWithBitMasking() {
260 this->splitType = splittingStrategy::bitMasking;
261 // Compute splits one row/column at a time.
262 auto numFracBits = computeNumFracBits<fp_t>();
263 auto bitsPerSlice = this->bitsPerSlice;
264 auto iStride = this->iStride();
265 auto jStride = this->jStride();
266 std::vector<uint_t> fraction (this->innerProductDimension());
267 std::vector<bool> sign (this->innerProductDimension());
268 for (size_t i = 0; i < this->otherDimension(); i++) {
269 // Get binary representation of significands of normalised row/column.
270 computeFixedPointRepresentationVector(fraction, sign, i);
271
272 // Create bitmask.
273 const uint_t smallBitmask = (1 << bitsPerSlice) - 1;
274 // Perform the split.
275 for (size_t j = 0; j < this->innerProductDimension(); j++) {
276 int16_t shiftCounter = numFracBits - bitsPerSlice;
277 int currentExponent;
278 frexp(this->matrix[i * iStride + j * jStride], &currentExponent);
279 int16_t exponentDifference = scalingExponents[i] - currentExponent;
280 for (size_t slice = 0; slice < numSplits; slice++) {
281 if (exponentDifference > (signed)bitsPerSlice) {
282 exponentDifference -= bitsPerSlice;
283 } else {
284 shiftCounter += exponentDifference;
285 exponentDifference = 0;
286 uint_t bitmask = shiftCounter > 0 ?
287 smallBitmask << shiftCounter :
288 smallBitmask >> -shiftCounter;
289 uint_t currentSlice = fraction[j] & bitmask;
290 uint_t currentSplit = shiftCounter > 0 ?
291 currentSlice >> shiftCounter :
292 currentSlice << -shiftCounter;
293 splitint_t value = (splitint_t)(currentSplit) * (sign[j] ? -1 : 1);
294 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = value;
295 shiftCounter -= bitsPerSlice;
296 }
297 }
298 }
299 }

313 void computeSplitsWithUnsignedEncoding() {
314 this->splitType = splittingStrategy::unsignedEncoding;
315 // Compute splits one row/column at a time.
316 auto numFracBits = computeNumFracBits<fp_t>();
317 auto bitsPerSlice = this->bitsPerSlice;
318 auto iStride = this->iStride();
319 auto jStride = this->jStride();
320 std::vector<uint_t> fraction (this->innerProductDimension());
321 std::vector<bool> sign (this->innerProductDimension());
322 for (size_t i = 0; i < this->otherDimension(); i++) {
323 // Get binary representation of significands of normalised row/column.
324 computeFixedPointRepresentationVector(fraction, sign, i);
325
326 // Create bitmasks.
327 // This algorithm uses bitsPerSlice bits for the first slice and bitsPerSlice + 1 bits for subsequent slices.
328 // This choice is to keep the algorithm consistent with the bitmasking approach above.
329 // NOTE: I am using a first slice with bitsPerSlice - 1 bits. This is not mentioned in sabb26, but it seems
330 // necessary to do this, if I want to ensure that the first slice does not overflow if the second slice
331 // is negative in two's complement.
332 const uint_t smallBitmask = (1 << (bitsPerSlice - 1)) - 1;
333 const uint_t largeBitmask = (1 << (bitsPerSlice + 1)) - 1;
334
335 for (size_t j = 0; j < this->innerProductDimension(); j++) {
336
337 int currentExponent;
338 frexp(this->matrix[i * iStride + j * jStride], &currentExponent);
339 int16_t exponentDifference = scalingExponents[i] - currentExponent;
340
341 // Boundary below slice 0, fully aligned to this entry.
342 int16_t firstShift = numFracBits - (bitsPerSlice - 1) + exponentDifference;
343
344 splitint_t value = 0;
345 uint_t remainder = 0;
346
347 if (!sign[j]) {
348 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
349 // Too small to contribute explicit bits to slice 0.
350 value = 0;
351 remainder = fraction[j];
352 } else {
353 // Positive number: leading digit is truncation.
354 uint_t bitmask = (firstShift > 0)
355 ? (smallBitmask << firstShift)
356 : (smallBitmask >> -firstShift);
357
358 uint_t currentSlice = fraction[j] & bitmask;
359 uint_t currentSplit = (firstShift > 0)
360 ? (currentSlice >> firstShift)
361 : (currentSlice << -firstShift);
362
363 value = static_cast<splitint_t>(currentSplit);
364 remainder = fraction[j] & ((uint_t(1) << firstShift) - 1);
365 }
366 } else {
367 // Negative number.
368 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
369 // Too small to contribute explicit bits to slice 0:
370 // round toward -infinity gives -1.
371 value = -1;
372 remainder = ((1ull << (sizeof(uint_t) * 8 - 1)) - fraction[j]) | (uint_t(1) << (sizeof(uint_t) * 8 - 1));
373 } else {
374 // Normal truncation path for the leading slice.
375 uint_t bitmask = (firstShift > 0)
376 ? (smallBitmask << firstShift)
377 : (smallBitmask >> -firstShift);
378
379 uint_t currentSlice = fraction[j] & bitmask;
380 uint_t currentSplit = (firstShift > 0)
381 ? (currentSlice >> firstShift)
382 : (currentSlice << -firstShift);
383
384 value = -static_cast<splitint_t>(currentSplit);
385
386 uint_t lowMask = (uint_t(1) << firstShift) - 1;
387 uint_t lowBits = fraction[j] & lowMask;
388
389 if (lowBits != 0) {
390 value -= 1; // round toward -infinity
391 remainder = (1ull << firstShift) - lowBits;
392 } else {
393 remainder = 0;
394 }
395 }
396 }
397
398 this->memory[i * iStride + j * jStride] = value;
399
400 // From here on, the remainder is already aligned and nonnegative.
401 int shiftCounter = firstShift - (bitsPerSlice + 1);
402 exponentDifference = 0;
403
404 // Split (nonnegative) remainder with unsigned slice encoding.
405 for (size_t slice = 1; slice < numSplits; slice++) {
406 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
407 exponentDifference -= (bitsPerSlice + 1);
408 if (sign[j]) {
409 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = static_cast<splitint_t>(largeBitmask);
410 }
411 } else {
412 shiftCounter += exponentDifference;
413 exponentDifference = 0;
414
415 uint_t bitmask = shiftCounter > 0 ?
416 largeBitmask << shiftCounter :
417 largeBitmask >> -shiftCounter;
418
419 uint_t currentSlice = remainder & bitmask;
420 uint_t currentSplit = shiftCounter > 0 ?
421 currentSlice >> shiftCounter :
422 currentSlice << -shiftCounter;
423
424 if (sign[j] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
425 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
426 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
427 uint_t maskToAdd = ((1ull << bitsLost) - 1) << bitsUsed;
428 currentSplit |= maskToAdd;
429 }
430
431 // Width of sub-leading slices is (bitsPerSlice + 1) bits.
432 const auto width = bitsPerSlice + 1;
433 const auto cutoff = static_cast<wideint_t>(uint_t(1) << bitsPerSlice); // 2^b
434 const auto base = static_cast<wideint_t>(uint_t(1) << width); // 2^(b+1)
435 const auto digitMax = cutoff - 1; // 2^b - 1
436 const auto digitMin = -cutoff; // -2^b
437
438 // If the unsigned digit is in the upper half, propagate carry upward.
439 if (currentSplit >= static_cast<uint_t>(cutoff)) {
440 int nextSlice = static_cast<int>(slice) - 1;
441 while (true) {
442 const size_t prevIdx =
443 i * iStride + j * jStride + nextSlice * this->matrix.size();
444
445 auto newValue =
446 static_cast<wideint_t>(this->memory[prevIdx]) + 1;
447
448 if (newValue <= digitMax) {
449 this->memory[prevIdx] = static_cast<splitint_t>(newValue);
450 break;
451 } else {
452 this->memory[prevIdx] = static_cast<splitint_t>(digitMin);
453 nextSlice--;
454 }
455 }
456 }
457
458 // Store the slice as a signed (b+1)-bit value:
459 // [0, 2^b-1] -> unchanged
460 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
461 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
462 if (signedDigit >= cutoff) {
463 signedDigit -= base;
464 }
465
466 this->memory[
467 i * iStride + j * jStride + slice * this->matrix.size()
468 ] = static_cast<splitint_t>(signedDigit);
469
470 shiftCounter -= (bitsPerSlice + 1);
471 }
472 }
473 }
474 }
475 }

486 void computeSplitsWithRoundToNearest() {
487 this->splitType = splittingStrategy::roundToNearest;
488 auto bitsPerSlice = this->bitsPerSlice;
489 auto iStride = this->iStride();
490 auto jStride = this->jStride();
491 auto localMatrix = this->matrix;
492 for (size_t slice = 0; slice < numSplits; slice++) {
493 for (size_t i = 0; i < this->otherDimension(); i++) {
494 fp_t sigma = ldexp(0.75, computeNumFracBits<fp_t>() - bitsPerSlice * slice + 1 - bitsPerSlice) * powersVector[i];
495 for (size_t j = 0; j < this->innerProductDimension(); j++) {
496 auto value = (localMatrix[i * iStride + j * jStride] + sigma);
497 value -= sigma;
498 localMatrix[i * iStride + j * jStride] -= value;
499 value = value / powersVector[i] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
500 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = value;
501 }
502 }
503 }
504 }

552template <typename splitint_t, typename accumulator_t, typename fp_t>
553std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
554 const MatrixSplit<splitint_t, fp_t> &B,
555 const size_t numDiagonals) {
556
557 std::vector<fp_t > C (A.m * B.n);
558
559 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
560 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
561 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
562 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
563 std::vector<accumulator_t> accumulator (A.m * B.n, 0.0);
564 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, Aindex, Bindex);
565 for (size_t i = 0; i < A.m; i++) {
566 for (size_t j = 0; j < B.n; j++) {
567 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
568 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[i + j * A.m]), -totalShift);
569 fp_t scalingFactor = A.powersVector[i] * B.powersVector[j];
570 C[i + j * A.m] += scaledSum * scalingFactor;
571 }
572 }
573 Aindex--;
574 Bindex++;
575 }
576 }
577
578 return C;
579}

601template <typename splitint_t, typename accumulator_t, typename fp_t>
602std::vector<fp_t> computeProductsWithIntegerAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
603 const MatrixSplit<splitint_t, fp_t> &B,
604 const size_t numDiagonals) {
605 std::vector<fp_t> C(A.m * B.n);
606
607 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
608 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
609 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
610
611 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
612
613 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
614 std::vector<accumulator_t> accumulator(A.m * B.n, 0);
615 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
616 // assert(A.sliceBitOffset(static_cast<size_t>(Aindex)) + B.sliceBitOffset(Bindex) == totalShift);
617
618 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, Aindex, Bindex);
619 Aindex--;
620 Bindex++;
621 }
622
623 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
624 for (size_t i = 0; i < A.m; i++) {
625 for (size_t j = 0; j < B.n; j++) {
626 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[i + j * A.m]), -totalShift);
627 fp_t scalingFactor = A.powersVector[i] * B.powersVector[j];
628 C[i + j * A.m] += scaledSum * scalingFactor;
629 }
630 }
631 }
632
633 return C;
634}

653template <typename fp_t, typename splitint_t, typename accumulator_t>
654std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const std::vector<fp_t> &B,
655 const size_t m, const size_t k, const size_t n,
656 const size_t numSplitsA, const size_t numSplitsB,
657 const splittingStrategy splitType = splittingStrategy::roundToNearest,
658 const accumulationStrategy accType = accumulationStrategy::floatingPoint,
659 const multiplicationStrategy multType = multiplicationStrategy::reduced) {
660
661 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
662 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
663 assert(bitsPerInteger <= bitsInAccumulator / 2);
664 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
665 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
666
667 auto splitA = MatrixSplit<splitint_t, fp_t>(A, m, k, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
668 auto splitB = MatrixSplit<splitint_t, fp_t>(B, k, n, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
669
670 size_t numDiagonals;
671 switch (multType) {
672 case multiplicationStrategy::reduced:
673 // Products below the main anti-diagonal are ignored.
674 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
675 break;
676 case multiplicationStrategy::full:
677 // All products are computed.
678 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
679 break;
680 default:
681 UNREACHABLE();
682 }
683
684 switch (accType) {
685 case accumulationStrategy::floatingPoint:
686 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, numDiagonals);
687 case accumulationStrategy::integer:
688 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, numDiagonals);
689 default:
690 UNREACHABLE();
691 }
692}
