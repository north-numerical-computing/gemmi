530 void computeNormalisationVectors() {
531 // Compute normalisation vector.
532 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
533 this->powersVector[outer] = 0.0;
534 for (size_t j = 0; j < this->innerDimension(); j++) {
535 this->powersVector[outer] = std::max(this->powersVector[outer],
536 std::abs(this->operand(outer, j)));
537 }
538 // Compute the smallest power of 2 that is strictly greater than the
539 // maximum value in the row/column.
540 // NOTE 1: This is not the technique used in uoi24.
541 // NOTE 2: I use exponents instead of powers of 2, as I need the former
542 // to shift correctly.
543 this->scalingExponents[outer] = getStoredFloatingPointExponent(this->powersVector[outer]);
544 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);
545 }

553 int computeSliceBitOffset(size_t slice) const {
554 switch (splitType) {
555 case splittingStrategy::truncation:
556 // Slice k -> k * b
557 return static_cast<int>((slice + 1) * bitsPerSlice);
558
559 case splittingStrategy::roundToNearest:
560 // Slice 0 -> b - 1
561 // Slice k -> k * b - 1
562 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
563
564 case splittingStrategy::unsignedEncoding:
565 // Slice 0 -> b - 1
566 // Slice k -> (b - 1) + k * (b + 1)
567 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));
568
569 // LCOV_EXCL_START
570 default:
571 throw std::logic_error("Unhandled splittingStrategy");
572 // LCOV_EXCL_STOP
573 }

583 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
584 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
585 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
586 fp_t value = this->operand(outer, inner);
587 fraction[inner] = std::bit_cast<uint_t>(value);
588 sign[inner] = std::signbit(value);
589 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
590 fraction[inner] = fraction[inner] & bitmask;
591 // Restore implicit bit for normal numbers.
592 // NOTE: NaNs and infs are currently not supported.
593 if (std::fpclassify(value) == FP_NORMAL)
594 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
595 }
596 }

607 void computeSplitsWithTruncation() {
608 this->splitType = splittingStrategy::truncation;
609 // Compute splits one row/column at a time.
610 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
611 auto bitsPerSlice = this->bitsPerSlice;
612 std::vector<uint_t> fraction (this->innerDimension());
613 std::vector<bool> sign (this->innerDimension());
614 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
615 // Get binary representation of significands of normalised row/column.
616 computeFixedPointRepresentationVector(fraction, sign, outer);
617
618 // Create bitmask.
619 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
620
621 // Perform the split.
622 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
623 // NOTE: I could have a special path for 0.
624 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
625 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
626 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
627 for (size_t slice = 0; slice < numSplits; slice++) {
628 if (exponentDifference > (signed)bitsPerSlice) {
629 exponentDifference -= bitsPerSlice;
630 } else {
631 shiftCounter += exponentDifference;
632 exponentDifference = 0;
633 uint_t bitmask = shiftCounter > 0 ?
634 smallBitmask << shiftCounter :
635 smallBitmask >> -shiftCounter;
636 uint_t currentSlice = fraction[inner] & bitmask;
637 uint_t currentSplit = shiftCounter > 0 ?
638 currentSlice >> shiftCounter :
639 currentSlice << -shiftCounter;
640 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
641 this->splitValue(outer, inner, slice) = value;
642 shiftCounter -= bitsPerSlice;
643 }
644 }
645 }
646 }

660 void computeSplitsWithUnsignedEncoding() {
661 this->splitType = splittingStrategy::unsignedEncoding;
662 // Compute splits one row/column at a time.
663 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
664 auto bitsPerSlice = this->bitsPerSlice;
665 std::vector<uint_t> fraction (this->innerDimension());
666 std::vector<bool> sign (this->innerDimension());
667 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
668 // Get binary representation of significands of normalised row/column.
669 computeFixedPointRepresentationVector(fraction, sign, outer);
670
671 // Create bitmasks.
672 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
673 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
674 // is to keep the algorithm consistent with the bitmasking approach
675 // above.
676 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
677 // is not mentioned in sabb26, but it seems necessary to do this to
678 // ensure that the first (signed) slice does not overflow if the full
679 // bitwidth of splitint_t is used and the second slice is negative in
680 // two's complement.
681 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
682 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
683
684 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
685
686 auto matrixIndex = operandIndex(outer, inner);
687
688 // NOTE: I could have a special path for 0.
689 int16_t shiftCounter;
690 int currentExponent = getStoredFloatingPointExponent(this->operand(outer, inner));
691 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
692
693 splitint_t value = 0;
694 uint_t remainder = 0;
695
696 // Slice 0.
697 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
698 // Value is too small to contribute explicit bits to slice 0.
699 value = sign[inner] ? -1 : 0;
700 remainder = sign[inner] ?
701 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
702 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
703 fraction[inner];
704 exponentDifference -= (bitsPerSlice - 1);
705 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
706 } else {
707 // Truncation.
708 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
709 exponentDifference = 0;
710 uint_t bitmask = (shiftCounter > 0)
711 ? (smallBitmask << shiftCounter)
712 : (smallBitmask >> -shiftCounter);
713 uint_t currentSlice = fraction[inner] & bitmask;
714 uint_t currentSplit = (shiftCounter > 0)
715 ? (currentSlice >> shiftCounter)
716 : (currentSlice << -shiftCounter);
717
718 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
719
720 // Rounding.
721 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
722 uint_t lowBits = fraction[inner] & lowMask;
723
724 if (lowBits != 0) {
725 value += (sign[inner] ? -1 : 0);
726 }
727 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
728 shiftCounter -= (bitsPerSlice + 1);
729 }
730 this->memory[matrixIndex] = value;
731
732 // Remaining slices.
733 const auto width = bitsPerSlice + 1;
734 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
735 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
736 const auto digitMax = cutoff - 1; // 2^b - 1
737 const auto digitMin = -cutoff; // -2^b
738 for (size_t slice = 1; slice < numSplits; slice++) {
739 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
740 exponentDifference -= (bitsPerSlice + 1);
741 if (sign[inner]) {
742 this->splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
743 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
744 }
745 } else {
746 shiftCounter += exponentDifference;
747 exponentDifference = 0;
748
749 uint_t bitmask = shiftCounter > 0 ?
750 largeBitmask << shiftCounter :
751 largeBitmask >> -shiftCounter;
752
753 uint_t currentSlice = remainder & bitmask;
754 uint_t currentSplit = shiftCounter > 0 ?
755 currentSlice >> shiftCounter :
756 currentSlice << -shiftCounter;
757
758 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
759 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
760 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
761 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
762 currentSplit |= maskToAdd;
763 }
764
765 // Update previous slices if currentSplit is negative in splitint_t.
766 // Carry must propoagate upward until slice 0 if necessary.
767 if (currentSplit > static_cast<uint_t>(digitMax)) {
768 int prevSliceIndex = static_cast<int>(slice) - 1;
769 while (true) {
770 auto newValue = static_cast<wideint_t>(this->splitValue(outer, inner, prevSliceIndex)) + 1;
771 if (newValue <= digitMax) {
772 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
773 break;
774 } else {
775 this->splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
776 prevSliceIndex--;
777 }
778 }
779 }
780
781 // Store the slice as a signed (b+1)-bit value:
782 // [0, 2^b-1] -> unchanged
783 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
784 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
785 if (signedDigit >= cutoff) {
786 signedDigit -= base;
787 }
788
789 this->splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
790 shiftCounter -= (bitsPerSlice + 1);
791 }
792 }
793 }
794 }
795 }

806 void computeSplitsWithRoundToNearest() {
807 this->splitType = splittingStrategy::roundToNearest;
808 auto bitsPerSlice = this->bitsPerSlice;
809 auto localMatrix = std::vector<fp_t>(this->matrix.data, this->matrix.data + this->matrix.size());
810 for (size_t slice = 0; slice < numSplits; slice++) {
811 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
812 fp_t sigma = ldexp(0.75,
813 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
814 powersVector[outer];
815 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
816 auto matrixIndex = operandIndex(outer, inner);
817 auto value = (localMatrix[matrixIndex] + sigma);
818 value -= sigma;
819 localMatrix[matrixIndex] -= value;
820 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
821 this->splitValue(outer, inner, slice) = value;
822 }
823 }
824 }

852 struct multiplicationSchedule {
853 size_t numSplitsA; // number of slices for A
854 size_t numSplitsB; // number of slices for B
855
856 std::vector<bool> mask;
857 bool operator()(size_t i, size_t j) const {
858 return mask[i * numSplitsB + j];
859 }
860};
861
870inline multiplicationSchedule makeSchedule(const config& config) {
871 multiplicationSchedule sched;
872 sched.numSplitsA = config.numSplitsA;
873 sched.numSplitsB = config.numSplitsB;
874 sched.mask.resize(config.numSplitsA * config.numSplitsB, false);
875
876 std::visit([&](auto&& spec) {
877 using T = std::decay_t<decltype(spec)>;
878
879 // Predefined strategy (full or reduced).
880 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
881
882 if (spec == multiplicationStrategy::full) {
883 std::fill(sched.mask.begin(), sched.mask.end(), true);
884 }
885 else { // reduced (anti‑diagonal rule)
886 size_t limit = std::max(config.numSplitsA, config.numSplitsB) - 1;
887
888 for (size_t i = 0; i < config.numSplitsA; ++i)
889 for (size_t j = 0; j < config.numSplitsB; ++j)
890 if (i + j <= limit)
891 sched.mask[i * config.numSplitsB + j] = true;
892 }
893 }
894
895 // Custom boolean mask.
896 else if constexpr (std::is_same_v<T, std::vector<bool>>) {
897
898 const auto& mask = spec;
899
900 if (mask.size() != config.numSplitsA * config.numSplitsB)
901 throw std::invalid_argument("Mask size mismatch.");
902
903 sched.mask = mask; // row‑major copy
904 }
905
906 }, config.multSpecification);
907
908 return sched;
909}

958template <typename splitint_t, typename accumulator_t, typename fp_t>
959std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const Decomposition<splitint_t, fp_t> &A,
960 const Decomposition<splitint_t, fp_t> &B,
961 const multiplicationSchedule &sched,
962 const matrixLayout layoutC) {
963 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
964 for (size_t diagonal = 0; diagonal <= A.numSplits + B.numSplits - 1; diagonal++) {
965 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
966 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
967 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
968 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
969 if (sched(Aindex, Bindex)) {
970 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
971 for (size_t row = 0; row < A.rows(); row++) {
972 for (size_t col = 0; col < B.cols(); col++) {
973 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
974 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
975 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
976 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
977 C[index] += scaledSum * scalingFactor;
978 }
979 }
980 }
981 Aindex--;
982 Bindex++;
983 }
984 }
985
986 return C;
987}

1011std::vector<fp_t> computeProductsWithIntegerAccumulation(const Decomposition<splitint_t, fp_t> &A,
1012 const Decomposition<splitint_t, fp_t> &B,
1013 const multiplicationSchedule &sched,
1014 const matrixLayout layoutC) {

1017 for (size_t diagonal = 0; diagonal <= A.numSplits + B.numSplits - 1; diagonal++) {
1018 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);

1025 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
1026 if (sched(Aindex, Bindex))
1027 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
1028 Aindex--;
1029 Bindex++;
1030 }
1031
1032 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
1033 for (size_t row = 0; row < A.rows(); row++) {
1034 for (size_t col = 0; col < B.cols(); col++) {
1035 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
1036 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
1037 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
1038 C[index] += scaledSum * scalingFactor;
1039 }
1040 }
1041 }
1042
1043 return C;
1044}
