46 if (value == 0.0) {

176 void computeNormalisationVectors() {
177 // Compute normalisation vector.
178 const size_t iStride = this->iStride();
179 const size_t jStride = this->jStride();
180 for (size_t i = 0; i < this->otherDimension(); i++) {
181 this->powersVector[i] = 0.0;
182 for (size_t j = 0; j < this->innerProductDimension(); j++) {
183 const size_t matrixIndex = i * iStride + j * jStride;
184 this->powersVector[i] = std::max(this->powersVector[i],
185 std::abs(this->matrix[matrixIndex]));
186 }
187 // Compute the smallest power of 2 that is strictly greater than the
188 // maximum value in the row/column.
189 // NOTE 1: This is not the technique used in uoi24.
190 // NOTE 2: I use exponents instead of powers of 2, as I need the former
191 // to shift correctly.
192 this->scalingExponents[i] = getFloatingPointExponent(this->powersVector[i]);
193 this->powersVector[i] = std::ldexp(1.0, this->scalingExponents[i]);
194 }

232 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t i) {
233 auto numFracBits = computeNumFracBits<fp_t>();
234 auto iStride = this->iStride();
235 auto jStride = this->jStride();
236 for (size_t j = 0; j < this->innerProductDimension(); j++) {
237 const size_t matrixIndex = i * iStride + j * jStride;
238 fp_t value = this->matrix[matrixIndex];
239 fraction[j] = std::bit_cast<uint_t>(value);
240 sign[j] = std::signbit(value);
241 uint_t bitmask = (static_cast<uint_t>(1) << (numFracBits - 1)) - 1;
242 fraction[j] = fraction[j] & bitmask;
243 // Restore implicit bit for normal numbers.
244 // NOTE: NaNs and infs are currently not supported.
245 if (std::fpclassify(value) == FP_NORMAL)
246 fraction[j] |= (static_cast<uint_t>(1) << (numFracBits - 1));
247 }

259 void computeSplitsWithBitMasking() {
260 this->splitType = splittingStrategy::bitMasking;
261 // Compute splits one row/column at a time.
262 auto numFracBits = computeNumFracBits<fp_t>();
263 auto bitsPerSlice = this->bitsPerSlice;
264 auto iStride = this->iStride();
265 auto jStride = this->jStride();
266 std::vector<uint_t> fraction (this->innerProductDimension());
267 std::vector<bool> sign (this->innerProductDimension());
268 for (size_t i = 0; i < this->otherDimension(); i++) {
269 // Get binary representation of significands of normalised row/column.
270 computeFixedPointRepresentationVector(fraction, sign, i);
271
272 // Create bitmask.
273 const uint_t smallBitmask = (1 << bitsPerSlice) - 1;
274
275 // Perform the split.
276 for (size_t j = 0; j < this->innerProductDimension(); j++) {
277 // NOTE: I could have a special path for 0.
278 int16_t shiftCounter = numFracBits - bitsPerSlice;
279 int currentExponent = getFloatingPointExponent(this->matrix[i * iStride + j * jStride]);
280 int16_t exponentDifference = scalingExponents[i] - currentExponent;
281 for (size_t slice = 0; slice < numSplits; slice++) {
282 if (exponentDifference > (signed)bitsPerSlice) {
283 exponentDifference -= bitsPerSlice;
284 } else {
285 shiftCounter += exponentDifference;
286 exponentDifference = 0;
287 uint_t bitmask = shiftCounter > 0 ?
288 smallBitmask << shiftCounter :
289 smallBitmask >> -shiftCounter;
290 uint_t currentSlice = fraction[j] & bitmask;
291 uint_t currentSplit = shiftCounter > 0 ?
292 currentSlice >> shiftCounter :
293 currentSlice << -shiftCounter;
294 splitint_t value = (splitint_t)(currentSplit) * (sign[j] ? -1 : 1);
295 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = value;
296 shiftCounter -= bitsPerSlice;
297 }
298 }
299 }
300 }

314 void computeSplitsWithUnsignedEncoding() {
315 this->splitType = splittingStrategy::unsignedEncoding;
316 // Compute splits one row/column at a time.
317 auto numFracBits = computeNumFracBits<fp_t>();
318 auto bitsPerSlice = this->bitsPerSlice;
319 auto iStride = this->iStride();
320 auto jStride = this->jStride();
321 std::vector<uint_t> fraction (this->innerProductDimension());
322 std::vector<bool> sign (this->innerProductDimension());
323 for (size_t i = 0; i < this->otherDimension(); i++) {
324 // Get binary representation of significands of normalised row/column.
325 computeFixedPointRepresentationVector(fraction, sign, i);
326
327 // Create bitmasks.
328 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
329 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
330 // is to keep the algorithm consistent with the bitmasking approach
331 // above.
332 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
333 // is not mentioned in sabb26, but it seems necessary to do this to
334 // ensure that the first (signed) slice does not overflow if the full
335 // bitwidth of splitint_t is used and the second slice is negative in
336 // two's complement.
337 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
338 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
339
340 for (size_t j = 0; j < this->innerProductDimension(); j++) {
341
342 auto matrixIndex = i * iStride + j * jStride;
343
344 // NOTE: I could have a special path for 0.
345 int16_t shiftCounter;
346 int currentExponent = getFloatingPointExponent(this->matrix[matrixIndex]);
347 int16_t exponentDifference = scalingExponents[i] - currentExponent;
348
349 splitint_t value = 0;
350 uint_t remainder = 0;
351
352 // Slice 0.
353 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
354 // Value is too small to contribute explicit bits to slice 0.
355 value = sign[j] ? -1 : 0;
356 remainder = sign[j] ?
357 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[j]) |
358 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
359 fraction[j];
360 exponentDifference -= (bitsPerSlice - 1);
361 shiftCounter = numFracBits - (bitsPerSlice + 1);
362 } else {
363 // Truncation.
364 shiftCounter = numFracBits - (bitsPerSlice - 1) + exponentDifference;
365 exponentDifference = 0;
366 uint_t bitmask = (shiftCounter > 0)
367 ? (smallBitmask << shiftCounter)
368 : (smallBitmask >> -shiftCounter);
369 uint_t currentSlice = fraction[j] & bitmask;
370 uint_t currentSplit = (shiftCounter > 0)
371 ? (currentSlice >> shiftCounter)
372 : (currentSlice << -shiftCounter);
373
374 value = (sign[j] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
375
376 // Rounding.
377 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
378 uint_t lowBits = fraction[j] & lowMask;
379
380 if (lowBits != 0) {
381 value += (sign[j] ? -1 : 0);
382 }
383 remainder = sign[j] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
384 shiftCounter -= (bitsPerSlice + 1);
385 }
386 this->memory[matrixIndex] = value;
387
388 // Remaining slices.
389 const auto width = bitsPerSlice + 1;
390 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
391 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
392 const auto digitMax = cutoff - 1; // 2^b - 1
393 const auto digitMin = -cutoff; // -2^b
394 for (size_t slice = 1; slice < numSplits; slice++) {
395 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
396 exponentDifference -= (bitsPerSlice + 1);
397 if (sign[j]) {
398 this->memory[matrixIndex + (slice - 1) * this->matrix.size()] = static_cast<splitint_t>(0);
399 this->memory[matrixIndex + slice * this->matrix.size()] = static_cast<splitint_t>(-1);
400 }
401 } else {
402 shiftCounter += exponentDifference;
403 exponentDifference = 0;
404
405 uint_t bitmask = shiftCounter > 0 ?
406 largeBitmask << shiftCounter :
407 largeBitmask >> -shiftCounter;
408
409 uint_t currentSlice = remainder & bitmask;
410 uint_t currentSplit = shiftCounter > 0 ?
411 currentSlice >> shiftCounter :
412 currentSlice << -shiftCounter;
413
414 if (sign[j] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
415 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
416 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
417 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
418 currentSplit |= maskToAdd;
419 }
420
421 // Update previous slices if currentSplit is negative in splitint_t.
422 // Carry must propoagate upward until slice 0 if necessary.
423 if (currentSplit > static_cast<uint_t>(digitMax)) {
424 int nextSlice = static_cast<int>(slice) - 1;
425 while (true) {
426 const size_t prevIdx = matrixIndex + nextSlice * this->matrix.size();
427 auto newValue = static_cast<wideint_t>(this->memory[prevIdx]) + 1;
428
429 if (newValue <= digitMax) {
430 this->memory[prevIdx] = static_cast<splitint_t>(newValue);
431 break;
432 } else {
433 this->memory[prevIdx] = static_cast<splitint_t>(digitMin);
434 nextSlice--;
435 }
436 }
437 }
438
439 // Store the slice as a signed (b+1)-bit value:
440 // [0, 2^b-1] -> unchanged
441 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
442 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
443 if (signedDigit >= cutoff) {
444 signedDigit -= base;
445 }
446
447 this->memory[matrixIndex + slice * this->matrix.size()] = static_cast<splitint_t>(signedDigit);
448 shiftCounter -= (bitsPerSlice + 1);
449 }
450 }
451 }
452 }
453 }

464 void computeSplitsWithRoundToNearest() {
465 this->splitType = splittingStrategy::roundToNearest;
466 auto bitsPerSlice = this->bitsPerSlice;
467 auto iStride = this->iStride();
468 auto jStride = this->jStride();
469 auto localMatrix = this->matrix;
470 for (size_t slice = 0; slice < numSplits; slice++) {
471 for (size_t i = 0; i < this->otherDimension(); i++) {
472 fp_t sigma = ldexp(0.75, computeNumFracBits<fp_t>() - bitsPerSlice * slice + 1 - bitsPerSlice) * powersVector[i];
473 for (size_t j = 0; j < this->innerProductDimension(); j++) {
474 auto matrixIndex = i * iStride + j * jStride;
475 auto value = (localMatrix[matrixIndex] + sigma);
476 value -= sigma;
477 localMatrix[matrixIndex] -= value;
478 value = value / powersVector[i] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
479 this->memory[matrixIndex + slice * this->matrix.size()] = value;
480 }
481 }
482 }
483 }

532template <typename splitint_t, typename accumulator_t, typename fp_t>
533std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
534 const MatrixSplit<splitint_t, fp_t> &B,
535 const size_t numDiagonals) {
536
537 std::vector<fp_t > C (A.m * B.n);
538
539 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
540 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
541 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
542 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
543 std::vector<accumulator_t> accumulator (A.m * B.n, 0.0);
544 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, Aindex, Bindex);
545 for (size_t i = 0; i < A.m; i++) {
546 for (size_t j = 0; j < B.n; j++) {
547 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
548 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[i + j * A.m]), -totalShift);
549 fp_t scalingFactor = A.powersVector[i] * B.powersVector[j];
550 C[i + j * A.m] += scaledSum * scalingFactor;
551 }
552 }
553 Aindex--;
554 Bindex++;
555 }
556 }
557
558 return C;
559}

581template <typename splitint_t, typename accumulator_t, typename fp_t>
582std::vector<fp_t> computeProductsWithIntegerAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
583 const MatrixSplit<splitint_t, fp_t> &B,
584 const size_t numDiagonals) {
585 std::vector<fp_t> C(A.m * B.n);
586
587 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
588 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
589 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
590
591 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
592
593 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
594 std::vector<accumulator_t> accumulator(A.m * B.n, 0);
595 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
596 // assert(A.sliceBitOffset(static_cast<size_t>(Aindex)) + B.sliceBitOffset(Bindex) == totalShift);
597
598 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, Aindex, Bindex);
599 Aindex--;
600 Bindex++;
601 }
602
603 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
604 for (size_t i = 0; i < A.m; i++) {
605 for (size_t j = 0; j < B.n; j++) {
606 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[i + j * A.m]), -totalShift);
607 fp_t scalingFactor = A.powersVector[i] * B.powersVector[j];
608 C[i + j * A.m] += scaledSum * scalingFactor;
609 }
610 }
611 }
612
613 return C;
614}

633template <typename fp_t, typename splitint_t, typename accumulator_t>
634std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const std::vector<fp_t> &B,
635 const size_t m, const size_t k, const size_t n,
636 const size_t numSplitsA, const size_t numSplitsB,
637 const splittingStrategy splitType = splittingStrategy::roundToNearest,
638 const accumulationStrategy accType = accumulationStrategy::floatingPoint,
639 const multiplicationStrategy multType = multiplicationStrategy::reduced) {
640
641 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
642 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
643 assert(bitsPerInteger <= bitsInAccumulator / 2);
644 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
645 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
646
647 auto splitA = MatrixSplit<splitint_t, fp_t>(A, m, k, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
648 auto splitB = MatrixSplit<splitint_t, fp_t>(B, k, n, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
649
650 size_t numDiagonals;
651 switch (multType) {
652 case multiplicationStrategy::reduced:
653 // Products below the main anti-diagonal are ignored.
654 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
655 break;
656 case multiplicationStrategy::full:
657 // All products are computed.
658 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
659 break;
660 // LCOV_EXCL_START
661 default:
662
663 std::abort();
664 // LCOV_EXCL_STOP
665 }
666
667 switch (accType) {
668 case accumulationStrategy::floatingPoint:
669 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, numDiagonals);
670 case accumulationStrategy::integer:
671 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, numDiagonals);
672 // LCOV_EXCL_START
673 default:
674 std::abort();
675 // LCOV_EXCL_STOP
676 }
