49template <typename fp_t>
50int getFloatingPointExponent(fp_t value) {
51 if (value == 0.0) {
52 return 0; // Exponent for zero is typically represented as all zeros.
53 } else {
54 return std::max(std::numeric_limits<fp_t>::min_exponent,
55 std::ilogb(std::abs(value)) + 1);
56 }

210 void computeNormalisationVectors() {
211 // Compute normalisation vector.
212 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
213 this->powersVector[outer] = 0.0;
214 for (size_t j = 0; j < this->innerDimension(); j++) {
215 const size_t matrixIndex = operandIndex(outer, j);
216 this->powersVector[outer] = std::max(this->powersVector[outer],
217 std::abs(this->matrix[matrixIndex]));
218 }
219 // Compute the smallest power of 2 that is strictly greater than the
220 // maximum value in the row/column.
221 // NOTE 1: This is not the technique used in uoi24.
222 // NOTE 2: I use exponents instead of powers of 2, as I need the former
223 // to shift correctly.
224 this->scalingExponents[outer] = getFloatingPointExponent(this->powersVector[outer]);
225 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);
226 }
227 }

234 int computeSliceBitOffset(size_t slice) const {
235 switch (splitType) {
236 case splittingStrategy::truncation:
237 // Slice k -> k * b
238 return static_cast<int>((slice + 1) * bitsPerSlice);
239
240 case splittingStrategy::roundToNearest:
241 // Slice 0 -> b - 1
242 // Slice k -> k * b - 1
243 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
244
245 case splittingStrategy::unsignedEncoding:
246 // Slice 0 -> b - 1
247 // Slice k -> (b - 1) + k * (b + 1)
248 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));
249
250 // LCOV_EXCL_START
251 default:
252 throw std::logic_error("Unhandled splittingStrategy");
253 // LCOV_EXCL_STOP
254 }

264 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
265 auto numFracBits = computeNumFracBits<fp_t>();
266 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
267 const size_t matrixIndex = operandIndex(outer, inner);
268 fp_t value = this->matrix[matrixIndex];
269 fraction[inner] = std::bit_cast<uint_t>(value);
270 sign[inner] = std::signbit(value);
271 uint_t bitmask = (static_cast<uint_t>(1) << (numFracBits - 1)) - 1;
272 fraction[inner] = fraction[inner] & bitmask;
273 // Restore implicit bit for normal numbers.
274 // NOTE: NaNs and infs are currently not supported.
275 if (std::fpclassify(value) == FP_NORMAL)
276 fraction[inner] |= (static_cast<uint_t>(1) << (numFracBits - 1));
277 }

289 void computeSplitsWithTruncation() {
290 this->splitType = splittingStrategy::truncation;
291 // Compute splits one row/column at a time.
292 auto numFracBits = computeNumFracBits<fp_t>();
293 auto bitsPerSlice = this->bitsPerSlice;
294 std::vector<uint_t> fraction (this->innerDimension());
295 std::vector<bool> sign (this->innerDimension());
296 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
297 // Get binary representation of significands of normalised row/column.
298 computeFixedPointRepresentationVector(fraction, sign, outer);
299
300 // Create bitmask.
301 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
302
303 // Perform the split.
304 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
305 // NOTE: I could have a special path for 0.
306 int16_t shiftCounter = numFracBits - bitsPerSlice;
307 int currentExponent = getFloatingPointExponent(this->matrix[operandIndex(outer, inner)]);
308 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
309 for (size_t slice = 0; slice < numSplits; slice++) {
310 if (exponentDifference > (signed)bitsPerSlice) {
311 exponentDifference -= bitsPerSlice;
312 } else {
313 shiftCounter += exponentDifference;
314 exponentDifference = 0;
315 uint_t bitmask = shiftCounter > 0 ?
316 smallBitmask << shiftCounter :
317 smallBitmask >> -shiftCounter;
318 uint_t currentSlice = fraction[inner] & bitmask;
319 uint_t currentSplit = shiftCounter > 0 ?
320 currentSlice >> shiftCounter :
321 currentSlice << -shiftCounter;
322 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
323 this->memory[splitIndex(outer, inner, slice)] = value;
324 shiftCounter -= bitsPerSlice;
325 }
326 }
327 }
328 }

342 void computeSplitsWithUnsignedEncoding() {
343 this->splitType = splittingStrategy::unsignedEncoding;
344 // Compute splits one row/column at a time.
345 auto numFracBits = computeNumFracBits<fp_t>();
346 auto bitsPerSlice = this->bitsPerSlice;
347 std::vector<uint_t> fraction (this->innerDimension());
348 std::vector<bool> sign (this->innerDimension());
349 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
350 // Get binary representation of significands of normalised row/column.
351 computeFixedPointRepresentationVector(fraction, sign, outer);
352
353 // Create bitmasks.
354 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
355 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
356 // is to keep the algorithm consistent with the bitmasking approach
357 // above.
358 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
359 // is not mentioned in sabb26, but it seems necessary to do this to
360 // ensure that the first (signed) slice does not overflow if the full
361 // bitwidth of splitint_t is used and the second slice is negative in
362 // two's complement.
363 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
364 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
365
366 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
367
368 auto matrixIndex = operandIndex(outer, inner);
369
370 // NOTE: I could have a special path for 0.
371 int16_t shiftCounter;
372 int currentExponent = getFloatingPointExponent(this->matrix[matrixIndex]);
373 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
374
375 splitint_t value = 0;
376 uint_t remainder = 0;
377
378 // Slice 0.
379 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
380 // Value is too small to contribute explicit bits to slice 0.
381 value = sign[inner] ? -1 : 0;
382 remainder = sign[inner] ?
383 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
384 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
385 fraction[inner];
386 exponentDifference -= (bitsPerSlice - 1);
387 shiftCounter = numFracBits - (bitsPerSlice + 1);
388 } else {
389 // Truncation.
390 shiftCounter = numFracBits - (bitsPerSlice - 1) + exponentDifference;
391 exponentDifference = 0;
392 uint_t bitmask = (shiftCounter > 0)
393 ? (smallBitmask << shiftCounter)
394 : (smallBitmask >> -shiftCounter);
395 uint_t currentSlice = fraction[inner] & bitmask;
396 uint_t currentSplit = (shiftCounter > 0)
397 ? (currentSlice >> shiftCounter)
398 : (currentSlice << -shiftCounter);
399
400 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
401
402 // Rounding.
403 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
404 uint_t lowBits = fraction[inner] & lowMask;
405
406 if (lowBits != 0) {
407 value += (sign[inner] ? -1 : 0);
408 }
409 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
410 shiftCounter -= (bitsPerSlice + 1);
411 }
412 this->memory[matrixIndex] = value;
413
414 // Remaining slices.
415 const auto width = bitsPerSlice + 1;
416 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
417 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
418 const auto digitMax = cutoff - 1; // 2^b - 1
419 const auto digitMin = -cutoff; // -2^b
420 for (size_t slice = 1; slice < numSplits; slice++) {
421 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
422 exponentDifference -= (bitsPerSlice + 1);
423 if (sign[inner]) {
424 this->memory[splitIndex(outer, inner, slice - 1)] = static_cast<splitint_t>(0);
425 this->memory[splitIndex(outer, inner, slice)] = static_cast<splitint_t>(-1);
426 }
427 } else {
428 shiftCounter += exponentDifference;
429 exponentDifference = 0;
430
431 uint_t bitmask = shiftCounter > 0 ?
432 largeBitmask << shiftCounter :
433 largeBitmask >> -shiftCounter;
434
435 uint_t currentSlice = remainder & bitmask;
436 uint_t currentSplit = shiftCounter > 0 ?
437 currentSlice >> shiftCounter :
438 currentSlice << -shiftCounter;
439
440 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
441 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
442 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
443 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
444 currentSplit |= maskToAdd;
445 }
446
447 // Update previous slices if currentSplit is negative in splitint_t.
448 // Carry must propoagate upward until slice 0 if necessary.
449 if (currentSplit > static_cast<uint_t>(digitMax)) {
450 int nextSlice = static_cast<int>(slice) - 1;
451 while (true) {
452 const size_t prevIdx = splitIndex(outer, inner, nextSlice);
453 auto newValue = static_cast<wideint_t>(this->memory[prevIdx]) + 1;
454
455 if (newValue <= digitMax) {
456 this->memory[prevIdx] = static_cast<splitint_t>(newValue);
457 break;
458 } else {
459 this->memory[prevIdx] = static_cast<splitint_t>(digitMin);
460 nextSlice--;
461 }
462 }
463 }
464
465 // Store the slice as a signed (b+1)-bit value:
466 // [0, 2^b-1] -> unchanged
467 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
468 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
469 if (signedDigit >= cutoff) {
470 signedDigit -= base;
471 }
472
473 this->memory[splitIndex(outer, inner, slice)] = static_cast<splitint_t>(signedDigit);
474 shiftCounter -= (bitsPerSlice + 1);
475 }
476 }
477 }
478 }
479 }

490 void computeSplitsWithRoundToNearest() {
491 this->splitType = splittingStrategy::roundToNearest;
492 auto bitsPerSlice = this->bitsPerSlice;
493 auto localMatrix = this->matrix;
494 for (size_t slice = 0; slice < numSplits; slice++) {
495 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
496 fp_t sigma = ldexp(0.75, computeNumFracBits<fp_t>() - bitsPerSlice * slice + 1 - bitsPerSlice) * powersVector[outer];
497 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
498 auto matrixIndex = operandIndex(outer, inner);
499 auto value = (localMatrix[matrixIndex] + sigma);
500 value -= sigma;
501 localMatrix[matrixIndex] -= value;
502 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
503 this->memory[splitIndex(outer, inner, slice)] = value;
504 }
505 }
506 }
507 }

525 const matrixLayout layoutC,
526 size_t iBlock, size_t jBlock) {
527

529 for (size_t col = 0; col < B.n; col++) {

533 B.memory[B.splitIndex(ell, col, jBlock)];
534 }

559template <typename splitint_t, typename accumulator_t, typename fp_t>
560std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
561 const MatrixSplit<splitint_t, fp_t> &B,
562 const matrixLayout layoutC,
563 const size_t numDiagonals) {
564 std::vector<fp_t> C (A.m * B.n, 0.0);
565 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
566 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
567 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
568 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
569 std::vector<accumulator_t> accumulator (A.m * B.n, 0.0);
570 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
571 for (size_t row = 0; row < A.m; row++) {
572 for (size_t col = 0; col < B.n; col++) {
573 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
574 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[row + col * A.m]), -totalShift);
575 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
576 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.m) : (col + row * B.n);
577 C[index] += scaledSum * scalingFactor;
578 }
579 }
580 Aindex--;
581 Bindex++;
582 }
583 }
584
585 return C;
586}

608template <typename splitint_t, typename accumulator_t, typename fp_t>
609std::vector<fp_t> computeProductsWithIntegerAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
610 const MatrixSplit<splitint_t, fp_t> &B,
611 const matrixLayout layoutC,
612 const size_t numDiagonals) {
613
614 std::vector<fp_t> C (A.m * B.n, 0.0);
615 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
616 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
617 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
618
619 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
620
621 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
622 std::vector<accumulator_t> accumulator(A.m * B.n, 0);
623 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
624 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
625 Aindex--;
626 Bindex++;
627 }
628
629 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
630 for (size_t row = 0; row < A.m; row++) {
631 for (size_t col = 0; col < B.n; col++) {
632 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[row + col * A.m]), -totalShift);
633 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
634 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.m) : (col + row * B.n);
635 C[index] += scaledSum * scalingFactor;
636 }
637 }
638 }
639
640 return C;
641}

660template <typename fp_t, typename splitint_t, typename accumulator_t>
661std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const matrixLayout layoutA,
662 const std::vector<fp_t> &B, const matrixLayout layoutB,
663 const size_t m, const size_t k, const size_t n,
664 const size_t numSplitsA, const size_t numSplitsB,
665 const splittingStrategy splitType = splittingStrategy::roundToNearest,
666 const accumulationStrategy accType = accumulationStrategy::floatingPoint,
667 const multiplicationStrategy multType = multiplicationStrategy::reduced) {
668
669 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
670 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
671 if (bitsPerInteger > bitsInAccumulator / 2) {
672 throw std::invalid_argument("Split integer type is too wide for the chosen accumulator type");
673 }
674 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
675 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
676
677 auto splitA = MatrixSplit<splitint_t, fp_t>(A, m, k, layoutA, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
678 auto splitB = MatrixSplit<splitint_t, fp_t>(B, k, n, layoutB, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
679
680 size_t numDiagonals;
681 switch (multType) {
682 case multiplicationStrategy::reduced:
683 // Products below the main anti-diagonal are ignored.
684 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
685 break;
686 case multiplicationStrategy::full:
687 // All products are computed.
688 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
689 break;
690 // LCOV_EXCL_START
691 default:
692 throw std::logic_error("Unhandled multiplicationStrategy");
693 // LCOV_EXCL_STOP
694 }
695
696 auto layoutC = matrixLayout::columnMajor;
697 switch (accType) {
698 case accumulationStrategy::floatingPoint:
699 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
700 case accumulationStrategy::integer:
701 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
702 // LCOV_EXCL_START
703 default:
704 throw std::logic_error("Unhandled accumulationStrategy");
705 // LCOV_EXCL_STOP
706 }
707}
