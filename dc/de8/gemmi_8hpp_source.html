382 void computeNormalisationVectors() {
383 // Compute normalisation vector.
384 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
385 this->powersVector[outer] = 0.0;
386 for (size_t j = 0; j < this->innerDimension(); j++) {
387 const size_t matrixIndex = operandIndex(outer, j);
388 this->powersVector[outer] = std::max(this->powersVector[outer],
389 std::abs(this->matrix.data[matrixIndex]));
390 }
391 // Compute the smallest power of 2 that is strictly greater than the
392 // maximum value in the row/column.
393 // NOTE 1: This is not the technique used in uoi24.
394 // NOTE 2: I use exponents instead of powers of 2, as I need the former
395 // to shift correctly.
396 this->scalingExponents[outer] = getStoredFloatingPointExponent(this->powersVector[outer]);
397 this->powersVector[outer] = std::ldexp(1.0, this->scalingExponents[outer]);
398 }

406 int computeSliceBitOffset(size_t slice) const {
407 switch (splitType) {
408 case splittingStrategy::truncation:
409 // Slice k -> k * b
410 return static_cast<int>((slice + 1) * bitsPerSlice);
411
412 case splittingStrategy::roundToNearest:
413 // Slice 0 -> b - 1
414 // Slice k -> k * b - 1
415 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
416
417 case splittingStrategy::unsignedEncoding:
418 // Slice 0 -> b - 1
419 // Slice k -> (b - 1) + k * (b + 1)
420 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));
421
422 // LCOV_EXCL_START
423 default:
424 throw std::logic_error("Unhandled splittingStrategy");
425 // LCOV_EXCL_STOP
426 }

436 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t outer) {
437 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
438 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
439 const size_t matrixIndex = operandIndex(outer, inner);
440 fp_t value = this->matrix.data[matrixIndex];
441 fraction[inner] = std::bit_cast<uint_t>(value);
442 sign[inner] = std::signbit(value);
443 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
444 fraction[inner] = fraction[inner] & bitmask;
445 // Restore implicit bit for normal numbers.
446 // NOTE: NaNs and infs are currently not supported.
447 if (std::fpclassify(value) == FP_NORMAL)
448 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
449 }

461 void computeSplitsWithTruncation() {
462 this->splitType = splittingStrategy::truncation;
463 // Compute splits one row/column at a time.
464 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
465 auto bitsPerSlice = this->bitsPerSlice;
466 std::vector<uint_t> fraction (this->innerDimension());
467 std::vector<bool> sign (this->innerDimension());
468 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
469 // Get binary representation of significands of normalised row/column.
470 computeFixedPointRepresentationVector(fraction, sign, outer);
471
472 // Create bitmask.
473 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
474
475 // Perform the split.
476 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
477 // NOTE: I could have a special path for 0.
478 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
479 int currentExponent = getStoredFloatingPointExponent(this->matrix.data[operandIndex(outer, inner)]);
480 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
481 for (size_t slice = 0; slice < numSplits; slice++) {
482 if (exponentDifference > (signed)bitsPerSlice) {
483 exponentDifference -= bitsPerSlice;
484 } else {
485 shiftCounter += exponentDifference;
486 exponentDifference = 0;
487 uint_t bitmask = shiftCounter > 0 ?
488 smallBitmask << shiftCounter :
489 smallBitmask >> -shiftCounter;
490 uint_t currentSlice = fraction[inner] & bitmask;
491 uint_t currentSplit = shiftCounter > 0 ?
492 currentSlice >> shiftCounter :
493 currentSlice << -shiftCounter;
494 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
495 this->memory[splitIndex(outer, inner, slice)] = value;
496 shiftCounter -= bitsPerSlice;
497 }
498 }
499 }
500 }
501 }

514 void computeSplitsWithUnsignedEncoding() {
515 this->splitType = splittingStrategy::unsignedEncoding;
516 // Compute splits one row/column at a time.
517 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
518 auto bitsPerSlice = this->bitsPerSlice;
519 std::vector<uint_t> fraction (this->innerDimension());
520 std::vector<bool> sign (this->innerDimension());
521 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
522 // Get binary representation of significands of normalised row/column.
523 computeFixedPointRepresentationVector(fraction, sign, outer);
524
525 // Create bitmasks.
526 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
527 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
528 // is to keep the algorithm consistent with the bitmasking approach
529 // above.
530 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
531 // is not mentioned in sabb26, but it seems necessary to do this to
532 // ensure that the first (signed) slice does not overflow if the full
533 // bitwidth of splitint_t is used and the second slice is negative in
534 // two's complement.
535 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
536 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
537
538 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
539
540 auto matrixIndex = operandIndex(outer, inner);
541
542 // NOTE: I could have a special path for 0.
543 int16_t shiftCounter;
544 int currentExponent = getStoredFloatingPointExponent(this->matrix.data[matrixIndex]);
545 int16_t exponentDifference = scalingExponents[outer] - currentExponent;
546
547 splitint_t value = 0;
548 uint_t remainder = 0;
549
550 // Slice 0.
551 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
552 // Value is too small to contribute explicit bits to slice 0.
553 value = sign[inner] ? -1 : 0;
554 remainder = sign[inner] ?
555 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
556 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
557 fraction[inner];
558 exponentDifference -= (bitsPerSlice - 1);
559 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
560 } else {
561 // Truncation.
562 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
563 exponentDifference = 0;
564 uint_t bitmask = (shiftCounter > 0)
565 ? (smallBitmask << shiftCounter)
566 : (smallBitmask >> -shiftCounter);
567 uint_t currentSlice = fraction[inner] & bitmask;
568 uint_t currentSplit = (shiftCounter > 0)
569 ? (currentSlice >> shiftCounter)
570 : (currentSlice << -shiftCounter);
571
572 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
573
574 // Rounding.
575 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
576 uint_t lowBits = fraction[inner] & lowMask;
577
578 if (lowBits != 0) {
579 value += (sign[inner] ? -1 : 0);
580 }
581 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
582 shiftCounter -= (bitsPerSlice + 1);
583 }
584 this->memory[matrixIndex] = value;
585
586 // Remaining slices.
587 const auto width = bitsPerSlice + 1;
588 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
589 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
590 const auto digitMax = cutoff - 1; // 2^b - 1
591 const auto digitMin = -cutoff; // -2^b
592 for (size_t slice = 1; slice < numSplits; slice++) {
593 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
594 exponentDifference -= (bitsPerSlice + 1);
595 if (sign[inner]) {
596 this->memory[splitIndex(outer, inner, slice - 1)] = static_cast<splitint_t>(0);
597 this->memory[splitIndex(outer, inner, slice)] = static_cast<splitint_t>(-1);
598 }
599 } else {
600 shiftCounter += exponentDifference;
601 exponentDifference = 0;
602
603 uint_t bitmask = shiftCounter > 0 ?
604 largeBitmask << shiftCounter :
605 largeBitmask >> -shiftCounter;
606
607 uint_t currentSlice = remainder & bitmask;
608 uint_t currentSplit = shiftCounter > 0 ?
609 currentSlice >> shiftCounter :
610 currentSlice << -shiftCounter;
611
612 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
613 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
614 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
615 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
616 currentSplit |= maskToAdd;
617 }
618
619 // Update previous slices if currentSplit is negative in splitint_t.
620 // Carry must propoagate upward until slice 0 if necessary.
621 if (currentSplit > static_cast<uint_t>(digitMax)) {
622 int nextSlice = static_cast<int>(slice) - 1;
623 while (true) {
624 const size_t prevIdx = splitIndex(outer, inner, nextSlice);
625 auto newValue = static_cast<wideint_t>(this->memory[prevIdx]) + 1;
626
627 if (newValue <= digitMax) {
628 this->memory[prevIdx] = static_cast<splitint_t>(newValue);
629 break;
630 } else {
631 this->memory[prevIdx] = static_cast<splitint_t>(digitMin);
632 nextSlice--;
633 }
634 }
635 }
636
637 // Store the slice as a signed (b+1)-bit value:
638 // [0, 2^b-1] -> unchanged
639 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
640 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
641 if (signedDigit >= cutoff) {
642 signedDigit -= base;
643 }
644
645 this->memory[splitIndex(outer, inner, slice)] = static_cast<splitint_t>(signedDigit);
646 shiftCounter -= (bitsPerSlice + 1);
647 }
648 }
649 }
650 }
651 }

662 void computeSplitsWithRoundToNearest() {
663 this->splitType = splittingStrategy::roundToNearest;
664 auto bitsPerSlice = this->bitsPerSlice;
665 auto localMatrix = std::vector<fp_t>(this->matrix.data, this->matrix.data + this->matrix.size());
666 for (size_t slice = 0; slice < numSplits; slice++) {
667 for (size_t outer = 0; outer < this->outerDimension(); outer++) {
668 fp_t sigma = ldexp(0.75,
669 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
670 powersVector[outer];
671 for (size_t inner = 0; inner < this->innerDimension(); inner++) {
672 auto matrixIndex = operandIndex(outer, inner);
673 auto value = (localMatrix[matrixIndex] + sigma);
674 value -= sigma;
675 localMatrix[matrixIndex] -= value;
676 value = value / powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
677 this->memory[splitIndex(outer, inner, slice)] = value;
678 }
679 }
680 }
681 }

747template <typename splitint_t, typename accumulator_t, typename fp_t>
748std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
749 const MatrixSplit<splitint_t, fp_t> &B,
750 const matrixLayout layoutC,
751 const size_t numDiagonals) {
752 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
753 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
754 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
755 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
756 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
757 std::vector<accumulator_t> accumulator (A.rows() * B.cols(), 0.0);
758 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
759 for (size_t row = 0; row < A.rows(); row++) {
760 for (size_t col = 0; col < B.cols(); col++) {
761 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
762 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
763 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
764 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
765 C[index] += scaledSum * scalingFactor;
766 }
767 }
768 Aindex--;
769 Bindex++;
770 }
771 }
772
773 return C;
774}

796template <typename splitint_t, typename accumulator_t, typename fp_t>
797std::vector<fp_t> computeProductsWithIntegerAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
798 const MatrixSplit<splitint_t, fp_t> &B,
799 const matrixLayout layoutC,
800 const size_t numDiagonals) {
801
802 std::vector<fp_t> C (A.rows() * B.cols(), 0.0);
803 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
804 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
805 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
806
807 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
808
809 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
810 std::vector<accumulator_t> accumulator(A.rows() * B.cols(), 0);
811 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
812 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, layoutC, Aindex, Bindex);
813 Aindex--;
814 Bindex++;
815 }
816
817 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
818 for (size_t row = 0; row < A.rows(); row++) {
819 for (size_t col = 0; col < B.cols(); col++) {
820 auto index = (layoutC == matrixLayout::columnMajor) ? (row + col * A.rows()) : (col + row * B.cols());
821 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[index]), -totalShift);
822 fp_t scalingFactor = A.powersVector[row] * B.powersVector[col];
823 C[index] += scaledSum * scalingFactor;
824 }
825 }
826 }
827
828 return C;
829}

848template <typename fp_t, typename splitint_t, typename accumulator_t>
849std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const matrixLayout layoutA,
850 const std::vector<fp_t> &B, const matrixLayout layoutB,
851 const size_t m, const size_t k, const size_t n,
852 const size_t numSplitsA, const size_t numSplitsB,
853 const matrixLayout layoutC = matrixLayout::columnMajor,
854 const splittingStrategy splitType = splittingStrategy::roundToNearest,
855 const accumulationStrategy accType = accumulationStrategy::floatingPoint,
856 const multiplicationStrategy multType = multiplicationStrategy::reduced) {
857
858 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
859 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
860 if (bitsPerInteger > bitsInAccumulator / 2) {
861 throw std::invalid_argument("Split integer type is too wide for the chosen accumulator type");
862 }
863 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
864 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
865
866 auto viewA = makeMatrixView(A, m, k, layoutA);
867 auto viewB = makeMatrixView(B, k, n, layoutB);
868
869 auto splitA = MatrixSplit<splitint_t, fp_t>(viewA, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
870 auto splitB = MatrixSplit<splitint_t, fp_t>(viewB, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
871
872 prepareMatrixSplit(splitA);
873 prepareMatrixSplit(splitB);
874
875 size_t numDiagonals;
876 switch (multType) {
877 case multiplicationStrategy::reduced:
878 // Products below the main anti-diagonal are ignored.
879 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
880 break;
881 case multiplicationStrategy::full:
882 // All products are computed.
883 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
884 break;
885 // LCOV_EXCL_START
886 default:
887 throw std::logic_error("Unhandled multiplicationStrategy");
888 // LCOV_EXCL_STOP
889 }
890
891 switch (accType) {
892 case accumulationStrategy::floatingPoint:
893 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
894 case accumulationStrategy::integer:
895 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, layoutC, numDiagonals);
896 // LCOV_EXCL_START
897 default:
898 throw std::logic_error("Unhandled accumulationStrategy");
899 // LCOV_EXCL_STOP
900 }
901}

903template <typename fp_t, typename splitint_t, typename accumulator_t>
