159 void computeNormalisationVectors() {
160 // Compute normalisation vector.
161 const size_t iStride = this->iStride();
162 const size_t jStride = this->jStride();
163 for (size_t i = 0; i < this->otherDimension(); i++) {
164 this->powersVector[i] = 0.0;
165 for (size_t j = 0; j < this->innerProductDimension(); j++) {
166 const size_t index = i * iStride + j * jStride;
167 this->powersVector[i] = std::max(this->powersVector[i],
168 std::abs(this->matrix[index]));
169 }
170 // Compute the smallest power of 2 that is strictly greater than the
171 // maximum value in the row/column.
172 // NOTE 1: This is not the technique used in uoi24.
173 // NOTE 2: I use exponents instead of powers of 2, as I need the former
174 // to shift correctly.
175 frexp(this->powersVector[i], this->scalingExponents.data() + i);
176 this->powersVector[i] = std::ldexp(1.0, this->scalingExponents[i]);
177 }

185 int computeSliceBitOffset(size_t slice) const {
186 switch (splitType) {
187 case splittingStrategy::bitMasking:
188 // Slice k -> k * b
189 return static_cast<int>((slice + 1) * bitsPerSlice);
190
191 case splittingStrategy::roundToNearest:
192 // Slice 0 -> b - 1
193 // Slice k -> k * b - 1
194 return static_cast<int>((slice + 1) * bitsPerSlice - 1);
195
196 case splittingStrategy::unsignedEncoding:
197 // Slice 0 -> b - 1
198 // Slice k -> (b - 1) + k * (b + 1)
199 return static_cast<int>((bitsPerSlice - 1) + slice * (bitsPerSlice + 1));
200 }

202 std::cerr << "Unknown splitting strategy requested.";
203 std::exit(1);

213 void computeFixedPointRepresentationVector(std::vector<uint_t> &fraction, std::vector<bool> &sign, size_t i) {
214 auto numExpBits = computeNumExpBits<fp_t>();
215 auto numFracBits = computeNumFracBits<fp_t>();
216 auto iStride = this->iStride();
217 auto jStride = this->jStride();
218 for (size_t j = 0; j < this->innerProductDimension(); j++) {
219 const size_t index = i * iStride + j * jStride;
220 fp_t value = this->matrix[index]; // powersVector[i];
221 fraction[j] = std::bit_cast<uint_t>(value); // To bitstring.
222 sign[j] = std::signbit(value); // Extract sign.
223 uint_t bitmask = (~((uint_t)(0))) >> (numExpBits + 1);
224 fraction[j] = fraction[j] & bitmask; // Remove exponent and sign.
225 // Restore implicit bit for normal numbers.
226 // NOTE: NaNs and infs are currently not supported.
227 if (std::fpclassify(value) == FP_NORMAL)
228 fraction[j] |= ((uint_t)1 << (numFracBits - 1));
229 }

241 void computeSplitsWithBitMasking() {
242 this->splitType = splittingStrategy::bitMasking;
243 // Compute splits one row/column at a time.
244 auto numFracBits = computeNumFracBits<fp_t>();
245 auto bitsPerSlice = this->bitsPerSlice;
246 auto iStride = this->iStride();
247 auto jStride = this->jStride();
248 std::vector<uint_t> fraction (this->innerProductDimension());
249 std::vector<bool> sign (this->innerProductDimension());
250 for (size_t i = 0; i < this->otherDimension(); i++) {
251 // Get binary representation of significands of normalised row/column.
252 computeFixedPointRepresentationVector(fraction, sign, i);
253
254 // Create bitmask.
255 const uint_t smallBitmask = (1 << bitsPerSlice) - 1;
256 // Perform the split.
257 for (size_t j = 0; j < this->innerProductDimension(); j++) {
258 int16_t shiftCounter = numFracBits - bitsPerSlice;
259 int currentExponent;
260 frexp(this->matrix[i * iStride + j * jStride], &currentExponent);
261 int16_t exponentDifference = scalingExponents[i] - currentExponent;
262 for (size_t slice = 0; slice < numSplits; slice++) {
263 if (exponentDifference > (signed)bitsPerSlice) {
264 exponentDifference -= bitsPerSlice;
265 } else {
266 shiftCounter += exponentDifference;
267 exponentDifference = 0;
268 uint_t bitmask = shiftCounter > 0 ?
269 smallBitmask << shiftCounter :
270 smallBitmask >> -shiftCounter;
271 uint_t currentSlice = fraction[j] & bitmask;
272 uint_t currentSplit = shiftCounter > 0 ?
273 currentSlice >> shiftCounter :
274 currentSlice << -shiftCounter;
275 splitint_t value = (splitint_t)(currentSplit) * (sign[j] ? -1 : 1);
276 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = value;
277 shiftCounter -= bitsPerSlice;
278 }
279 }
280 }
281 }

295 void computeSplitsWithUnsignedEncoding() {
296 this->splitType = splittingStrategy::unsignedEncoding;
297 // Compute splits one row/column at a time.
298 auto numFracBits = computeNumFracBits<fp_t>();
299 auto bitsPerSlice = this->bitsPerSlice;
300 auto iStride = this->iStride();
301 auto jStride = this->jStride();
302 std::vector<uint_t> fraction (this->innerProductDimension());
303 std::vector<bool> sign (this->innerProductDimension());
304 for (size_t i = 0; i < this->otherDimension(); i++) {
305 // Get binary representation of significands of normalised row/column.
306 computeFixedPointRepresentationVector(fraction, sign, i);
307
308 // Create bitmasks.
309 // This algorithm uses bitsPerSlice bits for the first slice and bitsPerSlice + 1 bits for subsequent slices.
310 // This choice is to keep the algorithm consistent with the bitmasking approach above.
311 // NOTE: I am using a first slice with bitsPerSlice - 1 bits. This is not mentioned in sabb26, but it seems
312 // necessary to do this, if I want to ensure that the first slice does not overflow if the second slice
313 // is negative in two's complement.
314 const uint_t smallBitmask = (1 << (bitsPerSlice - 1)) - 1;
315 const uint_t largeBitmask = (1 << (bitsPerSlice + 1)) - 1;
316
317 for (size_t j = 0; j < this->innerProductDimension(); j++) {
318
319 int currentExponent;
320 frexp(this->matrix[i * iStride + j * jStride], &currentExponent);
321 int16_t exponentDifference = scalingExponents[i] - currentExponent;
322
323 // Boundary below slice 0, fully aligned to this entry.
324 int16_t firstShift = numFracBits - (bitsPerSlice - 1) + exponentDifference;
325
326 splitint_t value = 0;
327 uint_t remainder = 0;
328
329 if (!sign[j]) {
330 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
331 // Too small to contribute explicit bits to slice 0.
332 value = 0;
333 remainder = fraction[j];
334 } else {
335 // Positive number: leading digit is truncation.
336 uint_t bitmask = (firstShift > 0)
337 ? (smallBitmask << firstShift)
338 : (smallBitmask >> -firstShift);
339
340 uint_t currentSlice = fraction[j] & bitmask;
341 uint_t currentSplit = (firstShift > 0)
342 ? (currentSlice >> firstShift)
343 : (currentSlice << -firstShift);
344
345 value = static_cast<splitint_t>(currentSplit);
346 remainder = fraction[j] & ((uint_t(1) << firstShift) - 1);
347 }
348 } else {
349 // Negative number.
350 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
351 // Too small to contribute explicit bits to slice 0:
352 // round toward -infinity gives -1.
353 value = -1;
354 remainder = ((1ull << (sizeof(uint_t) * 8 - 1)) - fraction[j]) | (uint_t(1) << (sizeof(uint_t) * 8 - 1));
355 } else {
356 // Normal truncation path for the leading slice.
357 uint_t bitmask = (firstShift > 0)
358 ? (smallBitmask << firstShift)
359 : (smallBitmask >> -firstShift);
360
361 uint_t currentSlice = fraction[j] & bitmask;
362 uint_t currentSplit = (firstShift > 0)
363 ? (currentSlice >> firstShift)
364 : (currentSlice << -firstShift);
365
366 value = -static_cast<splitint_t>(currentSplit);
367
368 uint_t lowMask = (uint_t(1) << firstShift) - 1;
369 uint_t lowBits = fraction[j] & lowMask;
370
371 if (lowBits != 0) {
372 value -= 1; // round toward -infinity
373 remainder = (1ull << firstShift) - lowBits;
374 } else {
375 remainder = 0;
376 }
377 }
378 }
379
380 this->memory[i * iStride + j * jStride] = value;
381
382 // From here on, the remainder is already aligned and nonnegative.
383 int shiftCounter = firstShift - (bitsPerSlice + 1);
384 exponentDifference = 0;
385
386 // Split (nonnegative) remainder with unsigned slice encoding.
387 for (size_t slice = 1; slice < numSplits; slice++) {
388 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
389 exponentDifference -= (bitsPerSlice + 1);
390 if (sign[j]) {
391 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = static_cast<splitint_t>(largeBitmask);
392 }
393 } else {
394 shiftCounter += exponentDifference;
395 exponentDifference = 0;
396
397 uint_t bitmask = shiftCounter > 0 ?
398 largeBitmask << shiftCounter :
399 largeBitmask >> -shiftCounter;
400
401 uint_t currentSlice = remainder & bitmask;
402 uint_t currentSplit = shiftCounter > 0 ?
403 currentSlice >> shiftCounter :
404 currentSlice << -shiftCounter;
405
406 if (sign[j] && shiftCounter + bitsPerSlice + 1 > sizeof(uint_t) * 8) {
407 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
408 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
409 uint_t maskToAdd = ((1ull << bitsLost) - 1) << bitsUsed;
410 currentSplit |= maskToAdd;
411 }
412
413 // Width of sub-leading slices is (bitsPerSlice + 1) bits.
414 const auto width = bitsPerSlice + 1;
415 const auto cutoff = static_cast<wideint_t>(uint_t(1) << bitsPerSlice); // 2^b
416 const auto base = static_cast<wideint_t>(uint_t(1) << width); // 2^(b+1)
417 const auto digitMax = cutoff - 1; // 2^b - 1
418 const auto digitMin = -cutoff; // -2^b
419
420 // If the unsigned digit is in the upper half, propagate carry upward.
421 if (currentSplit >= static_cast<uint_t>(cutoff)) {
422 int nextSlice = static_cast<int>(slice) - 1;
423 while (true) {
424 const size_t prevIdx =
425 i * iStride + j * jStride + nextSlice * this->matrix.size();
426
427 auto newValue =
428 static_cast<wideint_t>(this->memory[prevIdx]) + 1;
429
430 if (newValue <= digitMax) {
431 this->memory[prevIdx] = static_cast<splitint_t>(newValue);
432 break;
433 } else {
434 this->memory[prevIdx] = static_cast<splitint_t>(digitMin);
435 nextSlice--;
436 }
437 }
438 }
439
440 // Store the slice as a signed (b+1)-bit value:
441 // [0, 2^b-1] -> unchanged
442 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
443 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
444 if (signedDigit >= cutoff) {
445 signedDigit -= base;
446 }
447
448 this->memory[
449 i * iStride + j * jStride + slice * this->matrix.size()
450 ] = static_cast<splitint_t>(signedDigit);
451
452 shiftCounter -= (bitsPerSlice + 1);
453 }
454 }
455 }
456 }
457 }

468 void computeSplitsWithRoundToNearest() {
469 this->splitType = splittingStrategy::roundToNearest;
470 auto bitsPerSlice = this->bitsPerSlice;
471 auto iStride = this->iStride();
472 auto jStride = this->jStride();
473 auto localMatrix = this->matrix;
474 for (size_t slice = 0; slice < numSplits; slice++) {
475 for (size_t i = 0; i < this->otherDimension(); i++) {
476 fp_t sigma = ldexp(0.75, computeNumFracBits<fp_t>() - bitsPerSlice * slice + 1 - bitsPerSlice) * powersVector[i];
477 for (size_t j = 0; j < this->innerProductDimension(); j++) {
478 auto value = (localMatrix[i * iStride + j * jStride] + sigma);
479 value -= sigma;
480 localMatrix[i * iStride + j * jStride] -= value;
481 value = value / powersVector[i] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
482 this->memory[i * iStride + j * jStride + slice * this->matrix.size()] = value;
483 }
484 }
485 }

534template <typename splitint_t, typename accumulator_t, typename fp_t>
535std::vector<fp_t> computeProductsWithFloatingPointAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
536 const MatrixSplit<splitint_t, fp_t> &B,
537 const size_t numDiagonals) {
538
539 std::vector<fp_t > C (A.m * B.n);
540
541 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
542 int Aindex = diagonal < A.numSplits - 1 ? diagonal : A.numSplits - 1;
543 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
544 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
545 std::vector<accumulator_t> accumulator (A.m * B.n, 0.0);
546 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, Aindex, Bindex);
547 for (size_t i = 0; i < A.m; i++) {
548 for (size_t j = 0; j < B.n; j++) {
549 int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
550 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[i + j * A.m]), -totalShift);
551 fp_t scalingFactor = A.powersVector[i] * B.powersVector[j];
552 C[i + j * A.m] += scaledSum * scalingFactor;
553 }
554 }
555 Aindex--;
556 Bindex++;
557 }
558 }
559
560 return C;
561}

583template <typename splitint_t, typename accumulator_t, typename fp_t>
584std::vector<fp_t> computeProductsWithIntegerAccumulation(const MatrixSplit<splitint_t, fp_t> &A,
585 const MatrixSplit<splitint_t, fp_t> &B,
586 const size_t numDiagonals) {
587 std::vector<fp_t> C(A.m * B.n);
588
589 for (size_t diagonal = 0; diagonal <= numDiagonals; diagonal++) {
590 int Aindex = diagonal < A.numSplits ? static_cast<int>(diagonal) : static_cast<int>(A.numSplits - 1);
591 size_t Bindex = diagonal > A.numSplits - 1 ? diagonal - A.numSplits + 1 : 0;
592
593 const int totalShift = A.computeSliceBitOffset(static_cast<size_t>(Aindex)) + B.computeSliceBitOffset(Bindex);
594
595 // Compute and accumulate all products along this anti-diagonal in integer arithmetic.
596 std::vector<accumulator_t> accumulator(A.m * B.n, 0);
597 while (Aindex >= 0 && Bindex <= std::min(diagonal, B.numSplits - 1)) {
598 // assert(A.sliceBitOffset(static_cast<size_t>(Aindex)) + B.sliceBitOffset(Bindex) == totalShift);
599
600 computeExactIntegerGEMM<splitint_t, accumulator_t, fp_t>(A, B, accumulator, Aindex, Bindex);
601 Aindex--;
602 Bindex++;
603 }
604
605 // Scale the accumulated products and accumulate in floating-point arithmetic across diagonals.
606 for (size_t i = 0; i < A.m; i++) {
607 for (size_t j = 0; j < B.n; j++) {
608 fp_t scaledSum = std::ldexp(static_cast<fp_t>(accumulator[i + j * A.m]), -totalShift);
609 fp_t scalingFactor = A.powersVector[i] * B.powersVector[j];
610 C[i + j * A.m] += scaledSum * scalingFactor;
611 }
612 }
613 }
614
615 return C;
616}

635template <typename fp_t, typename splitint_t, typename accumulator_t>
636std::vector<fp_t> gemmi (const std::vector<fp_t> &A, const std::vector<fp_t> &B,
637 const size_t m, const size_t k, const size_t n,
638 const size_t numSplitsA, const size_t numSplitsB,
639 const splittingStrategy splitType = splittingStrategy::roundToNearest,
640 const accumulationStrategy accType = accumulationStrategy::floatingPoint,
641 const multiplicationStrategy multType = multiplicationStrategy::reduced) {
642
643 const size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
644 const size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
645 assert(bitsPerInteger <= bitsInAccumulator / 2);
646 const size_t alpha = std::floor((bitsInAccumulator - log2(k)) / 2);
647 const size_t bitsPerSlice = std::min(bitsPerInteger, static_cast<size_t>(alpha));
648
649 auto splitA = MatrixSplit<splitint_t, fp_t>(A, m, k, splitType, numSplitsA, bitsPerSlice, normalisationDimension::byRows);
650 auto splitB = MatrixSplit<splitint_t, fp_t>(B, k, n, splitType, numSplitsB, bitsPerSlice, normalisationDimension::byCols);
651
652 size_t numDiagonals;
653 switch (multType) {
654 case multiplicationStrategy::reduced:
655 // Products below the main anti-diagonal are ignored.
656 numDiagonals = std::max(splitA.numSplits, splitB.numSplits) - 1;
657 break;
658 case multiplicationStrategy::full:
659 // All products are computed.
660 numDiagonals = splitA.numSplits + splitB.numSplits - 1;
661 break;
662 default:
663 std::cerr << "Unknown multiplication strategy requested.";
664 exit(1);
665 }
666
667 switch (accType) {
668 case accumulationStrategy::floatingPoint:
669 return computeProductsWithFloatingPointAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, numDiagonals);
670 case accumulationStrategy::integer:
671 return computeProductsWithIntegerAccumulation<splitint_t, accumulator_t, fp_t>(splitA, splitB, numDiagonals);
672 default:
673 std::cerr << "Unknown accumulation strategy requested.";
674 exit(1);
675 }
676}
