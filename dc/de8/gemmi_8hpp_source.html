409template <typename fp_t, typename splitint_t, typename accumulator_t>
410DerivedParameters deriveParameters(MatrixView<const fp_t> A,
411 MatrixView<const fp_t> B,
412 const config& config) {
413
414 // Compile-time type checks.
415 static_assert(std::is_floating_point_v<fp_t>,
416 "fp_t must be a floating-point type");
417 static_assert(std::is_integral_v<splitint_t> && std::is_signed_v<splitint_t>,
418 "splitint_t must be a signed integer type");
419 static_assert(std::is_integral_v<accumulator_t> && std::is_signed_v<accumulator_t>,
420 "accumulator_t must be a signed integer type");
421
422 // Matrix checks.
423 if (A.data == nullptr)
424 throw std::invalid_argument("Matrix A has a null data pointer");
425 if (B.data == nullptr)
426 throw std::invalid_argument("Matrix B has a null data pointer");
427 if (A.empty())
428 throw std::invalid_argument("Matrix A is empty (rows or cols is 0)");
429 if (B.empty())
430 throw std::invalid_argument("Matrix B is empty (rows or cols is 0)");
431
432 if (A.cols != B.rows)
433 throw std::invalid_argument(
434 "Dimension mismatch: A.cols (" + std::to_string(A.cols) +
435 ") != B.rows (" + std::to_string(B.rows) + ")");
436
437 // Split counts.
438 if (config.numSplitsA == 0)
439 throw std::invalid_argument("numSplitsA must be >= 1");
440 if (config.numSplitsB == 0)
441 throw std::invalid_argument("numSplitsB must be >= 1");
442
443 // Custom mask size.
444 if (std::holds_alternative<std::vector<bool>>(config.multSpecification)) {
445 const auto& mask = std::get<std::vector<bool>>(config.multSpecification);
446 const size_t expected = config.numSplitsA * config.numSplitsB;
447 if (mask.size() != expected)
448 throw std::invalid_argument(
449 "Custom mask size (" + std::to_string(mask.size()) +
450 ") does not match numSplitsA * numSplitsB (" +
451 std::to_string(expected) + ")");
452 }
453
454 // Type width constraints.
455 constexpr size_t bitsInAccumulator = std::numeric_limits<accumulator_t>::digits;
456 constexpr size_t bitsPerInteger = std::numeric_limits<splitint_t>::digits;
457 if (bitsPerInteger > bitsInAccumulator / 2)
458 throw std::invalid_argument(
459 "splitint_t (" + std::to_string(bitsPerInteger) + " bits) is too wide "
460 "for accumulator_t (" + std::to_string(bitsInAccumulator) + " bits): "
461 "require bitsPerInteger <= bitsInAccumulator / 2");
462
463 const size_t k = A.cols;
464 const double log2k = (k > 1) ? std::log2(static_cast<double>(k)) : 0.0;
465 const auto alpha = static_cast<size_t>(
466 std::max(0.0, std::floor((static_cast<double>(bitsInAccumulator) - log2k) / 2.0)));
467 const size_t bitsPerSlice = std::min(bitsPerInteger, alpha);
468 if (bitsPerSlice == 0)
469 throw std::invalid_argument(
470 "Computed bitsPerSlice is 0: inner dimension k=" + std::to_string(k) +
471 " is too large for accumulator_t (" +
472 std::to_string(bitsInAccumulator) + " bits)");
473
474 return DerivedParameters{bitsPerSlice};
475}

627template <typename splitint_t, typename fp_t>
628void computeNormalisationVectors(preparedOperand<splitint_t, fp_t>& operand) {
629 // Compute normalisation vector.
630 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
631 operand.powersVector[outer] = 0.0;
632 for (size_t j = 0; j < operand.innerDimension(); j++) {
633 operand.powersVector[outer] = std::max(operand.powersVector[outer],
634 std::abs(operand.operand(outer, j)));
635 }
636 // Compute the smallest power of 2 that is strictly greater than the
637 // maximum value in the row/column.
638 // NOTE 1: This is not the technique used in uoi24.
639 // NOTE 2: I use exponents instead of powers of 2, as I need the former
640 // to shift correctly.
641 operand.scalingExponents[outer] = getStoredFloatingPointExponent(operand.powersVector[outer]);
642 operand.powersVector[outer] = std::ldexp(1.0, operand.scalingExponents[outer]);
643 }
644}

657template <typename splitint_t, typename fp_t>
658void computeFixedPointRepresentationVector(std::vector<typename FloatingPointTraits<fp_t>::StorageType> &fraction,
659 std::vector<bool> &sign, size_t outer,
660 const preparedOperand<splitint_t, fp_t>& operand) {
661 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
662 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
663 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
664 fp_t value = operand.operand(outer, inner);
665 fraction[inner] = std::bit_cast<uint_t>(value);
666 sign[inner] = std::signbit(value);
667 uint_t bitmask = (static_cast<uint_t>(1) << (numSignificandBits - 1)) - 1;
668 fraction[inner] = fraction[inner] & bitmask;
669 // Restore implicit bit for normal numbers.
670 // NOTE: NaNs and infs are currently not supported.
671 if (std::fpclassify(value) == FP_NORMAL)
672 fraction[inner] |= (static_cast<uint_t>(1) << (numSignificandBits - 1));
673 }
674}

689template <typename splitint_t, typename fp_t>
690void computeSplitsWithTruncation(preparedOperand<splitint_t, fp_t>& operand) {
691 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
692
693 // Compute splits one row/column at a time.
694 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
695 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
696 std::vector<uint_t> fraction (operand.innerDimension());
697 std::vector<bool> sign (operand.innerDimension());
698 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
699 // Get binary representation of significands of normalised row/column.
700 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
701
702 // Create bitmask.
703 const uint_t smallBitmask = (static_cast<uint_t>(1) << bitsPerSlice) - 1;
704
705 // Perform the split.
706 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
707 // NOTE: I could have a special path for 0.
708 int16_t shiftCounter = numSignificandBits - bitsPerSlice;
709 int currentExponent = getStoredFloatingPointExponent(operand.operand(outer, inner));
710 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
711 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
712 if (exponentDifference > (signed)bitsPerSlice) {
713 exponentDifference -= bitsPerSlice;
714 } else {
715 shiftCounter += exponentDifference;
716 exponentDifference = 0;
717 uint_t bitmask = shiftCounter > 0 ?
718 smallBitmask << shiftCounter :
719 smallBitmask >> -shiftCounter;
720 uint_t currentSlice = fraction[inner] & bitmask;
721 uint_t currentSplit = shiftCounter > 0 ?
722 currentSlice >> shiftCounter :
723 currentSlice << -shiftCounter;
724 splitint_t value = (splitint_t)(currentSplit) * (sign[inner] ? -1 : 1);
725 operand.splitValue(outer, inner, slice) = value;
726 shiftCounter -= bitsPerSlice;
727 }
728 }
729 }
730 }
731}

748template <typename splitint_t, typename fp_t>
749void computeSplitsWithUnsignedEncoding(preparedOperand<splitint_t, fp_t>& operand) {
750 using uint_t = typename FloatingPointTraits<fp_t>::StorageType;
751 using wideint_t = std::conditional_t<(sizeof(splitint_t) < sizeof(int)), int, std::intmax_t>;
752
753 // Compute splits one row/column at a time.
754 constexpr size_t numSignificandBits = FloatingPointTraits<fp_t>::numSignificandBits;
755 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
756 std::vector<uint_t> fraction (operand.innerDimension());
757 std::vector<bool> sign (operand.innerDimension());
758 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
759 // Get binary representation of significands of normalised row/column.
760 computeFixedPointRepresentationVector(fraction, sign, outer, operand);
761
762 // Create bitmasks.
763 // This algorithm uses (bitsPerSlice - 1) bits for the first slice
764 // and (bitsPerSlice + 1) bits for subsequent slices. This choice
765 // is to keep the algorithm consistent with the bitmasking approach
766 // above.
767 // NOTE: I am using a first slice with (bitsPerSlice - 1) bits. This
768 // is not mentioned in sabb26, but it seems necessary to do this to
769 // ensure that the first (signed) slice does not overflow if the full
770 // bitwidth of splitint_t is used and the second slice is negative in
771 // two's complement.
772 const uint_t smallBitmask = (static_cast<uint_t>(1) << (bitsPerSlice - 1)) - 1;
773 const uint_t largeBitmask = (static_cast<uint_t>(1) << (bitsPerSlice + 1)) - 1;
774
775 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
776
777 auto matrixIndex = operand.operandIndex(outer, inner);
778
779 // NOTE: I could have a special path for 0.
780 int16_t shiftCounter;
781 int currentExponent = getStoredFloatingPointExponent(operand.operand(outer, inner));
782 int16_t exponentDifference = operand.scalingExponents[outer] - currentExponent;
783
784 splitint_t value = 0;
785 uint_t remainder = 0;
786
787 // Slice 0.
788 if (exponentDifference > (signed)(bitsPerSlice - 1)) {
789 // Value is too small to contribute explicit bits to slice 0.
790 value = sign[inner] ? -1 : 0;
791 remainder = sign[inner] ?
792 ((static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) - fraction[inner]) |
793 (static_cast<uint_t>(1) << (sizeof(uint_t) * 8 - 1)) :
794 fraction[inner];
795 exponentDifference -= (bitsPerSlice - 1);
796 shiftCounter = numSignificandBits - (bitsPerSlice + 1);
797 } else {
798 // Truncation.
799 shiftCounter = numSignificandBits - (bitsPerSlice - 1) + exponentDifference;
800 exponentDifference = 0;
801 uint_t bitmask = (shiftCounter > 0)
802 ? (smallBitmask << shiftCounter)
803 : (smallBitmask >> -shiftCounter);
804 uint_t currentSlice = fraction[inner] & bitmask;
805 uint_t currentSplit = (shiftCounter > 0)
806 ? (currentSlice >> shiftCounter)
807 : (currentSlice << -shiftCounter);
808
809 value = (sign[inner] ? -1 : 1) * static_cast<splitint_t>(currentSplit);
810
811 // Rounding.
812 uint_t lowMask = (static_cast<uint_t>(1) << shiftCounter) - 1;
813 uint_t lowBits = fraction[inner] & lowMask;
814
815 if (lowBits != 0) {
816 value += (sign[inner] ? -1 : 0);
817 }
818 remainder = sign[inner] ? (static_cast<uint_t>(1) << shiftCounter) - lowBits : lowBits;
819 shiftCounter -= (bitsPerSlice + 1);
820 }
821 operand.memory[matrixIndex] = value;
822
823 // Remaining slices.
824 const auto width = bitsPerSlice + 1;
825 const auto cutoff = static_cast<wideint_t>(static_cast<uint_t>(1) << bitsPerSlice); // 2^b
826 const auto base = static_cast<wideint_t>(static_cast<uint_t>(1) << width); // 2^(b+1)
827 const auto digitMax = cutoff - 1; // 2^b - 1
828 const auto digitMin = -cutoff; // -2^b
829 for (size_t slice = 1; slice < operand.prepConfig.numSplits; slice++) {
830 if (exponentDifference > (signed)(bitsPerSlice + 1)) {
831 exponentDifference -= (bitsPerSlice + 1);
832 if (sign[inner]) {
833 operand.splitValue(outer, inner, slice - 1) = static_cast<splitint_t>(0);
834 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(-1);
835 }
836 } else {
837 shiftCounter += exponentDifference;
838 exponentDifference = 0;
839
840 uint_t bitmask = shiftCounter > 0 ?
841 largeBitmask << shiftCounter :
842 largeBitmask >> -shiftCounter;
843
844 uint_t currentSlice = remainder & bitmask;
845 uint_t currentSplit = shiftCounter > 0 ?
846 currentSlice >> shiftCounter :
847 currentSlice << -shiftCounter;
848
849 if (sign[inner] && (shiftCounter + static_cast<int>(bitsPerSlice) + 1) > static_cast<int>(sizeof(uint_t)) * 8) {
850 int bitsUsed = sizeof(uint_t) * 8 - shiftCounter;
851 int bitsLost = (bitsPerSlice + 1) - bitsUsed;
852 uint_t maskToAdd = ((static_cast<uint_t>(1) << bitsLost) - 1) << bitsUsed;
853 currentSplit |= maskToAdd;
854 }
855
856 // Update previous slices if currentSplit is negative in splitint_t.
857 // Carry must propoagate upward until slice 0 if necessary.
858 if (currentSplit > static_cast<uint_t>(digitMax)) {
859 int prevSliceIndex = static_cast<int>(slice) - 1;
860 while (true) {
861 auto newValue = static_cast<wideint_t>(operand.splitValue(outer, inner, prevSliceIndex)) + 1;
862 if (newValue <= digitMax) {
863 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(newValue);
864 break;
865 } else {
866 operand.splitValue(outer, inner, prevSliceIndex) = static_cast<splitint_t>(digitMin);
867 prevSliceIndex--;
868 }
869 }
870 }
871
872 // Store the slice as a signed (b+1)-bit value:
873 // [0, 2^b-1] -> unchanged
874 // [2^b, 2^(b+1)-1] -> subtract 2^(b+1), yielding [-2^b, -1]
875 wideint_t signedDigit = static_cast<wideint_t>(currentSplit);
876 if (signedDigit >= cutoff) {
877 signedDigit -= base;
878 }
879
880 operand.splitValue(outer, inner, slice) = static_cast<splitint_t>(signedDigit);
881 shiftCounter -= (bitsPerSlice + 1);
882 }
883 }
884 }
885 }
886}

901template <typename splitint_t, typename fp_t>
902void computeSplitsWithRoundToNearest(preparedOperand<splitint_t, fp_t>& operand) {
903 auto bitsPerSlice = operand.prepConfig.bitsPerSlice;
904 auto localMatrix = std::vector<fp_t>(operand.matrix.data, operand.matrix.data + operand.matrix.size());
905 for (size_t slice = 0; slice < operand.prepConfig.numSplits; slice++) {
906 for (size_t outer = 0; outer < operand.outerDimension(); outer++) {
907 fp_t sigma = ldexp(0.75,
908 FloatingPointTraits<fp_t>::numSignificandBits - bitsPerSlice * slice + 1 - bitsPerSlice) *
909 operand.powersVector[outer];
910 for (size_t inner = 0; inner < operand.innerDimension(); inner++) {
911 auto matrixIndex = operand.operandIndex(outer, inner);
912 auto value = (localMatrix[matrixIndex] + sigma);
913 value -= sigma;
914 localMatrix[matrixIndex] -= value;
915 value = value / operand.powersVector[outer] * ldexp(1.0, bitsPerSlice * slice + bitsPerSlice - 1);
916 operand.splitValue(outer, inner, slice) = value;
917 }
918 }
919 }

927template <typename splitint_t, typename fp_t>
928preparedOperand<splitint_t, fp_t> prepareOperand(MatrixView<const fp_t> matrix,
929 const OperandPreparationConfig& prepConfig) {
930 preparedOperand<splitint_t, fp_t> operand;
931 operand.matrix = matrix;
932 operand.prepConfig = prepConfig;
933 operand.memory.resize(operand.matrix.size() * prepConfig.numSplits);
934 operand.powersVector.resize(operand.prepConfig.dimension == normalisationDimension::byRows ?
935 operand.rows() : operand.cols());
936 operand.scalingExponents.resize(operand.powersVector.size());
937
938 computeNormalisationVectors(operand);
939
940 switch (prepConfig.splitType) {
941 case splittingStrategy::truncation:
942 computeSplitsWithTruncation(operand);
943 break;
944 case splittingStrategy::unsignedEncoding:
945 computeSplitsWithUnsignedEncoding(operand);
946 break;
947 case splittingStrategy::roundToNearest:
948 computeSplitsWithRoundToNearest(operand);
949 break;
950 }
951
952 return operand;
953}

964 struct multiplicationSchedule {
965 size_t numSplitsA; // number of slices for A
966 size_t numSplitsB; // number of slices for B
967
968 std::vector<bool> mask;
969 inline std::vector<bool>::reference operator()(size_t i, size_t j) {
970 return mask[i * numSplitsB + j];
971 }
972 inline bool operator()(size_t i, size_t j) const {
973 return mask[i * numSplitsB + j];
974 }
975};
976
985inline multiplicationSchedule makeSchedule(const config& config) {
986 multiplicationSchedule schedule;
987 schedule.numSplitsA = config.numSplitsA;
988 schedule.numSplitsB = config.numSplitsB;
989 schedule.mask.resize(config.numSplitsA * config.numSplitsB, false);
990
991 std::visit([&](auto&& spec) {
992 using T = std::decay_t<decltype(spec)>;
993
994 if constexpr (std::is_same_v<T, multiplicationStrategy>) {
995 if (spec == multiplicationStrategy::full) {
996 std::fill(schedule.mask.begin(), schedule.mask.end(), true);
997 } else {
998 size_t limit = std::max(config.numSplitsA, config.numSplitsB) - 1;
999 for (size_t i = 0; i < config.numSplitsA; ++i)
1000 for (size_t j = 0; j < config.numSplitsB; ++j)
1001 schedule(i, j) = i + j <= limit;
1002 }
1003 } else if constexpr (std::is_same_v<T, std::vector<bool>>) {
1004
1005 const auto& mask = spec;
1006
1007 if (mask.size() != config.numSplitsA * config.numSplitsB)
1008 throw std::invalid_argument("Mask size mismatch.");
1009

1086 C[index] += scaledSum * scalingFactor;
1087 }
1088 }
1089 }
1090 Aindex--;
1091 Bindex++;
1092 }
